## 第221题：线程基础（简单）

**问题**：理解Python线程的基本概念和创建方式

**答案**：
```python
import threading
import time

# 方式1：使用Thread类
def worker(name, delay):
    """线程工作函数"""
    print(f"{name} 开始工作")
    time.sleep(delay)
    print(f"{name} 完成工作")

# 创建线程
thread1 = threading.Thread(target=worker, args=("线程1", 2))
thread2 = threading.Thread(target=worker, args=("线程2", 1))

# 启动线程
thread1.start()
thread2.start()

# 等待线程完成
thread1.join()
thread2.join()

print("所有线程完成")

# 方式2：继承Thread类
class WorkerThread(threading.Thread):
    def __init__(self, name, delay):
        super().__init__()
        self.name = name
        self.delay = delay

    def run(self):
        print(f"{self.name} 开始工作")
        time.sleep(self.delay)
        print(f"{self.name} 完成工作")

# 使用自定义线程类
t1 = WorkerThread("自定义线程1", 1)
t2 = WorkerThread("自定义线程2", 2)

t1.start()
t2.start()
t1.join()
t2.join()

# 线程属性
print(f"线程名称: {t1.name}")
print(f"线程ID: {t1.ident}")
print(f"线程是否存活: {t1.is_alive()}")
print(f"当前活跃线程数: {threading.active_count()}")
print(f"当前线程: {threading.current_thread().name}")
```

**面试要点**：
- Python线程是操作系统原生线程的封装
- 两种创建方式：Thread类和继承Thread
- start()启动线程，join()等待线程完成
- 守护线程（daemon）会在主线程结束时自动终止

## 第222题：全局解释器锁GIL（简单）

**问题**：什么是GIL？它对Python多线程有什么影响？

**答案**：
```python
import threading
import time

# GIL的影响演示
def cpu_bound_task(n):
    """CPU密集型任务"""
    count = 0
    for i in range(n):
        count += i * i
    return count

# 单线程执行
start = time.time()
result1 = cpu_bound_task(10000000)
result2 = cpu_bound_task(10000000)
print(f"单线程时间: {time.time() - start:.2f}秒")

# 多线程执行（因为GIL，不会更快）
start = time.time()
t1 = threading.Thread(target=cpu_bound_task, args=(10000000,))
t2 = threading.Thread(target=cpu_bound_task, args=(10000000,))
t1.start()
t2.start()
t1.join()
t2.join()
print(f"多线程时间: {time.time() - start:.2f}秒")

# IO密集型任务（多线程有效）
def io_bound_task():
    """IO密集型任务"""
    time.sleep(1)  # 模拟IO等待

# 单线程IO任务
start = time.time()
io_bound_task()
io_bound_task()
print(f"单线程IO时间: {time.time() - start:.2f}秒")

# 多线程IO任务（会更快）
start = time.time()
threads = [threading.Thread(target=io_bound_task) for _ in range(2)]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"多线程IO时间: {time.time() - start:.2f}秒")
```

**GIL关键点**：
- **GIL是什么**：全局解释器锁，确保同一时刻只有一个线程执行Python字节码
- **为什么存在**：简化CPython内存管理，避免多线程竞争
- **影响**：CPU密集型任务无法利用多核，但IO密集型任务仍然有效
- **解决方案**：
  - CPU密集型：使用multiprocessing（多进程）
  - IO密集型：使用threading或asyncio

## 第223题：线程同步 - Lock（简单）

**问题**：使用Lock解决线程安全问题

**答案**：
```python
import threading

# 不使用锁的问题
counter = 0

def increment_unsafe():
    global counter
    for _ in range(100000):
        counter += 1

threads = [threading.Thread(target=increment_unsafe) for _ in range(10)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"不使用锁的结果: {counter}")  # 结果不确定，通常小于1000000

# 使用Lock
counter = 0
lock = threading.Lock()

def increment_safe():
    global counter
    for _ in range(100000):
        lock.acquire()  # 获取锁
        try:
            counter += 1
        finally:
            lock.release()  # 释放锁

threads = [threading.Thread(target=increment_safe) for _ in range(10)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"使用锁的结果: {counter}")  # 1000000

# 使用with语句（推荐）
counter = 0

def increment_with_context():
    global counter
    for _ in range(100000):
        with lock:  # 自动获取和释放锁
            counter += 1

threads = [threading.Thread(target=increment_with_context) for _ in range(10)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"使用with的结果: {counter}")  # 1000000

# Lock的方法
lock = threading.Lock()
print(f"尝试获取锁: {lock.acquire()}")  # True
print(f"锁是否被持有: {lock.locked()}")  # True
print(f"非阻塞获取: {lock.acquire(blocking=False)}")  # False
lock.release()

# 超时获取
result = lock.acquire(timeout=1)  # 等待1秒
if result:
    try:
        # 临界区代码
        pass
    finally:
        lock.release()
```

**Lock要点**：
- Lock是最基本的同步原语
- acquire()获取锁，release()释放锁
- 使用with语句可以自动管理锁
- 可以设置超时和非阻塞模式

## 第224题：RLock递归锁（简单）

**问题**：RLock与Lock的区别及使用场景

**答案**：
```python
import threading

# Lock的限制：不能重复获取
lock = threading.Lock()

def problematic_recursive():
    lock.acquire()
    # 这里如果再次acquire会死锁
    # lock.acquire()  # 会阻塞
    lock.release()

# RLock：可重入锁
rlock = threading.RLock()

def safe_recursive(count):
    """递归函数中使用RLock"""
    with rlock:
        print(f"递归深度: {count}")
        if count > 0:
            safe_recursive(count - 1)

thread = threading.Thread(target=safe_recursive, args=(5,))
thread.start()
thread.join()

# RLock的典型应用：嵌套调用
class BankAccount:
    def __init__(self):
        self.balance = 0
        self.lock = threading.RLock()

    def deposit(self, amount):
        with self.lock:
            self.balance += amount
            print(f"存款 {amount}，余额 {self.balance}")

    def withdraw(self, amount):
        with self.lock:
            if self.balance >= amount:
                self.balance -= amount
                print(f"取款 {amount}，余额 {self.balance}")
                return True
            return False

    def transfer_to(self, other, amount):
        """转账需要同时获取两个账户的锁"""
        with self.lock:  # 第一次获取锁
            if self.withdraw(amount):  # withdraw内部会再次获取锁
                other.deposit(amount)
                return True
            return False

# 使用示例
account1 = BankAccount()
account2 = BankAccount()
account1.deposit(1000)
account1.transfer_to(account2, 300)

# RLock的计数机制
rlock = threading.RLock()
rlock.acquire()
rlock.acquire()  # 可以多次获取
print(f"需要释放的次数: 2")
rlock.release()
rlock.release()  # 必须释放相同次数
```

**RLock要点**：
- RLock（Reentrant Lock）可以被同一线程多次获取
- 记录获取次数，必须释放相同次数才能完全释放
- 适用于递归调用和嵌套锁场景
- 比Lock稍慢，但更灵活

## 第225题：Condition条件变量（简单）

**问题**：使用Condition实现生产者-消费者模式

**答案**：
```python
import threading
import time
import random

# 生产者-消费者模式
class Queue:
    def __init__(self, max_size=5):
        self.queue = []
        self.max_size = max_size
        self.condition = threading.Condition()

    def produce(self, item):
        with self.condition:
            # 等待队列有空间
            while len(self.queue) >= self.max_size:
                print(f"队列满，生产者等待...")
                self.condition.wait()

            self.queue.append(item)
            print(f"生产: {item}, 队列大小: {len(self.queue)}")

            # 通知消费者
            self.condition.notify()

    def consume(self):
        with self.condition:
            # 等待队列有数据
            while len(self.queue) == 0:
                print(f"队列空，消费者等待...")
                self.condition.wait()

            item = self.queue.pop(0)
            print(f"消费: {item}, 队列大小: {len(self.queue)}")

            # 通知生产者
            self.condition.notify()
            return item

# 生产者
def producer(queue, count):
    for i in range(count):
        time.sleep(random.random())
        queue.produce(f"商品{i}")

# 消费者
def consumer(queue, count):
    for _ in range(count):
        time.sleep(random.random() * 2)
        queue.consume()

# 测试
q = Queue(max_size=3)
p_thread = threading.Thread(target=producer, args=(q, 10))
c_thread = threading.Thread(target=consumer, args=(q, 10))

p_thread.start()
c_thread.start()
p_thread.join()
c_thread.join()

# Condition的高级用法
condition = threading.Condition()
ready = False

def waiter():
    with condition:
        while not ready:
            print("等待条件满足...")
            condition.wait()
        print("条件满足，继续执行")

def notifier():
    global ready
    time.sleep(2)
    with condition:
        ready = True
        print("条件已满足，通知等待者")
        condition.notify_all()  # 通知所有等待的线程

t1 = threading.Thread(target=waiter)
t2 = threading.Thread(target=waiter)
t3 = threading.Thread(target=notifier)

t1.start()
t2.start()
t3.start()

t1.join()
t2.join()
t3.join()
```

**Condition要点**：
- Condition结合了Lock和通知机制
- wait()释放锁并等待通知
- notify()唤醒一个等待的线程
- notify_all()唤醒所有等待的线程
- 适用于生产者-消费者等复杂同步场景

## 第226题：Semaphore信号量（简单）

**问题**：使用Semaphore限制并发访问数量

**答案**：
```python
import threading
import time
import random

# Semaphore基础
semaphore = threading.Semaphore(3)  # 允许3个线程同时访问

def limited_access(worker_id):
    print(f"工作者 {worker_id} 等待访问...")
    semaphore.acquire()
    try:
        print(f"工作者 {worker_id} 获得访问权限")
        time.sleep(random.randint(1, 3))
        print(f"工作者 {worker_id} 完成工作")
    finally:
        semaphore.release()
        print(f"工作者 {worker_id} 释放访问权限")

# 创建10个线程，但只允许3个同时执行
threads = [threading.Thread(target=limited_access, args=(i,)) for i in range(10)]
for t in threads:
    t.start()
for t in threads:
    t.join()

# 实际应用：数据库连接池
class ConnectionPool:
    def __init__(self, max_connections=5):
        self.semaphore = threading.Semaphore(max_connections)
        self.connections = []

    def get_connection(self):
        self.semaphore.acquire()
        print(f"获取数据库连接 (线程: {threading.current_thread().name})")
        return f"Connection-{threading.current_thread().ident}"

    def release_connection(self, conn):
        print(f"释放数据库连接: {conn}")
        self.semaphore.release()

def database_operation(pool, query_id):
    conn = pool.get_connection()
    try:
        print(f"执行查询 {query_id}")
        time.sleep(random.uniform(0.5, 2))
        print(f"查询 {query_id} 完成")
    finally:
        pool.release_connection(conn)

# 测试连接池
pool = ConnectionPool(max_connections=3)
threads = [threading.Thread(target=database_operation, args=(pool, i))
           for i in range(10)]

for t in threads:
    t.start()
for t in threads:
    t.join()

# BoundedSemaphore：防止release超过初始值
bounded_sem = threading.BoundedSemaphore(2)
bounded_sem.acquire()
bounded_sem.release()
# bounded_sem.release()  # 会抛出ValueError

# with语句使用
with semaphore:
    # 临界区代码
    print("在信号量保护下执行")
```

**Semaphore要点**：
- Semaphore允许多个线程同时访问资源
- 内部维护计数器，acquire()减1，release()加1
- BoundedSemaphore防止过度release
- 适用于限制并发数量（如连接池、限流）

## 第227题：Event事件（简单）

**问题**：使用Event进行线程间通信

**答案**：
```python
import threading
import time

# Event基础
event = threading.Event()

def waiter(name):
    print(f"{name} 等待事件...")
    event.wait()  # 阻塞直到事件被设置
    print(f"{name} 收到事件，继续执行")

def setter():
    time.sleep(2)
    print("设置事件...")
    event.set()  # 设置事件，唤醒所有等待的线程

# 创建多个等待线程
threads = [threading.Thread(target=waiter, args=(f"线程{i}",)) for i in range(3)]
setter_thread = threading.Thread(target=setter)

for t in threads:
    t.start()
setter_thread.start()

for t in threads:
    t.join()
setter_thread.join()

# Event的方法
event = threading.Event()
print(f"事件是否设置: {event.is_set()}")  # False
event.set()  # 设置事件
print(f"事件是否设置: {event.is_set()}")  # True
event.clear()  # 清除事件
print(f"事件是否设置: {event.is_set()}")  # False

# 带超时的等待
event = threading.Event()

def timed_waiter():
    print("等待事件（5秒超时）...")
    result = event.wait(timeout=5)
    if result:
        print("事件已设置")
    else:
        print("等待超时")

thread = threading.Thread(target=timed_waiter)
thread.start()
thread.join()

# 实际应用：服务器启动信号
class Server:
    def __init__(self):
        self.ready = threading.Event()

    def start(self):
        print("服务器正在启动...")
        time.sleep(2)  # 模拟启动过程
        print("服务器启动完成")
        self.ready.set()  # 通知服务器已就绪

    def wait_until_ready(self):
        print("等待服务器就绪...")
        self.ready.wait()
        print("服务器已就绪，可以处理请求")

server = Server()
server_thread = threading.Thread(target=server.start)
client_thread = threading.Thread(target=server.wait_until_ready)

client_thread.start()
server_thread.start()

server_thread.join()
client_thread.join()
```

**Event要点**：
- Event用于线程间的简单信号通知
- set()设置事件，所有wait()的线程被唤醒
- clear()清除事件
- wait()阻塞直到事件被设置，支持超时
- 适用于启动信号、停止信号等场景

## 第228题：Queue线程安全队列（简单）

**问题**：使用Queue实现线程安全的数据传递

**答案**：
```python
import threading
import queue
import time
import random

# Queue基础
q = queue.Queue(maxsize=5)  # 最大容量5

def producer(q, items):
    for item in items:
        print(f"生产: {item}")
        q.put(item)  # 如果队列满，会阻塞
        time.sleep(random.random())
    print("生产者完成")

def consumer(q, count):
    for _ in range(count):
        item = q.get()  # 如果队列空，会阻塞
        print(f"消费: {item}")
        q.task_done()  # 标记任务完成
        time.sleep(random.random())
    print("消费者完成")

# 测试
items = [f"任务{i}" for i in range(10)]
p = threading.Thread(target=producer, args=(q, items))
c = threading.Thread(target=consumer, args=(q, 10))

p.start()
c.start()
p.join()
q.join()  # 等待所有任务完成
c.join()

# Queue的方法
q = queue.Queue()

# put方法
q.put("item1")  # 阻塞直到有空间
q.put("item2", block=False)  # 非阻塞，满了抛出queue.Full
q.put("item3", timeout=1)  # 超时1秒

# get方法
item = q.get()  # 阻塞直到有数据
item = q.get(block=False)  # 非阻塞，空了抛出queue.Empty
item = q.get(timeout=1)  # 超时1秒

# 队列状态
print(f"队列大小: {q.qsize()}")
print(f"队列是否为空: {q.empty()}")
print(f"队列是否已满: {q.full()}")

# LifoQueue：后进先出
lifo_q = queue.LifoQueue()
lifo_q.put(1)
lifo_q.put(2)
lifo_q.put(3)
print(lifo_q.get())  # 3
print(lifo_q.get())  # 2

# PriorityQueue：优先级队列
pq = queue.PriorityQueue()
pq.put((2, "低优先级"))
pq.put((1, "高优先级"))
pq.put((3, "最低优先级"))
print(pq.get())  # (1, "高优先级")
print(pq.get())  # (2, "低优先级")

# 实际应用：任务队列
class TaskQueue:
    def __init__(self, num_workers=3):
        self.queue = queue.Queue()
        self.workers = []
        for i in range(num_workers):
            worker = threading.Thread(target=self._worker, args=(i,))
            worker.daemon = True
            worker.start()
            self.workers.append(worker)

    def _worker(self, worker_id):
        while True:
            task = self.queue.get()
            if task is None:  # 停止信号
                break
            print(f"工作者 {worker_id} 执行任务: {task}")
            time.sleep(random.random())
            self.queue.task_done()

    def add_task(self, task):
        self.queue.put(task)

    def wait_completion(self):
        self.queue.join()

    def stop(self):
        for _ in self.workers:
            self.queue.put(None)
        for worker in self.workers:
            worker.join()

# 使用任务队列
task_queue = TaskQueue(num_workers=3)
for i in range(10):
    task_queue.add_task(f"任务{i}")

task_queue.wait_completion()
task_queue.stop()
print("所有任务完成")
```

**Queue要点**：
- Queue是线程安全的队列实现
- put()添加数据，get()获取数据
- task_done()标记任务完成，join()等待所有任务
- LifoQueue（栈）、PriorityQueue（优先级队列）
- 适用于生产者-消费者模式

## 第229题：进程基础（中等）

**问题**：理解multiprocessing模块和进程创建

**答案**：
```python
import multiprocessing
import os
import time

# 进程基础
def worker(name, delay):
    """进程工作函数"""
    print(f"{name} 启动 (PID: {os.getpid()}, 父进程: {os.getppid()})")
    time.sleep(delay)
    print(f"{name} 完成")

if __name__ == '__main__':
    print(f"主进程 PID: {os.getpid()}")

    # 创建进程
    p1 = multiprocessing.Process(target=worker, args=("进程1", 2))
    p2 = multiprocessing.Process(target=worker, args=("进程2", 1))

    # 启动进程
    p1.start()
    p2.start()

    # 等待进程完成
    p1.join()
    p2.join()

    print("所有进程完成")

# 进程类继承
class WorkerProcess(multiprocessing.Process):
    def __init__(self, name, delay):
        super().__init__()
        self.worker_name = name
        self.delay = delay

    def run(self):
        print(f"{self.worker_name} 开始 (PID: {os.getpid()})")
        time.sleep(self.delay)
        print(f"{self.worker_name} 结束")

if __name__ == '__main__':
    p1 = WorkerProcess("自定义进程1", 1)
    p2 = WorkerProcess("自定义进程2", 2)

    p1.start()
    p2.start()
    p1.join()
    p2.join()

# 进程属性和方法
if __name__ == '__main__':
    p = multiprocessing.Process(target=worker, args=("测试进程", 1))

    print(f"进程名称: {p.name}")
    print(f"进程是否存活: {p.is_alive()}")

    p.start()
    print(f"进程PID: {p.pid}")
    print(f"进程是否存活: {p.is_alive()}")

    p.join()
    print(f"进程退出码: {p.exitcode}")

# 守护进程
def daemon_worker():
    print("守护进程启动")
    time.sleep(5)
    print("守护进程结束")  # 不会打印

if __name__ == '__main__':
    p = multiprocessing.Process(target=daemon_worker)
    p.daemon = True  # 设置为守护进程
    p.start()
    time.sleep(1)
    print("主进程结束")
    # 守护进程会被强制终止

# CPU密集型任务对比
```

```python
    messages = ['Hello', 'World', 'Python']

        print(f"发送: {msg}")

def receiver(conn):
    """接收端"""
    while True:
        msg = conn.recv()
        if msg is None:
            break
        print(f"收到: {msg}")

    conn.close()

    parent_conn, child_conn = multiprocessing.Pipe()

    p1 = multiprocessing.Process(target=sender, args=(parent_conn,))
    p2 = multiprocessing.Process(target=receiver, args=(child_conn,))

    p2.start()
    p1.join()
    p2.join()

Pipe:
```

```python
class ReadWriteLock:
    def __init__(self):

        self._writers = 0
        self._read_ready = threading.Condition(threading.Lock())
        self._write_ready = threading.Condition(threading.Lock())

        """获取读锁"""
        self._read_ready.acquire()
        while self._writers > 0:
            self._read_ready.wait()
        self._readers += 1
        self._read_ready.release()

    def release_read(self):
        """释放读锁"""
        self._read_ready.acquire()
        self._readers -= 1
        if self._readers == 0:
            self._write_ready.notify()
        self._read_ready.release()

    def acquire_write(self):
        """获取写锁"""
        self._write_ready.acquire()
        while self._readers > 0 or self._writers > 0:
            self._write_ready.wait()
        self._writers = 1

    def release_write(self):
        """释放写锁"""
        self._writers = 0
        self._write_ready.notify()
        self._read_ready.notifyAll()
        self._write_ready.release()
```

**答案**：
```python
import threading
import time
from queue import PriorityQueue, Queue
from datetime import datetime, timedelta
from typing import Callable, Any

# 配置日志

    level=logging.INFO,
    format='%(asctime)s - %(threadName)s - %(message)s'
)

# 1. 任务定义
class Task:
    """任务类"""
    task_id_counter = 0
    task_id_lock = threading.Lock()

    def __init__(self, func: Callable, args: tuple = (), kwargs: dict = None,
                 priority: int = 0, delay: float = 0, retry: int = 0):
        with Task.task_id_lock:
            self.task_id = Task.task_id_counter
            Task.task_id_counter += 1

        self.func = func
        self.args = args
        self.kwargs = kwargs or {}
        self.priority = priority
        self.scheduled_time = time.time() + delay
        self.retry_count = retry
        self.max_retries = retry
        self.status = 'pending'
        self.result = None
        self.error = None

    def __lt__(self, other):
        """优先级比较（用于优先级队列）"""
        if self.scheduled_time != other.scheduled_time:
            return self.scheduled_time < other.scheduled_time
        return self.priority < other.priority

    def execute(self):
        """执行任务"""
        try:
            logging.info(f"执行任务 {self.task_id}: {self.func.__name__}")
            self.status = 'running'
            self.result = self.func(*self.args, **self.kwargs)
            self.status = 'completed'
            logging.info(f"任务 {self.task_id} 完成")
            return True
        except Exception as e:
            self.error = str(e)
            self.status = 'failed'
            logging.error(f"任务 {self.task_id} 失败: {e}")
            return False

# 2. 线程安全的任务调度器
class TaskScheduler:
    """线程安全的任务调度器"""

    def __init__(self, num_workers: int = 3):
        self.num_workers = num_workers
        self.task_queue = PriorityQueue()
        self.workers = []
        self.running = False

        # 任务跟踪
        self.tasks = {}
        self.tasks_lock = threading.RLock()

        # 统计信息
        self.stats = {
            'submitted': 0,
            'completed': 0,
            'failed': 0,
            'retried': 0
        }
        self.stats_lock = threading.Lock()

    def start(self):
        """启动调度器"""
        if self.running:
            return

        self.running = True
        logging.info(f"启动调度器，工作线程数: {self.num_workers}")

        # 创建工作线程
        for i in range(self.num_workers):
            worker = threading.Thread(
                target=self._worker,
                name=f"Worker-{i}",
                daemon=True
            )
            self.workers.append(worker)
            worker.start()

    def stop(self):
        """停止调度器"""
        logging.info("停止调度器...")
        self.running = False

        # 发送停止信号
        for _ in range(self.num_workers):
            self.task_queue.put((float('inf'), None))

        # 等待所有工作线程结束
        for worker in self.workers:
            worker.join()

        logging.info("调度器已停止")

    def submit(self, func: Callable, args: tuple = (), kwargs: dict = None,
               priority: int = 0, delay: float = 0, retry: int = 0) -> int:
        """提交任务"""
        task = Task(func, args, kwargs, priority, delay, retry)

        with self.tasks_lock:
            self.tasks[task.task_id] = task

        self.task_queue.put((task.scheduled_time, task))

        with self.stats_lock:
            self.stats['submitted'] += 1

        logging.info(f"提交任务 {task.task_id}, 优先级: {priority}, 延迟: {delay}s")
        return task.task_id

    def _worker(self):
        """工作线程"""
        while self.running:
            try:
                # 获取任务（带超时避免阻塞）
                scheduled_time, task = self.task_queue.get(timeout=1)

                if task is None:  # 停止信号
                    break

                # 等待到调度时间
                wait_time = scheduled_time - time.time()
                if wait_time > 0:
                    time.sleep(wait_time)

                # 执行任务
                success = task.execute()

                # 更新统计
                with self.stats_lock:
                    if success:
                        self.stats['completed'] += 1
                    else:
                        self.stats['failed'] += 1

                        # 重试逻辑
                        if task.retry_count > 0:
                            task.retry_count -= 1
                            task.status = 'pending'
                            task.scheduled_time = time.time() + 1  # 1秒后重试
                            self.task_queue.put((task.scheduled_time, task))
                            self.stats['retried'] += 1
                            logging.info(f"任务 {task.task_id} 将重试，剩余次数: {task.retry_count}")

            except Exception as e:
                if self.running:
                    logging.error(f"工作线程错误: {e}")

    def get_task_status(self, task_id: int) -> dict:
        """获取任务状态"""
        with self.tasks_lock:
            task = self.tasks.get(task_id)
            if task:
                return {
                    'task_id': task.task_id,
                    'status': task.status,
                    'result': task.result,
                    'error': task.error
                }
        return None

    def get_stats(self) -> dict:
        """获取统计信息"""
        with self.stats_lock:
            return self.stats.copy()

    def wait_completion(self, timeout: float = None):
        """等待所有任务完成"""
        start_time = time.time()
        while True:
            if timeout and (time.time() - start_time) > timeout:
                logging.warning("等待超时")
                break

            if self.task_queue.empty():
                time.sleep(0.1)
                if self.task_queue.empty():  # 再次检查
                    break

            time.sleep(0.5)

class AdvancedScheduler(TaskScheduler):
    """高级调度器（添加更多功能）"""

    def __init__(self, num_workers: int = 3):
        super().__init__(num_workers)
        self.periodic_tasks = {}
        self.periodic_lock = threading.Lock()

    def submit_periodic(self, func: Callable, interval: float,
                       args: tuple = (), kwargs: dict = None) -> str:
        """提交周期性任务"""
        task_id = f"periodic_{len(self.periodic_tasks)}"

        def periodic_wrapper():
            while self.running:
                try:
                    func(*args, **(kwargs or {}))
                except Exception as e:
                    logging.error(f"周期任务错误: {e}")
                time.sleep(interval)

        with self.periodic_lock:
            thread = threading.Thread(
                target=periodic_wrapper,
                name=task_id,
                daemon=True
            )
            self.periodic_tasks[task_id] = thread
            thread.start()

        logging.info(f"启动周期任务 {task_id}, 间隔: {interval}秒")
        return task_id

    def cancel_periodic(self, task_id: str):
        """取消周期性任务"""
        with self.periodic_lock:
            if task_id in self.periodic_tasks:
                # 注意：Thread没有直接停止方法，需要配合标志位
                # 这里简化处理
                del self.periodic_tasks[task_id]
                logging.info(f"取消周期任务 {task_id}")

# 5. 性能监控
class MonitoredScheduler(AdvancedScheduler):
    """带监控的调度器"""

    def __init__(self, num_workers: int = 3):
        super().__init__(num_workers)
        self.task_times = []
        self.times_lock = threading.Lock()

    def _worker(self):
        """重写worker添加监控"""
        while self.running:
            try:
                scheduled_time, task = self.task_queue.get(timeout=1)

                if task is None:
                    break

                wait_time = scheduled_time - time.time()
                if wait_time > 0:
                    time.sleep(wait_time)

                # 记录执行时间
                start_time = time.time()
                success = task.execute()
                execution_time = time.time() - start_time

                with self.times_lock:
                    self.task_times.append(execution_time)

                with self.stats_lock:
                    if success:
                        self.stats['completed'] += 1
                    else:
                        self.stats['failed'] += 1
                        if task.retry_count > 0:
                            task.retry_count -= 1
                            task.status = 'pending'
                            task.scheduled_time = time.time() + 1
                            self.task_queue.put((task.scheduled_time, task))
                            self.stats['retried'] += 1

            except Exception as e:
                if self.running:
                    logging.error(f"工作线程错误: {e}")

    def get_performance_stats(self) -> dict:
        """获取性能统计"""
        with self.times_lock:
            if not self.task_times:
                return {}

            avg_time = sum(self.task_times) / len(self.task_times)
            max_time = max(self.task_times)
            min_time = min(self.task_times)

            return {
                'average_execution_time': avg_time,
                'max_execution_time': max_time,
                'min_execution_time': min_time,
                'total_tasks': len(self.task_times)
            }

    scheduler.stop()
```
