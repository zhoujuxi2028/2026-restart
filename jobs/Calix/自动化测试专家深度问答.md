```python
import requests
import logging
from typing import Optional, Dict, Any
from urllib3.util.retry import Retry
from requests.adapters import HTTPAdapter

class EnterpriseAPIClient:
    def __init__(self, base_url: str, config: dict):
        self.base_url = base_url
        self.session = requests.Session()
        self.config = config
        self.logger = logging.getLogger(__name__)

        # 配置重试策略
        retry_strategy = Retry(
            total=3,
            status_forcelist=[429, 500, 502, 503, 504],
            backoff_factor=1,
            raise_on_status=False
        )

        adapter = HTTPAdapter(max_retries=retry_strategy)
        self.session.mount("http://", adapter)
        self.session.mount("https://", adapter)

        # 设置默认headers
        self.session.headers.update({
            'Content-Type': 'application/json',
            'Accept': 'application/json',
            'User-Agent': 'Enterprise-Test-Framework/1.0'
        })

    def authenticate(self, username: str, password: str) -> bool:
        """统一认证方法"""
        auth_payload = {
            'username': username,
            'password': password
        }

        try:
            response = self.session.post(
                f"{self.base_url}/auth/login",
                json=auth_payload,
                timeout=self.config.get('timeout', 30)
            )

            if response.status_code == 200:
                token = response.json().get('access_token')
                self.session.headers.update({
                    'Authorization': f'Bearer {token}'
                })
                self.logger.info("Authentication successful")
                return True
            else:
                self.logger.error(f"Authentication failed: {response.status_code}")
                return False

        except requests.exceptions.RequestException as e:
            self.logger.error(f"Authentication error: {str(e)}")
            return False

    def request(self, method: str, endpoint: str, **kwargs) -> requests.Response:
        """统一请求方法，包含完整的日志和错误处理"""
        url = f"{self.base_url.rstrip('/')}/{endpoint.lstrip('/')}"

        # 请求前处理
        self._log_request(method, url, kwargs)

        try:
            # 发送请求
            response = self.session.request(method, url, **kwargs)

            # 响应后处理
            self._log_response(response)
            self._validate_response(response)

            return response

        except requests.exceptions.Timeout:
            self.logger.error(f"Request timeout for {method} {url}")
            raise
        except requests.exceptions.ConnectionError:
            self.logger.error(f"Connection error for {method} {url}")
            raise
        except requests.exceptions.RequestException as e:
            self.logger.error(f"Request error: {str(e)}")
            raise

    def _log_request(self, method: str, url: str, kwargs: dict):
        """详细的请求日志"""
        self.logger.info(f"=== REQUEST ===")
        self.logger.info(f"{method.upper()} {url}")

        if kwargs.get('headers'):
            self.logger.info(f"Headers: {kwargs['headers']}")

        if kwargs.get('json'):
            self.logger.info(f"JSON Body: {kwargs['json']}")
        elif kwargs.get('data'):
            self.logger.info(f"Data: {kwargs['data']}")

    def _log_response(self, response: requests.Response):
        """详细的响应日志"""
        self.logger.info(f"=== RESPONSE ===")
        self.logger.info(f"Status Code: {response.status_code}")
        self.logger.info(f"Headers: {dict(response.headers)}")

        try:
            self.logger.info(f"JSON Body: {response.json()}")
        except ValueError:
            self.logger.info(f"Text Body: {response.text[:500]}...")

    def _validate_response(self, response: requests.Response):
        """响应验证"""
        # 检查状态码
        if response.status_code >= 500:
            raise requests.exceptions.HTTPError(
                f"Server error: {response.status_code}"
            )

        # 检查响应时间
        if response.elapsed.total_seconds() > self.config.get('max_response_time', 10):
            self.logger.warning(
                f"Slow response: {response.elapsed.total_seconds()}s"
            )

    # 便捷方法
    def get(self, endpoint: str, **kwargs) -> requests.Response:
        return self.request('GET', endpoint, **kwargs)

    def post(self, endpoint: str, **kwargs) -> requests.Response:
        return self.request('POST', endpoint, **kwargs)

    def put(self, endpoint: str, **kwargs) -> requests.Response:
        return self.request('PUT', endpoint, **kwargs)

    def delete(self, endpoint: str, **kwargs) -> requests.Response:
        return self.request('DELETE', endpoint, **kwargs)
```

```python
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By
from selenium.common.exceptions import TimeoutException
import allure

class BasePage:
    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)

    def find_element(self, locator, timeout=None):
        """智能元素查找"""
        wait_time = timeout or 10
        try:
            element = WebDriverWait(self.driver, wait_time).until(
                EC.presence_of_element_located(locator)
            )
            self._highlight_element(element)
            return element
        except TimeoutException:
            self._capture_failure_evidence(locator)
            raise AssertionError(f"Element {locator} not found within {wait_time} seconds")

    def find_elements(self, locator, timeout=None):
        """查找多个元素"""
        wait_time = timeout or 10
        try:
            return WebDriverWait(self.driver, wait_time).until(
                EC.presence_of_all_elements_located(locator)
            )
        except TimeoutException:
            return []

    def click(self, locator, timeout=None):
        """智能点击"""
        element = self.wait_for_clickable(locator, timeout)
        try:
            # 滚动到元素位置
            self.driver.execute_script("arguments[0].scrollIntoView(true);", element)

            # 等待一下确保滚动完成
            self.driver.implicitly_wait(0.5)

            # 尝试普通点击
            element.click()

        except Exception as e:
            # 如果普通点击失败，使用JavaScript点击
            self.driver.execute_script("arguments[0].click();", element)
            allure.attach(f"Used JavaScript click due to: {str(e)}",
                         name="Click Method",
                         attachment_type=allure.attachment_type.TEXT)

    def enter_text(self, locator, text, clear=True, timeout=None):
        """输入文本"""
        element = self.find_element(locator, timeout)

        if clear:
            element.clear()

        element.send_keys(text)

        # 验证输入
        actual_value = element.get_attribute('value')
        if actual_value != text:
            allure.attach(f"Expected: {text}, Actual: {actual_value}",
                         name="Text Input Verification",
                         attachment_type=allure.attachment_type.TEXT)

    def wait_for_clickable(self, locator, timeout=None):
        """等待元素可点击"""
        wait_time = timeout or 10
        try:
            return WebDriverWait(self.driver, wait_time).until(
                EC.element_to_be_clickable(locator)
            )
        except TimeoutException:
            self._capture_failure_evidence(locator)
            raise AssertionError(f"Element {locator} not clickable within {wait_time} seconds")

    def wait_for_page_load(self, timeout=30):
        """等待页面完全加载"""
        WebDriverWait(self.driver, timeout).until(
            lambda driver: driver.execute_script("return document.readyState") == "complete"
        )

    def wait_for_ajax_complete(self, timeout=30):
        """等待AJAX请求完成"""
        WebDriverWait(self.driver, timeout).until(
            lambda driver: driver.execute_script("return window.jQuery && jQuery.active === 0")
        )

    def _highlight_element(self, element):
        """高亮显示元素（用于调试）"""
        if self.driver.name != 'htmlunit':  # 只在真实浏览器中高亮
            self.driver.execute_script(
                "arguments[0].style.border='3px solid red'", element
            )

    def _capture_failure_evidence(self, locator):
        """捕获失败证据"""
        # 截图
        screenshot = self.driver.get_screenshot_as_png()
        allure.attach(screenshot,
                     name=f"Failure_Screenshot_{locator}",
                     attachment_type=allure.attachment_type.PNG)

        # 页面源码
        page_source = self.driver.page_source
        allure.attach(page_source,
                     name="Page_Source",
                     attachment_type=allure.attachment_type.HTML)

class ComponentBase:
```

#### 2. 智能等待策略
```python
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.support.ui import WebDriverWait
from selenium.common.exceptions import TimeoutException

class SmartWaits:
    def __init__(self, driver, default_timeout=10):
        self.driver = driver
        self.default_timeout = default_timeout

    def wait_for_element_visible_and_clickable(self, locator, timeout=None):
        """等待元素既可见又可点击"""
        wait_time = timeout or self.default_timeout

        try:
            # 首先等待可见
            WebDriverWait(self.driver, wait_time).until(
                EC.visibility_of_element_located(locator)
            )

            # 然后等待可点击
            element = WebDriverWait(self.driver, wait_time).until(
                EC.element_to_be_clickable(locator)
            )

            return element

        except TimeoutException:
            raise AssertionError(
                f"Element {locator} not visible and clickable within {wait_time} seconds"
            )

    def wait_for_text_to_be_present(self, locator, text, timeout=None):
        """等待元素包含特定文本"""
        wait_time = timeout or self.default_timeout

        try:
            WebDriverWait(self.driver, wait_time).until(
                EC.text_to_be_present_in_element(locator, text)
            )
            return True
        except TimeoutException:
            current_text = self.driver.find_element(*locator).text
            raise AssertionError(
                f"Expected text '{text}' not found in element {locator}. "
                f"Current text: '{current_text}'"
            )

    def wait_for_url_change(self, current_url, timeout=None):
        """等待URL变化"""
        wait_time = timeout or self.default_timeout

        try:
            WebDriverWait(self.driver, wait_time).until(
                lambda driver: driver.current_url != current_url
            )
            return True
        except TimeoutException:
            raise AssertionError(
                f"URL did not change from {current_url} within {wait_time} seconds"
            )

    def wait_for_element_staleness(self, element, timeout=None):
        """等待元素变为stale（页面刷新后）"""
        wait_time = timeout or self.default_timeout

        try:
            WebDriverWait(self.driver, wait_time).until(
                EC.staleness_of(element)
            )
            return True
        except TimeoutException:
            raise AssertionError(
                f"Element did not become stale within {wait_time} seconds"
            )

    def wait_for_custom_condition(self, condition_func, timeout=None, poll_frequency=0.5):
        """等待自定义条件"""
        wait_time = timeout or self.default_timeout

        try:
            WebDriverWait(self.driver, wait_time, poll_frequency).until(condition_func)
            return True
        except TimeoutException:
            raise AssertionError(
                f"Custom condition not met within {wait_time} seconds"
            )

class CustomExpectedConditions:
    @staticmethod
    def number_of_windows_to_be(num):
        """等待窗口数量达到指定数量"""
        def _predicate(driver):
            return len(driver.window_handles) == num
        return _predicate

    @staticmethod
    def element_attribute_to_include(locator, attribute, value):
        """等待元素属性包含特定值"""
        def _predicate(driver):
            element = driver.find_element(*locator)
            attr_value = element.get_attribute(attribute)
            return value in attr_value if attr_value else False
        return _predicate

    @staticmethod
    def ajax_requests_completed():
        """等待所有AJAX请求完成"""
        def _predicate(driver):
            jquery_active = driver.execute_script("return window.jQuery && jQuery.active === 0")
            angular_pending = driver.execute_script(
                "return window.angular && angular.element(document).injector().get('$http').pendingRequests.length === 0"
            )
            return jquery_active and angular_pending
        return _predicate
```

#### 1. 测试金字塔实现
```yaml
# .github/workflows/test-pipeline.yml
name: Comprehensive Test Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  unit-tests:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: [3.9, 3.10, 3.11]

    steps:
    - uses: actions/checkout@v4

    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: ${{ matrix.python-version }}

    - name: Install dependencies
      run: |
        pip install -r requirements.txt
        pip install -r requirements-test.txt

    - name: Run unit tests
      run: |
        pytest tests/unit/ \
          --junitxml=junit/unit-test-results.xml \
          --cov=src \
          --cov-report=xml \
          --cov-report=html

    - name: Upload test results
      uses: actions/upload-artifact@v3
      if: always()
      with:
        name: unit-test-results-${{ matrix.python-version }}
        path: |
          junit/unit-test-results.xml
          htmlcov/

  integration-tests:
    needs: unit-tests
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:13
        env:
          POSTGRES_PASSWORD: testpass
          POSTGRES_DB: testdb
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

      redis:
        image: redis:6
        options: >-
          --health-cmd "redis-cli ping"
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
    - uses: actions/checkout@v4

    - name: Set up test environment
      run: |
        docker-compose -f docker-compose.test.yml up -d
        sleep 30  # 等待服务启动

    - name: Run integration tests
      env:
        DATABASE_URL: postgresql://postgres:testpass@localhost:5432/testdb
        REDIS_URL: redis://localhost:6379
      run: |
        pytest tests/integration/ \
          --junitxml=junit/integration-test-results.xml \
          --alluredir=allure-results

    - name: Generate Allure report
      uses: simple-elf/allure-report-action@master
      if: always()
      with:
        allure_results: allure-results
        allure_history: allure-history

  api-tests:
    needs: integration-tests
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Start application
      run: |
        docker-compose up -d
        ./scripts/wait-for-service.sh http://localhost:8080/health

    - name: Run API tests
      run: |
        pytest tests/api/ \
          --junitxml=junit/api-test-results.xml \
          --html=reports/api-report.html \
          --self-contained-html

    - name: Run API performance tests
      run: |
        pytest tests/performance/api/ \
          --benchmark-json=benchmark-results.json

  ui-tests:
    needs: api-tests
    runs-on: ubuntu-latest

    strategy:
      matrix:
        browser: [chrome, firefox]

    steps:
    - uses: actions/checkout@v4

    - name: Set up browser
      if: matrix.browser == 'chrome'
      uses: browser-actions/setup-chrome@latest

    - name: Set up browser
      if: matrix.browser == 'firefox'
      uses: browser-actions/setup-firefox@latest

    - name: Start application
      run: |
        docker-compose up -d
        ./scripts/wait-for-service.sh http://localhost:8080

    - name: Run UI tests
      env:
        BROWSER: ${{ matrix.browser }}
        HEADLESS: true
      run: |
        pytest tests/ui/ \
          --junitxml=junit/ui-test-results-${{ matrix.browser }}.xml \
          --html=reports/ui-report-${{ matrix.browser }}.html

    - name: Upload screenshots
      uses: actions/upload-artifact@v3
      if: failure()
      with:
        name: screenshots-${{ matrix.browser }}
        path: screenshots/

  security-tests:
    needs: ui-tests
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Run OWASP ZAP security scan
      uses: zaproxy/action-full-scan@v0.4.0
      with:
        target: 'http://localhost:8080'
        rules_file_name: '.zap/rules.tsv'
        cmd_options: '-a'

  publish-results:
    needs: [unit-tests, integration-tests, api-tests, ui-tests, security-tests]
    runs-on: ubuntu-latest
    if: always()

    steps:
    - name: Download all artifacts
      uses: actions/download-artifact@v3

    - name: Publish test results
      uses: mikepenz/action-junit-report@v3
      if: always()
      with:
        report_paths: '**/junit/*.xml'
        github_token: ${{ secrets.GITHUB_TOKEN }}

    - name: Comment PR with test results
      uses: actions/github-script@v6
      if: github.event_name == 'pull_request'
      with:
        script: |
          const fs = require('fs');
          const path = require('path');

          let comment = '## Test Results Summary\n\n';

          // 读取各种测试结果
          const testTypes = ['unit', 'integration', 'api', 'ui'];
          for (const testType of testTypes) {
            try {
              const resultFile = `junit/${testType}-test-results.xml`;
              if (fs.existsSync(resultFile)) {
                comment += `✅ ${testType} tests passed\n`;
              }
            } catch (error) {
              comment += `❌ ${testType} tests failed\n`;
            }
          }

          github.rest.issues.createComment({
            issue_number: context.issue.number,
            owner: context.repo.owner,
            repo: context.repo.repo,
            body: comment
          });
```

#### 3. 测试报告和通知
```python
# scripts/test_reporter.py
import json
import requests
import xml.etree.ElementTree as ET
from typing import Dict, List
import os

class TestReporter:
    def __init__(self):
        self.webhook_url = os.getenv('SLACK_WEBHOOK_URL')
        self.github_token = os.getenv('GITHUB_TOKEN')

    def parse_junit_results(self, xml_file: str) -> Dict:
        """解析JUnit XML结果"""
        tree = ET.parse(xml_file)
        root = tree.getroot()

        results = {
            'total': int(root.get('tests', 0)),
            'passed': 0,
            'failed': int(root.get('failures', 0)),
            'errors': int(root.get('errors', 0)),
            'skipped': int(root.get('skipped', 0)),
            'time': float(root.get('time', 0)),
            'failure_details': []
        }

        results['passed'] = results['total'] - results['failed'] - results['errors'] - results['skipped']

        # 收集失败详情
        for testcase in root.findall('.//testcase'):
            failure = testcase.find('failure')
            error = testcase.find('error')

            if failure is not None or error is not None:
                failure_info = {
                    'test_name': testcase.get('name'),
                    'class_name': testcase.get('classname'),
                    'message': (failure or error).get('message', ''),
                    'details': (failure or error).text
                }
                results['failure_details'].append(failure_info)

        return results

    def generate_summary_report(self, test_results: Dict[str, Dict]) -> str:
        """生成测试摘要报告"""
        total_tests = sum(result['total'] for result in test_results.values())
        total_passed = sum(result['passed'] for result in test_results.values())
        total_failed = sum(result['failed'] for result in test_results.values())
        total_errors = sum(result['errors'] for result in test_results.values())

        success_rate = (total_passed / total_tests * 100) if total_tests > 0 else 0

        report = f"""
# 测试执行报告

## 总体统计
- **总测试数**: {total_tests}
- **通过**: {total_passed} ✅
- **失败**: {total_failed} ❌
- **错误**: {total_errors} ⚠️
- **成功率**: {success_rate:.2f}%

## 分类结果
"""

        for test_type, result in test_results.items():
            status_emoji = "✅" if result['failed'] == 0 and result['errors'] == 0 else "❌"
            report += f"### {test_type.upper()} {status_emoji}\n"
            report += f"- 通过: {result['passed']}/{result['total']}\n"
            report += f"- 执行时间: {result['time']:.2f}秒\n\n"

        # 添加失败详情
        if total_failed > 0 or total_errors > 0:
            report += "## 失败详情\n"
            for test_type, result in test_results.items():
                if result['failure_details']:
                    report += f"### {test_type.upper()}\n"
                    for failure in result['failure_details']:
                        report += f"- **{failure['test_name']}**: {failure['message']}\n"

        return report

    def send_slack_notification(self, report: str, success: bool):
        """发送Slack通知"""
        if not self.webhook_url:
            return

        color = "good" if success else "danger"
        icon = ":white_check_mark:" if success else ":x:"

        payload = {
            "attachments": [
                {
                    "color": color,
                    "title": f"{icon} 测试执行报告",
                    "text": report,
                    "mrkdwn_in": ["text"]
                }
            ]
        }

        requests.post(self.webhook_url, json=payload)

    def create_github_comment(self, pr_number: int, report: str):
        """在GitHub PR上创建评论"""
        if not self.github_token:
            return

        headers = {
            "Authorization": f"token {self.github_token}",
            "Accept": "application/vnd.github.v3+json"
        }

        comment_data = {
            "body": report
        }

        repo = os.getenv('GITHUB_REPOSITORY')
        url = f"https://api.github.com/repos/{repo}/issues/{pr_number}/comments"

        requests.post(url, json=comment_data, headers=headers)

# 使用示例
if __name__ == "__main__":
    reporter = TestReporter()

    # 收集所有测试结果
    test_results = {}

    result_files = [
        ('unit', 'junit/unit-test-results.xml'),
        ('integration', 'junit/integration-test-results.xml'),
        ('api', 'junit/api-test-results.xml'),
        ('ui', 'junit/ui-test-results.xml')
    ]

    for test_type, file_path in result_files:
        if os.path.exists(file_path):
            test_results[test_type] = reporter.parse_junit_results(file_path)

    # 生成报告
    summary_report = reporter.generate_summary_report(test_results)

    # 判断整体成功状态
    overall_success = all(
        result['failed'] == 0 and result['errors'] == 0
        for result in test_results.values()
    )

    # 发送通知
    reporter.send_slack_notification(summary_report, overall_success)

    # 如果是PR，创建GitHub评论
    pr_number = os.getenv('PR_NUMBER')
    if pr_number:
        reporter.create_github_comment(int(pr_number), summary_report)
```
