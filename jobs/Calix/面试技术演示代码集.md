```python
#!/usr/bin/env python3
"""
SNMP设备快速诊断工具
演示网络协议深度应用和故障排查能力
"""

import subprocess
import sys
import time
import json
from typing import Dict, List, Optional

class SNMPDiagnosticTool:
    """SNMP诊断工具 - 面试演示版本"""

    def __init__(self):
        self.common_oids = {
            'sysDescr': '1.3.6.1.2.1.1.1.0',
            'sysUpTime': '1.3.6.1.2.1.1.3.0',
            'sysName': '1.3.6.1.2.1.1.5.0',
            'sysLocation': '1.3.6.1.2.1.1.6.0',
            'ifNumber': '1.3.6.1.2.1.2.1.0',
            'ifInOctets': '1.3.6.1.2.1.2.2.1.10',
            'ifOutOctets': '1.3.6.1.2.1.2.2.1.16',
            'ifAdminStatus': '1.3.6.1.2.1.2.2.1.7',
            'ifOperStatus': '1.3.6.1.2.1.2.2.1.8'
        }

    def quick_snmp_test(self, host: str, community: str = 'public') -> Dict:
        """
        快速SNMP连通性和基础信息测试
        演示: 网络协议应用和故障诊断能力
        """
        print(f"🔍 开始SNMP诊断: {host}")
        result = {
            'host': host,
            'community': community,
            'timestamp': time.time(),
            'connectivity': False,
            'basic_info': {},
            'interface_summary': {},
            'performance_data': {},
            'errors': []
        }

        try:
            # 1. 基础连通性测试 - 获取系统描述
            print("  📋 测试基础连通性...")
            sys_info = self._snmp_get(host, community, self.common_oids['sysDescr'])

            if sys_info['success']:
                result['connectivity'] = True
                result['basic_info']['description'] = sys_info['value']
                print(f"  ✅ SNMP连通性正常")

                # 2. 收集系统基础信息
                print("  📊 收集系统信息...")
                basic_oids = ['sysUpTime', 'sysName', 'sysLocation']
                for oid_name in basic_oids:
                    info = self._snmp_get(host, community, self.common_oids[oid_name])
                    if info['success']:
                        result['basic_info'][oid_name] = info['value']

                # 3. 接口数量和状态检查
                print("  🔌 检查网络接口...")
                interface_data = self._check_interfaces(host, community)
                result['interface_summary'] = interface_data

                # 4. 性能数据采样
                print("  📈 采集性能数据...")
                perf_data = self._collect_performance_sample(host, community)
                result['performance_data'] = perf_data

                print(f"  ✅ 诊断完成 - 设备: {result['basic_info'].get('sysName', 'Unknown')}")

            else:
                result['errors'].append(f"SNMP连通性失败: {sys_info['error']}")
                print(f"  ❌ SNMP连通性失败: {sys_info['error']}")

        except Exception as e:
            result['errors'].append(f"诊断过程异常: {str(e)}")
            print(f"  ❌ 诊断异常: {e}")

        return result

    def _snmp_get(self, host: str, community: str, oid: str, timeout: int = 5) -> Dict:
        """执行SNMP GET操作"""
        try:
            cmd = ['snmpget', '-v2c', '-c', community, '-t', str(timeout), host, oid]
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=timeout+2)

            if result.returncode == 0 and result.stdout:
                # 解析SNMP输出
                output = result.stdout.strip()
                if '=' in output:
                    value_part = output.split('=', 1)[1].strip()
                    # 简化的值解析
                    if value_part.startswith('STRING:'):
                        value = value_part.replace('STRING:', '').strip().strip('"')
                    elif value_part.startswith('INTEGER:'):
                        value = value_part.replace('INTEGER:', '').strip()
                    elif value_part.startswith('Timeticks:'):
                        value = value_part.replace('Timeticks:', '').strip()
                    else:
                        value = value_part

                    return {'success': True, 'value': value, 'raw': output}

            return {'success': False, 'error': result.stderr or 'No response', 'raw': result.stdout}

        except subprocess.TimeoutExpired:
            return {'success': False, 'error': 'Timeout'}
        except Exception as e:
            return {'success': False, 'error': str(e)}

    def _check_interfaces(self, host: str, community: str) -> Dict:
        """检查网络接口状态"""
        interface_summary = {
            'total_interfaces': 0,
            'admin_up': 0,
            'oper_up': 0,
            'interfaces': []
        }

        try:
            # 获取接口数量
            if_num_result = self._snmp_get(host, community, self.common_oids['ifNumber'])
            if if_num_result['success']:
                interface_summary['total_interfaces'] = int(if_num_result['value'])

                # 获取前几个接口的状态作为示例
                for i in range(1, min(6, interface_summary['total_interfaces'] + 1)):
                    admin_oid = f"{self.common_oids['ifAdminStatus']}.{i}"
                    oper_oid = f"{self.common_oids['ifOperStatus']}.{i}"

                    admin_status = self._snmp_get(host, community, admin_oid)
                    oper_status = self._snmp_get(host, community, oper_oid)

                    if admin_status['success'] and oper_status['success']:
                        admin_val = int(admin_status['value'])
                        oper_val = int(oper_status['value'])

                        interface_summary['interfaces'].append({
                            'index': i,
                            'admin_status': 'up' if admin_val == 1 else 'down',
                            'oper_status': 'up' if oper_val == 1 else 'down'
                        })

                        if admin_val == 1:
                            interface_summary['admin_up'] += 1
                        if oper_val == 1:
                            interface_summary['oper_up'] += 1

        except Exception as e:
            interface_summary['error'] = str(e)

        return interface_summary

    def _collect_performance_sample(self, host: str, community: str) -> Dict:
        """收集性能数据样本"""
        perf_data = {'timestamp': time.time(), 'metrics': {}}

        try:
            # 获取第一个接口的流量计数器作为示例
            in_octets = self._snmp_get(host, community, f"{self.common_oids['ifInOctets']}.1")
            out_octets = self._snmp_get(host, community, f"{self.common_oids['ifOutOctets']}.1")

            if in_octets['success']:
                perf_data['metrics']['if1_in_octets'] = int(in_octets['value'])
            if out_octets['success']:
                perf_data['metrics']['if1_out_octets'] = int(out_octets['value'])

            # 系统运行时间
            uptime = self._snmp_get(host, community, self.common_oids['sysUpTime'])
            if uptime['success']:
                perf_data['metrics']['uptime_ticks'] = uptime['value']

        except Exception as e:
            perf_data['error'] = str(e)

        return perf_data

    def generate_report(self, results: List[Dict]) -> str:
        """生成诊断报告"""
        report = "="*60 + "\n"
        report += "📋 SNMP设备诊断报告\n"
        report += "="*60 + "\n\n"

        total_devices = len(results)
        successful_devices = len([r for r in results if r['connectivity']])

        report += f"📊 概况统计:\n"
        report += f"  测试设备数量: {total_devices}\n"
        report += f"  连通设备数量: {successful_devices}\n"
        report += f"  连通成功率: {successful_devices/total_devices*100:.1f}%\n\n"

        for i, result in enumerate(results, 1):
            report += f"📱 设备 {i}: {result['host']}\n"
            if result['connectivity']:
                basic_info = result['basic_info']
                report += f"  ✅ 连通性: 正常\n"
                report += f"  📋 系统描述: {basic_info.get('description', 'N/A')}\n"
                report += f"  🏷️  系统名称: {basic_info.get('sysName', 'N/A')}\n"
                report += f"  📍 位置信息: {basic_info.get('sysLocation', 'N/A')}\n"
                report += f"  ⏱️  运行时间: {basic_info.get('sysUpTime', 'N/A')}\n"

                if 'interface_summary' in result:
                    intf_summary = result['interface_summary']
                    report += f"  🔌 接口统计: 总数={intf_summary.get('total_interfaces', 0)}, "
                    report += f"管理UP={intf_summary.get('admin_up', 0)}, "
                    report += f"操作UP={intf_summary.get('oper_up', 0)}\n"
            else:
                report += f"  ❌ 连通性: 失败\n"
                report += f"  📝 错误信息: {'; '.join(result['errors'])}\n"

            report += "\n"

        report += "="*60 + "\n"
        return report

# 演示使用示例
if __name__ == "__main__":
    print("🚀 SNMP诊断工具演示")
    print("适用于网络设备快速健康检查")
    print()

    # 模拟测试设备列表
    test_devices = [
        {'host': '192.168.1.1', 'community': 'public'},
        {'host': '192.168.1.10', 'community': 'public'},
        {'host': '10.0.0.1', 'community': 'private'}
    ]

    tool = SNMPDiagnosticTool()
    results = []

    for device in test_devices:
        result = tool.quick_snmp_test(device['host'], device['community'])
        results.append(result)
        time.sleep(1)  # 避免过于频繁的请求

    # 生成并显示报告
    report = tool.generate_report(results)
    print(report)
```

#### 网络连通性测试工具
```python
#!/usr/bin/env python3
"""
网络连通性综合测试工具
演示: 网络故障排查的系统化方法
"""

import subprocess
import socket
import time
import concurrent.futures
from typing import Dict, List, Tuple

class NetworkConnectivityTester:
    """网络连通性测试工具"""

    def __init__(self):
        self.test_results = []

    def comprehensive_connectivity_test(self, target: str) -> Dict:
        """
        综合连通性测试
        演示: 分层网络诊断方法
        """
        print(f"🔍 开始综合连通性测试: {target}")

        result = {
            'target': target,
            'timestamp': time.time(),
            'tests': {}
        }

        # 并行执行多种测试
        with concurrent.futures.ThreadPoolExecutor(max_workers=4) as executor:
            futures = {
                'ping': executor.submit(self._ping_test, target),
                'tcp_ports': executor.submit(self._tcp_port_scan, target, [22, 80, 443, 161, 830]),
                'dns_resolution': executor.submit(self._dns_resolution_test, target),
                'traceroute': executor.submit(self._traceroute_test, target)
            }

            # 收集结果
            for test_name, future in futures.items():
                try:
                    result['tests'][test_name] = future.result(timeout=30)
                except Exception as e:
                    result['tests'][test_name] = {'error': str(e)}

        # 综合分析
        result['analysis'] = self._analyze_connectivity(result['tests'])

        return result

    def _ping_test(self, target: str, count: int = 4) -> Dict:
        """PING连通性测试"""
        print(f"  🏓 执行PING测试...")

        try:
            cmd = ['ping', '-c', str(count), target]
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=30)

            if result.returncode == 0:
                # 解析ping结果
                output_lines = result.stdout.split('\n')

                # 查找统计行
                stats_line = None
                for line in output_lines:
                    if 'packet loss' in line:
                        stats_line = line
                        break

                if stats_line:
                    # 提取丢包率
                    packet_loss_str = stats_line.split(',')[2].strip()
                    packet_loss = float(packet_loss_str.split('%')[0].split()[-1])

                    # 查找RTT统计
                    rtt_line = None
                    for line in output_lines:
                        if 'min/avg/max' in line:
                            rtt_line = line
                            break

                    rtt_stats = {}
                    if rtt_line:
                        rtt_part = rtt_line.split('=')[1].strip()
                        min_rtt, avg_rtt, max_rtt, stddev = rtt_part.split('/')
                        rtt_stats = {
                            'min': float(min_rtt),
                            'avg': float(avg_rtt),
                            'max': float(max_rtt),
                            'stddev': float(stddev.split()[0])
                        }

                    return {
                        'success': True,
                        'packet_loss_percent': packet_loss,
                        'rtt_stats': rtt_stats,
                        'packets_sent': count,
                        'packets_received': count - int(count * packet_loss / 100)
                    }

            return {
                'success': False,
                'error': result.stderr or 'Ping failed',
                'return_code': result.returncode
            }

        except subprocess.TimeoutExpired:
            return {'success': False, 'error': 'Ping timeout'}
        except Exception as e:
            return {'success': False, 'error': str(e)}

    def _tcp_port_scan(self, target: str, ports: List[int]) -> Dict:
        """TCP端口扫描"""
        print(f"  🔍 扫描TCP端口...")

        open_ports = []
        closed_ports = []
        scan_results = {}

        for port in ports:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(3)

            start_time = time.time()
            try:
                result = sock.connect_ex((target, port))
                connect_time = (time.time() - start_time) * 1000

                if result == 0:
                    open_ports.append(port)
                    scan_results[port] = {
                        'status': 'open',
                        'connect_time_ms': round(connect_time, 2)
                    }
                else:
                    closed_ports.append(port)
                    scan_results[port] = {'status': 'closed'}

            except socket.timeout:
                closed_ports.append(port)
                scan_results[port] = {'status': 'timeout'}
            except Exception as e:
                closed_ports.append(port)
                scan_results[port] = {'status': 'error', 'error': str(e)}
            finally:
                sock.close()

        return {
            'ports_scanned': ports,
            'open_ports': open_ports,
            'closed_ports': closed_ports,
            'scan_results': scan_results,
            'summary': f"{len(open_ports)}/{len(ports)} ports open"
        }

    def _dns_resolution_test(self, target: str) -> Dict:
        """DNS解析测试"""
        print(f"  🔍 测试DNS解析...")

        try:
            start_time = time.time()
            ip_addresses = socket.gethostbyname_ex(target)
            resolution_time = (time.time() - start_time) * 1000

            return {
                'success': True,
                'hostname': ip_addresses[0],
                'aliases': ip_addresses[1],
                'addresses': ip_addresses[2],
                'resolution_time_ms': round(resolution_time, 2)
            }

        except socket.gaierror as e:
            return {
                'success': False,
                'error': f"DNS resolution failed: {e}",
                'error_code': e.errno
            }
        except Exception as e:
            return {'success': False, 'error': str(e)}

    def _traceroute_test(self, target: str, max_hops: int = 15) -> Dict:
        """路由追踪测试"""
        print(f"  🛤️  执行路由追踪...")

        try:
            # 使用系统的traceroute命令
            cmd = ['traceroute', '-m', str(max_hops), target]
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=60)

            hops = []
            if result.returncode == 0:
                lines = result.stdout.split('\n')
                for line in lines[1:]:  # 跳过第一行标题
                    if line.strip() and not line.startswith('traceroute'):
                        # 简化的hop解析
                        parts = line.strip().split()
                        if len(parts) > 1:
                            hop_number = parts[0]
                            hop_info = ' '.join(parts[1:])
                            hops.append({
                                'hop': hop_number,
                                'info': hop_info
                            })

                return {
                    'success': True,
                    'hops': hops,
                    'total_hops': len(hops)
                }

            return {
                'success': False,
                'error': result.stderr or 'Traceroute failed'
            }

        except subprocess.TimeoutExpired:
            return {'success': False, 'error': 'Traceroute timeout'}
        except FileNotFoundError:
            return {'success': False, 'error': 'Traceroute command not found'}
        except Exception as e:
            return {'success': False, 'error': str(e)}

    def _analyze_connectivity(self, tests: Dict) -> Dict:
        """分析连通性测试结果"""
        analysis = {
            'overall_status': 'unknown',
            'issues_found': [],
            'recommendations': []
        }

        # 分析PING结果
        ping_result = tests.get('ping', {})
        if ping_result.get('success'):
            packet_loss = ping_result.get('packet_loss_percent', 0)
            avg_rtt = ping_result.get('rtt_stats', {}).get('avg', 0)

            if packet_loss == 0 and avg_rtt < 50:
                analysis['network_quality'] = 'excellent'
            elif packet_loss < 5 and avg_rtt < 100:
                analysis['network_quality'] = 'good'
            elif packet_loss < 15:
                analysis['network_quality'] = 'poor'
                analysis['issues_found'].append(f"High packet loss: {packet_loss}%")
            else:
                analysis['network_quality'] = 'very_poor'
                analysis['issues_found'].append(f"Very high packet loss: {packet_loss}%")

            if avg_rtt > 200:
                analysis['issues_found'].append(f"High latency: {avg_rtt}ms")
        else:
            analysis['issues_found'].append("PING connectivity failed")

        # 分析端口扫描结果
        port_result = tests.get('tcp_ports', {})
        open_ports = port_result.get('open_ports', [])

        if not open_ports:
            analysis['issues_found'].append("No TCP ports responding")
            analysis['recommendations'].append("Check firewall rules and service status")
        else:
            analysis['accessible_services'] = len(open_ports)

        # 分析DNS解析
        dns_result = tests.get('dns_resolution', {})
        if not dns_result.get('success'):
            analysis['issues_found'].append("DNS resolution failed")
            analysis['recommendations'].append("Check DNS server configuration")

        # 综合状态判断
        if not analysis['issues_found']:
            analysis['overall_status'] = 'healthy'
        elif len(analysis['issues_found']) <= 2:
            analysis['overall_status'] = 'warning'
        else:
            analysis['overall_status'] = 'critical'

        return analysis

    def generate_detailed_report(self, result: Dict) -> str:
        """生成详细的连通性报告"""
        target = result['target']
        tests = result['tests']
        analysis = result['analysis']

        report = f"="*70 + "\n"
        report += f"🔍 网络连通性测试报告\n"
        report += f"="*70 + "\n"
        report += f"目标地址: {target}\n"
        report += f"测试时间: {time.strftime('%Y-%m-%d %H:%M:%S', time.localtime(result['timestamp']))}\n"
        report += f"综合状态: {analysis['overall_status'].upper()}\n\n"

        # PING测试结果
        ping_result = tests.get('ping', {})
        report += "📡 PING连通性测试:\n"
        if ping_result.get('success'):
            report += f"  ✅ 状态: 成功\n"
            report += f"  📊 丢包率: {ping_result['packet_loss_percent']}%\n"
            rtt_stats = ping_result.get('rtt_stats', {})
            if rtt_stats:
                report += f"  ⏱️  平均延迟: {rtt_stats['avg']}ms\n"
                report += f"  📈 延迟范围: {rtt_stats['min']}-{rtt_stats['max']}ms\n"
        else:
            report += f"  ❌ 状态: 失败\n"
            report += f"  📝 错误: {ping_result.get('error', 'Unknown')}\n"
        report += "\n"

        # 端口扫描结果
        port_result = tests.get('tcp_ports', {})
        report += "🔍 TCP端口扫描:\n"
        if port_result:
            open_ports = port_result.get('open_ports', [])
            report += f"  📊 扫描结果: {port_result.get('summary', 'N/A')}\n"
            if open_ports:
                report += f"  ✅ 开放端口: {', '.join(map(str, open_ports))}\n"
            else:
                report += f"  ⚠️  开放端口: 无\n"
        report += "\n"

        # DNS解析结果
        dns_result = tests.get('dns_resolution', {})
        report += "🔍 DNS解析测试:\n"
        if dns_result.get('success'):
            report += f"  ✅ 状态: 成功\n"
            report += f"  📍 解析地址: {', '.join(dns_result['addresses'])}\n"
            report += f"  ⏱️  解析时间: {dns_result['resolution_time_ms']}ms\n"
        else:
            report += f"  ❌ 状态: 失败\n"
            report += f"  📝 错误: {dns_result.get('error', 'Unknown')}\n"
        report += "\n"

        # 路由追踪结果
        traceroute_result = tests.get('traceroute', {})
        report += "🛤️  路由追踪:\n"
        if traceroute_result.get('success'):
            report += f"  ✅ 状态: 成功\n"
            report += f"  🔢 跳数: {traceroute_result['total_hops']}\n"
            # 显示前几跳作为示例
            for hop in traceroute_result['hops'][:5]:
                report += f"    {hop['hop']}. {hop['info']}\n"
        else:
            report += f"  ❌ 状态: 失败\n"
            report += f"  📝 错误: {traceroute_result.get('error', 'Unknown')}\n"
        report += "\n"

        # 问题和建议
        issues = analysis.get('issues_found', [])
        recommendations = analysis.get('recommendations', [])

        if issues:
            report += "⚠️  发现的问题:\n"
            for issue in issues:
                report += f"  • {issue}\n"
            report += "\n"

        if recommendations:
            report += "💡 优化建议:\n"
            for rec in recommendations:
                report += f"  • {rec}\n"
            report += "\n"

        report += "="*70 + "\n"
        return report

# 演示使用示例
if __name__ == "__main__":
    print("🚀 网络连通性测试工具演示")

    tester = NetworkConnectivityTester()

    # 测试示例目标
    test_targets = ['8.8.8.8', 'github.com', 'localhost']

    for target in test_targets:
        print(f"\n🎯 测试目标: {target}")
        result = tester.comprehensive_connectivity_test(target)

        # 生成并显示报告
        report = tester.generate_detailed_report(result)
        print(report)

        time.sleep(2)  # 间隔测试
```

```python

"""

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.common.exceptions import TimeoutException
import time

class BasePage:
    """
    增强版BasePage类
    演示: 现代化Page Object设计模式
    """

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)
        self.logger = logging.getLogger(self.__class__.__name__)

    def smart_find_element(self, locator, timeout=None, wait_condition=None):
        """
        智能元素查找
        演示: 容错性设计和多种等待策略
        """
        wait_time = timeout or 10

        try:
            if wait_condition:
                # 自定义等待条件
                element = WebDriverWait(self.driver, wait_time).until(wait_condition)
            else:
                # 默认等待元素出现
                element = WebDriverWait(self.driver, wait_time).until(
                    EC.presence_of_element_located(locator)
                )

            # 滚动到元素位置
            self.driver.execute_script("arguments[0].scrollIntoView(true);", element)

            # 高亮元素（演示时可见）
            self._highlight_element(element)

            return element

        except TimeoutException:
            self.logger.error(f"Element not found: {locator}")
            self._take_debug_screenshot(f"element_not_found_{locator[1]}")
            raise

    def smart_click(self, locator, timeout=None):
        """
        智能点击 - 多种点击策略
        演示: 异常处理和备用方案
        """
        element = self.smart_find_element(locator, timeout)

        try:
            # 等待元素可点击
            clickable_element = WebDriverWait(self.driver, 5).until(
                EC.element_to_be_clickable(locator)
            )
            clickable_element.click()
            self.logger.info(f"Successfully clicked: {locator}")

        except Exception as e:
            # 备用点击方案
            self.logger.warning(f"Normal click failed, using JS click: {e}")
            self.driver.execute_script("arguments[0].click();", element)

    def smart_input(self, locator, text, clear=True, timeout=None):
        """
        智能文本输入
        演示: 输入验证和错误恢复
        """
        element = self.smart_find_element(locator, timeout)

        if clear:
            element.clear()

        element.send_keys(text)

        # 验证输入
        actual_value = element.get_attribute('value')
        if actual_value != text:
            self.logger.warning(f"Input verification failed. Expected: {text}, Got: {actual_value}")
            # 重试输入
            element.clear()
            element.send_keys(text)

        self.logger.info(f"Input completed: {text}")

    def wait_for_page_transition(self, expected_url_contains=None, timeout=10):
        """
        等待页面跳转完成
        演示: 页面状态管理
        """
        try:
            # 等待页面加载完成
            WebDriverWait(self.driver, timeout).until(
                lambda driver: driver.execute_script("return document.readyState") == "complete"
            )

            # 等待AJAX完成（如果页面使用jQuery）
            WebDriverWait(self.driver, timeout).until(
                lambda driver: driver.execute_script("return window.jQuery && jQuery.active === 0")
            )

            # 验证URL（如果指定）
            if expected_url_contains:
                WebDriverWait(self.driver, timeout).until(
                    lambda driver: expected_url_contains in driver.current_url
                )

            self.logger.info("Page transition completed")

        except TimeoutException:
            self.logger.warning("Page transition timeout")

    def _highlight_element(self, element):
        """高亮元素用于演示"""
        try:
            self.driver.execute_script(
                "arguments[0].style.border='3px solid red'",
                element
            )
            time.sleep(0.5)  # 短暂高亮
            self.driver.execute_script(
                "arguments[0].style.border=''",
                element
            )
        except:
            pass  # 忽略高亮失败

    def _take_debug_screenshot(self, name):
        """调试截图"""
        try:
            timestamp = int(time.time())
            filename = f"debug_{name}_{timestamp}.png"
            self.driver.save_screenshot(filename)
            self.logger.info(f"Debug screenshot saved: {filename}")
        except:
            pass

class ComponentBase:
    """
    页面组件基类
    演示: 组件化设计思想
    """

    def __init__(self, driver, root_locator):
        self.driver = driver
        self.root_locator = root_locator
        self.base = BasePage(driver)

    def get_component_root(self):
        """获取组件根元素"""
        return self.base.smart_find_element(self.root_locator)

    def find_element_in_component(self, child_locator):
        """在组件内查找子元素"""
        root = self.get_component_root()
        return root.find_element(*child_locator)

class DeviceManagementTable(ComponentBase):
    """
    设备管理表格组件
    演示: 复杂组件的封装
    """

    # 表格相关定位器
    HEADER_ROW = (By.CSS_SELECTOR, "thead tr")
    DATA_ROWS = (By.CSS_SELECTOR, "tbody tr")
    SEARCH_INPUT = (By.CSS_SELECTOR, ".table-search input")
    ADD_BUTTON = (By.CSS_SELECTOR, ".add-device-btn")

    def __init__(self, driver, root_locator=(By.ID, "device-table")):
        super().__init__(driver, root_locator)

    def search_device(self, device_name):
        """搜索设备"""
        search_input = self.base.smart_find_element(self.SEARCH_INPUT)
        self.base.smart_input(self.SEARCH_INPUT, device_name)

        # 等待搜索结果更新
        time.sleep(1)  # 简化的等待，实际项目中应使用更智能的等待

        self.base.logger.info(f"Searched for device: {device_name}")

    def get_device_count(self):
        """获取设备数量"""
        rows = self.base.driver.find_elements(*self.DATA_ROWS)
        return len(rows)

    def get_device_by_name(self, device_name):
        """根据名称获取设备行"""
        rows = self.base.driver.find_elements(*self.DATA_ROWS)

        for row in rows:
            if device_name in row.text:
                return DeviceRow(self.driver, row)

        return None

    def click_add_device(self):
        """点击添加设备按钮"""
        self.base.smart_click(self.ADD_BUTTON)
        return AddDeviceDialog(self.driver)

class AddDeviceDialog:
    """
    添加设备对话框
    演示: 对话框组件的流畅接口设计
    """

    def __init__(self, driver):
        self.driver = driver
        self.base = BasePage(driver)

        # 等待对话框出现
        self.base.smart_find_element((By.ID, "add-device-modal"))

    def enter_device_name(self, name):
        """输入设备名称"""
        self.base.smart_input((By.ID, "device-name"), name)
        return self  # 返回自身支持链式调用

    def enter_device_ip(self, ip):
        """输入设备IP"""
        self.base.smart_input((By.ID, "device-ip"), ip)
        return self

    def select_device_type(self, device_type):
        """选择设备类型"""
        type_dropdown = self.base.smart_find_element((By.ID, "device-type"))

        # 点击下拉框
        type_dropdown.click()

        # 选择选项
        option_locator = (By.XPATH, f"//option[@value='{device_type}']")
        self.base.smart_click(option_locator)

        return self

    def configure_snmp(self, community="public", port=161):
        """配置SNMP参数"""
        # 切换到SNMP配置标签
        snmp_tab = self.base.smart_find_element((By.ID, "snmp-config-tab"))
        snmp_tab.click()

        self.base.smart_input((By.ID, "snmp-community"), community)
        self.base.smart_input((By.ID, "snmp-port"), str(port))

        return self

    def save(self):
        """保存设备"""
        save_btn = self.base.smart_find_element((By.ID, "save-device-btn"))
        save_btn.click()

        # 等待保存完成（通过成功消息或对话框关闭）
        try:
            WebDriverWait(self.driver, 10).until(
                EC.presence_of_element_located((By.CLASS_NAME, "success-message"))
            )
            self.base.logger.info("Device saved successfully")
        except TimeoutException:
            self.base.logger.warning("Success message not found")

    def cancel(self):
        """取消操作"""
        cancel_btn = self.base.smart_find_element((By.ID, "cancel-btn"))
        cancel_btn.click()

class DeviceManagementPage(BasePage):
    """
    设备管理主页面
    演示: 流畅接口和高级页面交互
    """

    def __init__(self, driver):
        super().__init__(driver)
        self.device_table = DeviceManagementTable(driver)

    def navigate_to_device_management(self):
        """导航到设备管理页面"""
        menu_item = self.smart_find_element((By.XPATH, "//a[contains(text(),'Device Management')]"))
        menu_item.click()

        # 等待页面加载
        self.wait_for_page_transition(expected_url_contains="device-management")

        return self

    def add_device_complete_flow(self, device_name, device_ip, device_type="switch"):
        """
        添加设备的完整流程
        演示: 业务流程的封装
        """
        self.logger.info(f"Starting device creation flow: {device_name}")

        dialog = self.device_table.click_add_device()

        dialog.enter_device_name(device_name) \
              .enter_device_ip(device_ip) \
              .select_device_type(device_type) \
              .configure_snmp() \
              .save()

        # 验证设备是否出现在列表中
        time.sleep(2)  # 等待列表刷新
        device_row = self.device_table.get_device_by_name(device_name)

        if device_row:
            self.logger.info(f"Device {device_name} created successfully")
            return True
        else:
            self.logger.error(f"Device {device_name} not found after creation")
            return False
```

```python
#!/usr/bin/env python3
"""
面试现场演示 - 网络健康检查
可直接在面试官的机器上运行
"""

import subprocess
import socket
import time
import json
import sys

def quick_network_health_check():
    """快速网络健康检查"""

    print("🚀 网络健康检查工具")
    print("="*50)

    results = {
        'timestamp': time.time(),
        'tests': {}
    }

    # 1. 本地网络配置检查
    print("📋 1. 检查本地网络配置...")
    try:
        hostname = socket.gethostname()
        local_ip = socket.gethostbyname(hostname)
        print(f"   主机名: {hostname}")
        print(f"   本地IP: {local_ip}")
        results['tests']['local_config'] = {
            'hostname': hostname,
            'local_ip': local_ip,
            'status': 'ok'
        }
    except Exception as e:
        print(f"   ❌ 错误: {e}")
        results['tests']['local_config'] = {'status': 'error', 'error': str(e)}

    # 2. DNS解析测试
    print("\n🔍 2. DNS解析测试...")
    dns_targets = ['google.com', 'github.com', '8.8.8.8']
    dns_results = []

    for target in dns_targets:
        try:
            start_time = time.time()
            addr = socket.gethostbyname(target)
            resolution_time = (time.time() - start_time) * 1000

            print(f"   ✅ {target} -> {addr} ({resolution_time:.1f}ms)")
            dns_results.append({
                'target': target,
                'resolved_ip': addr,
                'resolution_time_ms': resolution_time,
                'status': 'success'
            })
        except Exception as e:
            print(f"   ❌ {target} -> 解析失败: {e}")
            dns_results.append({
                'target': target,
                'status': 'failed',
                'error': str(e)
            })

    results['tests']['dns'] = dns_results

    # 3. 连通性测试
    print("\n🏓 3. 连通性测试...")
    ping_targets = ['8.8.8.8', 'google.com']
    ping_results = []

    for target in ping_targets:
        try:
            # 执行ping命令
            cmd = ['ping', '-c', '3', target]
            result = subprocess.run(cmd, capture_output=True, text=True, timeout=15)

            if result.returncode == 0:
                # 简单解析ping结果
                lines = result.stdout.split('\n')
                for line in lines:
                    if 'packet loss' in line:
                        packet_loss = line.split(',')[2].strip().split('%')[0].split()[-1]
                        print(f"   ✅ {target} - 丢包率: {packet_loss}%")

                        ping_results.append({
                            'target': target,
                            'packet_loss': packet_loss + '%',
                            'status': 'success'
                        })
                        break
            else:
                print(f"   ❌ {target} - ping失败")
                ping_results.append({
                    'target': target,
                    'status': 'failed',
                    'error': 'ping command failed'
                })

        except subprocess.TimeoutExpired:
            print(f"   ⏱️  {target} - ping超时")
            ping_results.append({
                'target': target,
                'status': 'timeout'
            })
        except Exception as e:
            print(f"   ❌ {target} - 错误: {e}")
            ping_results.append({
                'target': target,
                'status': 'error',
                'error': str(e)
            })

    results['tests']['connectivity'] = ping_results

    # 4. 端口连通性测试
    print("\n🔌 4. 端口连通性测试...")
    port_tests = [
        ('google.com', 80),
        ('google.com', 443),
        ('8.8.8.8', 53)
    ]

    port_results = []
    for host, port in port_tests:
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(5)

            start_time = time.time()
            result = sock.connect_ex((host, port))
            connect_time = (time.time() - start_time) * 1000

            if result == 0:
                print(f"   ✅ {host}:{port} - 连接成功 ({connect_time:.1f}ms)")
                port_results.append({
                    'host': host,
                    'port': port,
                    'status': 'open',
                    'connect_time_ms': connect_time
                })
            else:
                print(f"   ❌ {host}:{port} - 连接失败")
                port_results.append({
                    'host': host,
                    'port': port,
                    'status': 'closed'
                })

            sock.close()

        except Exception as e:
            print(f"   ❌ {host}:{port} - 错误: {e}")
            port_results.append({
                'host': host,
                'port': port,
                'status': 'error',
                'error': str(e)
            })

    results['tests']['ports'] = port_results

    # 5. 生成总结
    print("\n📊 5. 测试总结")
    print("="*50)

    # 统计成功率
    dns_success = len([r for r in dns_results if r['status'] == 'success'])
    ping_success = len([r for r in ping_results if r['status'] == 'success'])
    port_success = len([r for r in port_results if r['status'] == 'open'])

    print(f"DNS解析: {dns_success}/{len(dns_results)} 成功")
    print(f"连通性测试: {ping_success}/{len(ping_results)} 成功")
    print(f"端口测试: {port_success}/{len(port_results)} 开放")

    # 整体健康状态
    total_tests = len(dns_results) + len(ping_results) + len(port_results)
    successful_tests = dns_success + ping_success + port_success
    success_rate = successful_tests / total_tests * 100

    print(f"\n🎯 整体成功率: {success_rate:.1f}%")

    if success_rate >= 90:
        print("✅ 网络状态: 优秀")
    elif success_rate >= 70:
        print("⚠️  网络状态: 良好")
    elif success_rate >= 50:
        print("⚠️  网络状态: 一般")
    else:
        print("❌ 网络状态: 需要关注")

    # 保存详细结果到JSON文件（可选）
    if '--save' in sys.argv:
        with open(f"network_health_{int(time.time())}.json", 'w') as f:
            json.dump(results, f, indent=2)
        print("\n💾 详细结果已保存到JSON文件")

    print("\n✨ 检查完成")
    return results

if __name__ == "__main__":

    print("展示网络诊断的系统化思路和实际操作能力\n")

    quick_network_health_check()
```
