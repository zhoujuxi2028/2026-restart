BASF Coatings Technology (Shanghai) Co., Ltd. (BCTS) is founded in 2023 in Shanghai, China. interface TestCase {
```typescript
  id: string;
  description: string;
  execute: () => Promise<void>;
}

class TestRunner {
  async runTests(tests: TestCase[]): Promise<void> {
    for (const test of tests) {
      try {
        await test.execute();
        console.log(`✓ ${test.description}`);
      } catch (error) {
        console.error(`✗ ${test.description}`, error);
      }
    }
  }
}
```
