### Page Object 基类
**pages/base_page.py**
```python
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By
from selenium.webdriver.common.action_chains import ActionChains
import logging

class BasePage:
    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)
        self.logger = logging.getLogger(__name__)

    def find_element(self, locator, timeout=10):
        """安全查找元素"""
        try:
            wait = WebDriverWait(self.driver, timeout)
            return wait.until(EC.presence_of_element_located(locator))
        except Exception as e:
            self.logger.error(f"Element not found: {locator}, Error: {e}")
            raise

    def click_element(self, locator, timeout=10):
        """安全点击元素"""
        element = self.wait_for_clickable(locator, timeout)
        try:
            element.click()
        except Exception as e:
            # 如果普通点击失败，使用JavaScript点击
            self.driver.execute_script("arguments[0].click();", element)
            self.logger.warning(f"Used JS click for {locator}")

    def wait_for_clickable(self, locator, timeout=10):
        """等待元素可点击"""
        wait = WebDriverWait(self.driver, timeout)
        return wait.until(EC.element_to_be_clickable(locator))

    def enter_text(self, locator, text, clear=True):
        """输入文本"""
        element = self.find_element(locator)
        if clear:
            element.clear()
        element.send_keys(text)

    def get_text(self, locator):
        """获取元素文本"""
        element = self.find_element(locator)
        return element.text

    def is_element_visible(self, locator, timeout=5):
        """检查元素是否可见"""
        try:
            wait = WebDriverWait(self.driver, timeout)
            wait.until(EC.visibility_of_element_located(locator))
            return True
        except:
            return False

    def scroll_to_element(self, locator):
        """滚动到元素"""
        element = self.find_element(locator)
        self.driver.execute_script("arguments[0].scrollIntoView(true);", element)

    def take_screenshot(self, filename):
        """截图"""
        self.driver.save_screenshot(filename)
        self.logger.info(f"Screenshot saved: {filename}")
```

### 登录页面模板
**pages/login_page.py**
```python
from selenium.webdriver.common.by import By
from .base_page import BasePage

class LoginPage(BasePage):
    # 页面元素定位器
    USERNAME_INPUT = (By.ID, "username")
    PASSWORD_INPUT = (By.ID, "password")
    LOGIN_BUTTON = (By.ID, "login-btn")
    ERROR_MESSAGE = (By.CLASS_NAME, "error-message")
    REMEMBER_ME_CHECKBOX = (By.ID, "remember-me")

    def __init__(self, driver):
        super().__init__(driver)
        self.url = "http://192.168.1.100:8080/login"

    def navigate_to_login(self):
        """导航到登录页面"""
        self.driver.get(self.url)
        self.logger.info(f"Navigated to {self.url}")

    def enter_username(self, username):
        """输入用户名"""
        self.enter_text(self.USERNAME_INPUT, username)
        self.logger.info(f"Entered username: {username}")

    def enter_password(self, password):
        """输入密码"""
        self.enter_text(self.PASSWORD_INPUT, password)
        self.logger.info("Entered password")

    def click_login(self):
        """点击登录按钮"""
        self.click_element(self.LOGIN_BUTTON)
        self.logger.info("Clicked login button")

    def login(self, username, password):
        """完整登录流程"""
        self.navigate_to_login()
        self.enter_username(username)
        self.enter_password(password)
        self.click_login()

    def is_login_successful(self):
        """检查登录是否成功"""
        try:
            # 等待页面跳转，检查是否出现Dashboard
            dashboard_locator = (By.CLASS_NAME, "dashboard")
            return self.is_element_visible(dashboard_locator, timeout=10)
        except:
            return False

    def get_error_message(self):
        """获取错误信息"""
        if self.is_element_visible(self.ERROR_MESSAGE, timeout=3):
            return self.get_text(self.ERROR_MESSAGE)
        return None

    def toggle_remember_me(self):
        """切换记住我选项"""
        self.click_element(self.REMEMBER_ME_CHECKBOX)
```

### 设备管理页面模板
**pages/device_management_page.py**
```python
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import Select
from .base_page import BasePage

class DeviceManagementPage(BasePage):
    # 页面元素
    ADD_DEVICE_BUTTON = (By.ID, "add-device-btn")
    DEVICE_TABLE = (By.ID, "device-table")
    SEARCH_INPUT = (By.ID, "device-search")
    FILTER_DROPDOWN = (By.ID, "device-filter")

    # 设备添加对话框
    DEVICE_NAME_INPUT = (By.ID, "device-name")
    DEVICE_IP_INPUT = (By.ID, "device-ip")
    DEVICE_TYPE_SELECT = (By.ID, "device-type")
    SAVE_DEVICE_BUTTON = (By.ID, "save-device-btn")
    CANCEL_BUTTON = (By.ID, "cancel-btn")

    # 成功/错误消息
    SUCCESS_MESSAGE = (By.CLASS_NAME, "success-message")
    ERROR_MESSAGE = (By.CLASS_NAME, "error-message")

    def navigate_to_device_management(self):
        """导航到设备管理页面"""
        device_menu = (By.XPATH, "//a[contains(text(),'Device Management')]")
        self.click_element(device_menu)
        self.wait_for_page_load()

    def wait_for_page_load(self):
        """等待页面加载完成"""
        self.find_element(self.DEVICE_TABLE, timeout=15)

    def click_add_device(self):
        """点击添加设备按钮"""
        self.click_element(self.ADD_DEVICE_BUTTON)

    def add_device(self, device_name, device_ip, device_type):
        """添加新设备"""
        self.click_add_device()

        # 等待对话框出现
        modal = (By.ID, "device-modal")
        self.find_element(modal)

        # 填写设备信息
        self.enter_text(self.DEVICE_NAME_INPUT, device_name)
        self.enter_text(self.DEVICE_IP_INPUT, device_ip)

        # 选择设备类型
        device_type_element = self.find_element(self.DEVICE_TYPE_SELECT)
        select = Select(device_type_element)
        select.select_by_value(device_type)

        # 保存设备
        self.click_element(self.SAVE_DEVICE_BUTTON)

        # 等待操作完成
        self.wait_for_message()

    def search_device(self, search_term):
        """搜索设备"""
        self.enter_text(self.SEARCH_INPUT, search_term)
        # 等待搜索结果
        import time
        time.sleep(2)

    def get_device_count(self):
        """获取设备数量"""
        table = self.find_element(self.DEVICE_TABLE)
        rows = table.find_elements(By.TAG_NAME, "tr")
        return len(rows) - 1  # 减去表头

    def is_device_in_table(self, device_name):
        """检查设备是否在表格中"""
        table_text = self.get_text(self.DEVICE_TABLE)
        return device_name in table_text

    def wait_for_message(self, timeout=10):
        """等待成功或错误消息"""
        from selenium.webdriver.support import expected_conditions as EC
        from selenium.common.exceptions import TimeoutException

        try:
            WebDriverWait(self.driver, timeout).until(
                EC.any_of(
                    EC.visibility_of_element_located(self.SUCCESS_MESSAGE),
                    EC.visibility_of_element_located(self.ERROR_MESSAGE)
                )
            )
        except TimeoutException:
            self.logger.warning("No message appeared within timeout")

    def get_success_message(self):
        """获取成功消息"""
        if self.is_element_visible(self.SUCCESS_MESSAGE, timeout=3):
            return self.get_text(self.SUCCESS_MESSAGE)
        return None

    def get_error_message(self):
        """获取错误消息"""
        if self.is_element_visible(self.ERROR_MESSAGE, timeout=3):
            return self.get_text(self.ERROR_MESSAGE)
        return None
```

### Page Object模式
**tests/pages/DeviceManagementPage.js**
```javascript
export class DeviceManagementPage {
  constructor(page) {
    this.page = page
    this.addDeviceButton = page.locator('#add-device-btn')
    this.deviceTable = page.locator('#device-table')
    this.searchInput = page.locator('#device-search')

    // 设备对话框
    this.deviceNameInput = page.locator('#device-name')
    this.deviceIPInput = page.locator('#device-ip')
    this.deviceTypeSelect = page.locator('#device-type')
    this.saveButton = page.locator('#save-device-btn')

    // 消息
    this.successMessage = page.locator('.success-message')
    this.errorMessage = page.locator('.error-message')
  }

  async navigate() {
    await this.page.click('text=Device Management')
    await this.deviceTable.waitFor()
  }

  async addDevice(name, ip, type) {
    await this.addDeviceButton.click()
    await this.page.locator('#device-modal').waitFor()

    await this.deviceNameInput.fill(name)
    await this.deviceIPInput.fill(ip)
    await this.deviceTypeSelect.selectOption(type)

    await this.saveButton.click()
  }

  async searchDevice(searchTerm) {
    await this.searchInput.fill(searchTerm)
    await this.page.waitForTimeout(1000) // 等待搜索结果
  }

  async getDeviceCount() {
    const rows = await this.deviceTable.locator('tr').count()
    return rows - 1 // 减去表头
  }

  async isDeviceVisible(deviceName) {
    return await this.deviceTable.textContent().then(text => text.includes(deviceName))
  }
}
```

import pytest
import allure
from selenium import webdriver
from webdriver_manager.chrome import ChromeDriverManager
from webdriver_manager.firefox import GeckoDriverManager

@pytest.fixture(scope="session")
def driver_init(request):
    browser = request.config.getoption("--browser", default="chrome")

    if browser == "chrome":
        options = webdriver.ChromeOptions()
        options.add_argument("--headless")
        driver = webdriver.Chrome(
            service=Service(ChromeDriverManager().install()),
            options=options
        )
    elif browser == "firefox":
        options = webdriver.FirefoxOptions()
        options.add_argument("--headless")
        driver = webdriver.Firefox(
            service=Service(GeckoDriverManager().install()),
            options=options
        )

    yield driver
    driver.quit()

@pytest.hookimpl(tryfirst=True, hookwrapper=True)
def pytest_runtest_makereport(item, call):
    outcome = yield
    rep = outcome.get_result()

    if rep.when == "call" and rep.failed:
        # 测试失败时截图
        driver = item.funcargs.get('driver')
        if driver:
            allure.attach(
                driver.get_screenshot_as_png(),
                name="screenshot",
                attachment_type=allure.attachment_type.PNG
            )

## 🎯 面试重点总结

### 框架选择考虑因素
1. **Selenium**: 跨浏览器支持好，生态成熟，适合复杂场景
2. **Cypress**: 前端友好，调试方便，JavaScript生态
3. **Playwright**: 现代化，速度快，多浏览器并行

### 自动化最佳实践
- **Page Object模式**: 提高代码复用性
- **数据驱动**: 参数化测试用例
- **等待策略**: 显式等待优于隐式等待
- **错误处理**: 优雅的异常处理和重试机制
- **并行执行**: 提高测试效率
- **CI/CD集成**: 自动化测试流水线

### 实战经验准备
准备能够演示的自动化测试项目，包括：
- 完整的测试框架设计
- 具体的业务场景测试用例
- CI/CD集成配置
- 测试报告和失败分析方法
