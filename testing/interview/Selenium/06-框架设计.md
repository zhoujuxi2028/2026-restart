```python
import yaml
import os

class Config:
    """配置管理类"""

    _config = None

    @classmethod
    def load(cls, config_file="config/config.yaml"):
        """加载配置"""
        if cls._config is None:
            with open(config_file, "r", encoding="utf-8") as f:
                cls._config = yaml.safe_load(f)
        return cls._config

    @classmethod
    def get(cls, key, default=None):
        """获取配置项"""
        config = cls.load()
        return config.get(key, default)

from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from utils.logger import Logger

class BasePage:
    """页面基类"""

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 20)
        self.logger = Logger.get_logger(self.__class__.__name__)

    def find_element(self, locator):
        """查找元素"""
        self.logger.info(f"查找元素: {locator}")
        return self.wait.until(EC.presence_of_element_located(locator))

    def click(self, locator):
        """点击元素"""
        self.logger.info(f"点击元素: {locator}")
        element = self.wait.until(EC.element_to_be_clickable(locator))
        element.click()

    def enter_text(self, locator, text):
        """输入文本"""
        self.logger.info(f"输入文本到 {locator}: {text}")
        element = self.find_element(locator)
        element.clear()
        element.send_keys(text)

    def get_text(self, locator):
        """获取文本"""
        element = self.find_element(locator)
        return element.text

import logging
import os
from datetime import datetime

class Logger:
    """日志管理类"""

    _loggers = {}

    @classmethod
    def get_logger(cls, name):
        """获取 logger 实例"""
        if name not in cls._loggers:
            logger = logging.getLogger(name)
            logger.setLevel(logging.DEBUG)

            # 控制台处理器
            console_handler = logging.StreamHandler()
            console_handler.setLevel(logging.INFO)

            # 文件处理器
            log_dir = "logs"
            os.makedirs(log_dir, exist_ok=True)
            log_file = os.path.join(
                log_dir,
                f"test_{datetime.now().strftime('%Y%m%d')}.log"
            )
            file_handler = logging.FileHandler(log_file, encoding="utf-8")
            file_handler.setLevel(logging.DEBUG)

            # 格式化
            formatter = logging.Formatter(
                '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
            )
            console_handler.setFormatter(formatter)
            file_handler.setFormatter(formatter)

            logger.addHandler(console_handler)
            logger.addHandler(file_handler)

            cls._loggers[name] = logger

        return cls._loggers[name]
```

## 第 2 题：如何实现关键字驱动测试框架？

**答案**：

关键字驱动框架通过预定义的关键字执行测试，非技术人员也可以编写测试用例。

```python
# 1. keywords.py - 关键字库
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

class Keywords:
    """关键字库"""

    def __init__(self):
        self.driver = None
        self.wait = None

    def open_browser(self, browser="chrome"):
        """打开浏览器"""
        if browser == "chrome":
            self.driver = webdriver.Chrome()
        self.driver.maximize_window()
        self.wait = WebDriverWait(self.driver, 10)
        return True

    def navigate_to(self, url):
        """导航到 URL"""
        self.driver.get(url)
        return True

    def input_text(self, locator_type, locator_value, text):
        """输入文本"""
        locator = self._get_locator(locator_type, locator_value)
        element = self.wait.until(EC.presence_of_element_located(locator))
        element.clear()
        element.send_keys(text)
        return True

    def click_element(self, locator_type, locator_value):
        """点击元素"""
        locator = self._get_locator(locator_type, locator_value)
        element = self.wait.until(EC.element_to_be_clickable(locator))
        element.click()
        return True

    def verify_text(self, locator_type, locator_value, expected_text):
        """验证文本"""
        locator = self._get_locator(locator_type, locator_value)
        element = self.wait.until(EC.presence_of_element_located(locator))
        actual_text = element.text
        return expected_text in actual_text

    def verify_page_title(self, expected_title):
        """验证页面标题"""
        return expected_title in self.driver.title

    def close_browser(self):
        """关闭浏览器"""
        if self.driver:
            self.driver.quit()
        return True

    def _get_locator(self, locator_type, locator_value):
        """获取定位器"""
        locator_map = {
            "id": By.ID,
            "name": By.NAME,
            "xpath": By.XPATH,
            "css": By.CSS_SELECTOR,
            "class": By.CLASS_NAME,
            "link": By.LINK_TEXT
        }
        by = locator_map.get(locator_type.lower(), By.ID)
        return (by, locator_value)

# 3. keyword_executor.py - 关键字执行器
import openpyxl
from keywords import Keywords

class KeywordExecutor:
    """关键字执行器"""

    def __init__(self):
        self.keywords = Keywords()
        self.results = []

    def load_test_case(self, excel_file, sheet_name="TestCase"):
        """从 Excel 加载测试用例"""
        workbook = openpyxl.load_workbook(excel_file)
        sheet = workbook[sheet_name]

        test_steps = []
        for row in sheet.iter_rows(min_row=2, values_only=True):
            if row[0]:  # 跳过空行
                test_steps.append({
                    "step": row[0],
                    "keyword": row[1],
                    "locator_type": row[2] or "",
                    "locator_value": row[3] or "",
                    "test_data": row[4] or "",
                    "expected": row[5] or ""
                })

        return test_steps

    def execute_test_case(self, test_steps):
        """执行测试用例"""
        for step in test_steps:
            keyword = step["keyword"]
            result = self._execute_keyword(step)

            self.results.append({
                "step": step["step"],
                "keyword": keyword,
                "status": "PASS" if result else "FAIL"
            })

            if not result:
                print(f"步骤 {step['step']} 失败: {keyword}")
                break

        return self.results

    def _execute_keyword(self, step):
        """执行单个关键字"""
        keyword = step["keyword"]
        method = getattr(self.keywords, keyword, None)

        if not method:
            print(f"未知关键字: {keyword}")
            return False

        try:
            # 根据关键字参数数量调用
            if keyword == "open_browser":
                return method(step["locator_type"] or "chrome")
            elif keyword == "navigate_to":
                return method(step["test_data"])
            elif keyword == "input_text":
                return method(
                    step["locator_type"],
                    step["locator_value"],
                    step["test_data"]
                )
            elif keyword == "click_element":
                return method(step["locator_type"], step["locator_value"])
            elif keyword == "verify_text":
                return method(
                    step["locator_type"],
                    step["locator_value"],
                    step["expected"]
                )
            elif keyword == "verify_page_title":
                return method(step["expected"])
            elif keyword == "close_browser":
                return method()
            else:
                return False
        except Exception as e:
            print(f"执行关键字 {keyword} 失败: {e}")
            return False

        </html>
        """
```

```python
import pymysql

# 1. 数据库连接类
class DatabaseHelper:
    """数据库辅助类"""

    def __init__(self, host, user, password, database, port=3306):
        self.connection = pymysql.connect(
            host=host,
            user=user,
            password=password,
            database=database,
            port=port,
            charset='utf8mb4',
            cursorclass=pymysql.cursors.DictCursor
        )

    def execute_query(self, query):
        """执行查询"""
        with self.connection.cursor() as cursor:
            cursor.execute(query)
            result = cursor.fetchall()
        return result

    def execute_update(self, query):
        """执行更新"""
        with self.connection.cursor() as cursor:
            cursor.execute(query)
            self.connection.commit()

    def close(self):
        """关闭连接"""
        self.connection.close()
```

```python
import requests

# 1. API 辅助类
class APIHelper:
    """API 辅助类"""

    def __init__(self, base_url):
        self.base_url = base_url
        self.session = requests.Session()

    def login_via_api(self, username, password):
        """通过 API 登录"""
        response = self.session.post(
            f"{self.base_url}/api/login",
            json={"username": username, "password": password}
        )
        assert response.status_code == 200
        return response.json()

    def create_user_via_api(self, user_data):
        """通过 API 创建用户"""
        response = self.session.post(
            f"{self.base_url}/api/users",
            json=user_data
        )
        return response.json()

    def get_cookies(self):
        """获取 Cookie"""
        return self.session.cookies.get_dict()
```

```python
from functools import wraps
import time

def retry_on_failure(max_attempts=3, delay=1, screenshot=True):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            driver = kwargs.get("driver") or (args[0] if args else None)

            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt < max_attempts - 1:
                        print(f"尝试 {attempt + 1} 失败: {e}")
                        if screenshot and driver:
                            take_screenshot(driver, func.__name__)
                        time.sleep(delay)
                    else:
                        raise
        return wrapper
    return decorator

@retry_on_failure(max_attempts=3, delay=2, screenshot=True)

    driver.get("https://example.com")
    assert "Example" in driver.title

class RetryableTest:
    """可重试的测试基类"""

    MAX_RETRIES = 3
    RETRY_DELAY = 2

    def run_with_retry(self, test_func, *args, **kwargs):
        """运行测试并支持重试"""
        for attempt in range(self.MAX_RETRIES):
            try:
                return test_func(*args, **kwargs)
            except Exception as e:
                if attempt < self.MAX_RETRIES - 1:
                    print(f"测试失败，{self.RETRY_DELAY}秒后重试...")
                    time.sleep(self.RETRY_DELAY)
                else:
                    raise

# 5. 智能重试（只重试特定异常）
from selenium.common.exceptions import TimeoutException, StaleElementReferenceException

def smart_retry(retryable_exceptions=(TimeoutException, StaleElementReferenceException)):
    """智能重试（只重试特定异常）"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            max_attempts = 3
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except retryable_exceptions as e:
                    if attempt < max_attempts - 1:
                        print(f"可重试的异常: {e.__class__.__name__}，正在重试...")
                        time.sleep(1)
                    else:
                        raise
                except Exception:
                    # 其他异常不重试
                    raise
        return wrapper
    return decorator
```

```python
import yaml
import os

class Environment:
    """环境管理类"""

    _current_env = None
    _config = None

    @classmethod
    def set_environment(cls, env_name):
        """设置当前环境"""
        cls._current_env = env_name
        cls._load_config()

    @classmethod
    def _load_config(cls):
        """加载配置"""
        with open("config/environments.yaml", "r") as f:
            all_config = yaml.safe_load(f)

        cls._config = all_config["environments"].get(cls._current_env)
        if not cls._config:
            raise ValueError(f"环境配置不存在: {cls._current_env}")

    @classmethod
    def get(cls, key, default=None):
        """获取配置项"""
        if cls._config is None:
            raise Exception("请先设置环境")

        keys = key.split(".")
        value = cls._config

        for k in keys:
            value = value.get(k)
            if value is None:
                return default

        return value

    @classmethod
    def get_current_env(cls):
        """获取当前环境名"""
        return cls._current_env

class CrossEnvironmentTest:
    """跨环境测试"""

    def __init__(self):
        self.environments = ["dev", "qa"]
        self.results = {}

    def run_on_all_environments(self, test_func):
        """在所有环境运行测试"""
        for env in self.environments:
            Environment.set_environment(env)
            driver = webdriver.Chrome()

            try:
                test_func(driver)
                self.results[env] = "PASS"
            except Exception as e:
                self.results[env] = f"FAIL: {e}"
            finally:
                driver.quit()

        return self.results
```
