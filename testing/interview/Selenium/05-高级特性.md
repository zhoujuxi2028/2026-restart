## 第 1 题：如何使用 Chrome DevTools Protocol (CDP) 进行高级操作？

**答案**：

CDP 允许直接与 Chrome 浏览器通信，实现 Selenium 原生不支持的功能。

```python
from selenium import webdriver
import json

driver = webdriver.Chrome()

# 1. 启用网络监控
driver.execute_cdp_cmd("Network.enable", {})

# 2. 拦截网络请求
driver.execute_cdp_cmd("Network.setRequestInterception", {
    "patterns": [{"urlPattern": "*"}]
})

# 3. 获取网络请求
logs = driver.get_log("performance")
for log in logs:
    message = json.loads(log["message"])["message"]
    if message["method"] == "Network.responseReceived":
        print(f"请求: {message['params']['response']['url']}")

# 4. 设置网络条件
driver.execute_cdp_cmd("Network.emulateNetworkConditions", {
    "offline": False,
    "downloadThroughput": 500 * 1024 / 8,  # 500 kbps
    "uploadThroughput": 500 * 1024 / 8,
    "latency": 100
})

# 5. 设置地理位置
driver.execute_cdp_cmd("Emulation.setGeolocationOverride", {
    "latitude": 37.774929,
    "longitude": -122.419418,
    "accuracy": 100
})

# 6. 设置设备方向
driver.execute_cdp_cmd("DeviceOrientation.setDeviceOrientationOverride", {
    "alpha": 0,
    "beta": 60,
    "gamma": 0
})

# 7. 启用 JavaScript 覆盖率
driver.execute_cdp_cmd("Profiler.enable", {})
driver.execute_cdp_cmd("Profiler.startPreciseCoverage", {
    "callCount": True,
    "detailed": True
})

# 访问页面
driver.get("https://example.com")

# 获取 JavaScript 覆盖率
coverage = driver.execute_cdp_cmd("Profiler.takePreciseCoverage", {})
print(f"JavaScript 覆盖率: {coverage}")

# 8. 截取全页面截图
def capture_full_page_screenshot(driver, filename):
    """使用 CDP 截取完整页面"""
    # 获取页面尺寸
    metrics = driver.execute_cdp_cmd("Page.getLayoutMetrics", {})
    width = metrics["contentSize"]["width"]
    height = metrics["contentSize"]["height"]

    # 设置视口并截图
    result = driver.execute_cdp_cmd("Page.captureScreenshot", {
        "format": "png",
        "clip": {
            "x": 0,
            "y": 0,
            "width": width,
            "height": height,
            "scale": 1
        }
    })

    # 保存截图
    import base64
    with open(filename, "wb") as f:
        f.write(base64.b64decode(result["data"]))

capture_full_page_screenshot(driver, "full_page.png")

# 9. 监控 Console 日志
driver.execute_cdp_cmd("Log.enable", {})
driver.get("https://example.com")

console_logs = driver.get_log("browser")
for log in console_logs:
    print(f"[{log['level']}] {log['message']}")

# 10. 注入 JavaScript
driver.execute_cdp_cmd("Page.addScriptToEvaluateOnNewDocument", {
    "source": """
        Object.defineProperty(navigator, 'webdriver', {
            get: () => undefined
        });
    """
})

# CDP 工具类
class CDPHelper:
    def __init__(self, driver):
        self.driver = driver

    def enable_network_monitoring(self):
        """启用网络监控"""
        self.driver.execute_cdp_cmd("Network.enable", {})

    def get_network_requests(self):
        """获取网络请求"""
        logs = self.driver.get_log("performance")
        requests = []

        for log in logs:
            message = json.loads(log["message"])["message"]
            if message["method"] == "Network.responseReceived":
                response = message["params"]["response"]
                requests.append({
                    "url": response["url"],
                    "status": response["status"],
                    "mimeType": response["mimeType"]
                })

        return requests

    def set_slow_network(self):
        """设置慢速网络"""
        self.driver.execute_cdp_cmd("Network.enable", {})
        self.driver.execute_cdp_cmd("Network.emulateNetworkConditions", {
            "offline": False,
            "downloadThroughput": 400 * 1024 / 8,
            "uploadThroughput": 400 * 1024 / 8,
            "latency": 400
        })

    def block_url_patterns(self, patterns):
        """阻止特定 URL 模式"""
        self.driver.execute_cdp_cmd("Network.enable", {})
        self.driver.execute_cdp_cmd("Network.setBlockedURLs", {
            "urls": patterns
        })

# 使用
cdp = CDPHelper(driver)
cdp.enable_network_monitoring()
cdp.block_url_patterns(["*analytics*", "*ads*"])
driver.get("https://example.com")
requests = cdp.get_network_requests()
print(f"网络请求: {len(requests)} 个")
```

**关键点**：
- CDP 只支持 Chrome/Chromium 浏览器
- 可以实现 Selenium 原生不支持的功能
- 常用于性能监控和高级调试

## 第 2 题：如何实现页面对象模式 (Page Object Model, POM)？

**答案**：

POM 是一种设计模式，将页面元素和操作封装到类中，提高代码可维护性。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# 1. 基础页面类
class BasePage:
    """所有页面的基类"""

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)

    def find_element(self, locator):
        """查找元素"""
        return self.wait.until(EC.presence_of_element_located(locator))

    def find_elements(self, locator):
        """查找多个元素"""
        return self.wait.until(EC.presence_of_all_elements_located(locator))

    def click(self, locator):
        """点击元素"""
        element = self.wait.until(EC.element_to_be_clickable(locator))
        element.click()

    def enter_text(self, locator, text):
        """输入文本"""
        element = self.find_element(locator)
        element.clear()
        element.send_keys(text)

    def get_text(self, locator):
        """获取文本"""
        return self.find_element(locator).text

    def is_visible(self, locator):
        """检查元素是否可见"""
        try:
            element = self.find_element(locator)
            return element.is_displayed()
        except:
            return False

# 2. 登录页面
class LoginPage(BasePage):
    """登录页面对象"""

    # 定位器
    USERNAME_INPUT = (By.ID, "username")
    PASSWORD_INPUT = (By.ID, "password")
    LOGIN_BUTTON = (By.ID, "login-submit")
    ERROR_MESSAGE = (By.CLASS_NAME, "error-message")

    def __init__(self, driver):
        super().__init__(driver)
        self.url = "https://example.com/login"

    def open(self):
        """打开登录页面"""
        self.driver.get(self.url)
        return self

    def enter_username(self, username):
        """输入用户名"""
        self.enter_text(self.USERNAME_INPUT, username)
        return self

    def enter_password(self, password):
        """输入密码"""
        self.enter_text(self.PASSWORD_INPUT, password)
        return self

    def click_login(self):
        """点击登录按钮"""
        self.click(self.LOGIN_BUTTON)
        return HomePage(self.driver)  # 返回主页对象

    def login(self, username, password):
        """完整登录流程"""
        self.enter_username(username)
        self.enter_password(password)
        return self.click_login()

    def get_error_message(self):
        """获取错误消息"""
        return self.get_text(self.ERROR_MESSAGE)

# 3. 主页
class HomePage(BasePage):
    """主页对象"""

    WELCOME_MESSAGE = (By.ID, "welcome")
    LOGOUT_BUTTON = (By.ID, "logout")
    PROFILE_LINK = (By.LINK_TEXT, "个人资料")

    def get_welcome_message(self):
        """获取欢迎消息"""
        return self.get_text(self.WELCOME_MESSAGE)

    def logout(self):
        """登出"""
        self.click(self.LOGOUT_BUTTON)
        return LoginPage(self.driver)

    def go_to_profile(self):
        """进入个人资料页"""
        self.click(self.PROFILE_LINK)
        return ProfilePage(self.driver)

# 4. 个人资料页
class ProfilePage(BasePage):
    """个人资料页"""

    NAME_INPUT = (By.ID, "name")
    EMAIL_INPUT = (By.ID, "email")
    SAVE_BUTTON = (By.ID, "save")
    SUCCESS_MESSAGE = (By.CLASS_NAME, "success-message")

    def update_profile(self, name, email):
        """更新个人资料"""
        self.enter_text(self.NAME_INPUT, name)
        self.enter_text(self.EMAIL_INPUT, email)
        self.click(self.SAVE_BUTTON)

    def get_success_message(self):
        """获取成功消息"""
        return self.get_text(self.SUCCESS_MESSAGE)

# 5. 测试使用
driver = webdriver.Chrome()

# 链式调用
login_page = LoginPage(driver)
home_page = login_page.open().login("testuser", "password123")

assert "欢迎" in home_page.get_welcome_message()

profile_page = home_page.go_to_profile()
profile_page.update_profile("新名字", "new@example.com")

assert "保存成功" in profile_page.get_success_message()

driver.quit()

# 6. 高级特性：组件对象
class HeaderComponent(BasePage):
    """页面头部组件"""

    SEARCH_BOX = (By.ID, "search")
    USER_MENU = (By.ID, "user-menu")
    NOTIFICATIONS = (By.ID, "notifications")

    def search(self, keyword):
        """搜索"""
        self.enter_text(self.SEARCH_BOX, keyword)
        from selenium.webdriver.common.keys import Keys
        self.find_element(self.SEARCH_BOX).send_keys(Keys.ENTER)

    def open_user_menu(self):
        """打开用户菜单"""
        self.click(self.USER_MENU)

# 7. 使用组件
class PageWithHeader(BasePage):
    """带头部的页面"""

    def __init__(self, driver):
        super().__init__(driver)
        self.header = HeaderComponent(driver)

# 实际使用
page = PageWithHeader(driver)
page.header.search("Selenium")
```

```python
def read_excel_data(filename, sheet_name):
    """从 Excel 读取数据"""
    workbook = openpyxl.load_workbook(filename)
    sheet = workbook[sheet_name]

    data = []
    headers = [cell.value for cell in sheet[1]]

    for row in sheet.iter_rows(min_row=2, values_only=True):
        data.append(dict(zip(headers, row)))

    return data

# 方法 6：数据驱动框架
class DataProvider:
    """数据提供者"""

    @staticmethod
    def load_csv(filename):
        """加载 CSV"""
        with open(filename, "r", encoding="utf-8") as f:
            return list(csv.DictReader(f))

    @staticmethod
    def load_json(filename):
        """加载 JSON"""
        with open(filename, "r", encoding="utf-8") as f:
            return json.load(f)

    @staticmethod
    def load_excel(filename, sheet_name):
        """加载 Excel"""
        return read_excel_data(filename, sheet_name)

class DataDrivenTest:
    """数据驱动测试基类"""

    def __init__(self, driver):
        self.driver = driver

    def run_tests(self, test_data, test_func):
        """运行数据驱动测试"""
        results = []

        for i, data in enumerate(test_data, 1):
            print(f"执行测试用例 {i}/{len(test_data)}")
            try:
                test_func(data)
                results.append({"case": i, "data": data, "status": "PASS"})
            except AssertionError as e:
                results.append({"case": i, "data": data, "status": "FAIL", "error": str(e)})

        return results

# 使用
```

```python
# 方法 3：自定义测试报告
class TestReport:
    """自定义测试报告"""

    def __init__(self, report_name="test_report"):
        self.report_name = report_name
        self.test_results = []
        self.start_time = datetime.now()

    def add_test_result(self, test_name, status, duration, screenshot=None):
        """添加测试结果"""
        self.test_results.append({
            "test_name": test_name,
            "status": status,
            "duration": duration,
            "screenshot": screenshot,
            "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        })

    def generate_html_report(self):
        """生成 HTML 报告"""
        html_template = """
        <!DOCTYPE html>
        <html>
        <head>
            <title>{title}</title>
            <style>
                body {{ font-family: Arial, sans-serif; }}
                table {{ border-collapse: collapse; width: 100%; }}
                th, td {{ border: 1px solid #ddd; padding: 8px; text-align: left; }}
                th {{ background-color: #4CAF50; color: white; }}
                .pass {{ color: green; }}
                .fail {{ color: red; }}
            </style>
        </head>
        <body>
            <h1>测试报告</h1>
            <p>执行时间: {start_time}</p>
            <p>总用例数: {total}, 通过: {passed}, 失败: {failed}</p>
            <table>
                <tr>
                    <th>用例名称</th>
                    <th>状态</th>
                    <th>耗时</th>
                    <th>时间戳</th>
                </tr>
                {test_rows}
            </table>
        </body>
        </html>
        """

        passed = sum(1 for r in self.test_results if r["status"] == "PASS")
        failed = len(self.test_results) - passed

        test_rows = ""
        for result in self.test_results:
            status_class = "pass" if result["status"] == "PASS" else "fail"
            test_rows += f"""
                <tr>
                    <td>{result['test_name']}</td>
                    <td class="{status_class}">{result['status']}</td>
                    <td>{result['duration']:.2f}s</td>
                    <td>{result['timestamp']}</td>
                </tr>
            """

        html_content = html_template.format(
            title=self.report_name,
            start_time=self.start_time.strftime("%Y-%m-%d %H:%M:%S"),
            total=len(self.test_results),
            passed=passed,
            failed=failed,
            test_rows=test_rows
        )

        filename = f"{self.report_name}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.html"
        with open(filename, "w", encoding="utf-8") as f:
            f.write(html_content)

        return filename

# conftest.py
@pytest.hookimpl(tryfirst=True, hookwrapper=True)
def pytest_runtest_makereport(item, call):
    """失败时截图"""
    outcome = yield
    rep = outcome.get_result()

    if rep.when == "call" and rep.failed:
        driver = item.funcargs.get("driver")
        if driver:
            screenshot_dir = "screenshots"
            os.makedirs(screenshot_dir, exist_ok=True)
            screenshot_path = os.path.join(screenshot_dir, f"{item.name}.png")
            driver.save_screenshot(screenshot_path)

            # 附加到 Allure 报告
            allure.attach.file(screenshot_path, name="失败截图", attachment_type=allure.attachment_type.PNG)
```

```python
# 并发测试框架
class ParallelTestRunner:
    """并行测试运行器"""

    def __init__(self, max_workers=4):
        self.max_workers = max_workers
        self.results = []

    def run_test_case(self, test_func, test_data):
        """运行单个测试"""
        driver = webdriver.Chrome()
        try:
            result = test_func(driver, test_data)
            self.results.append({"data": test_data, "status": "PASS", "result": result})
        except Exception as e:
            self.results.append({"data": test_data, "status": "FAIL", "error": str(e)})
        finally:
            driver.quit()

    def run_parallel(self, test_func, test_data_list):
        """并行运行所有测试"""
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [executor.submit(self.run_test_case, test_func, data) for data in test_data_list]
            for future in futures:
                future.result()

        return self.results

    driver.find_element(By.ID, "username").send_keys(data["username"])
    driver.find_element(By.ID, "password").send_keys(data["password"])
    driver.find_element(By.ID, "submit").click()
    return "欢迎" in driver.page_source

runner = ParallelTestRunner(max_workers=5)
test_data = [
    {"username": "user1", "password": "pass1"},
    {"username": "user2", "password": "pass2"},
    {"username": "user3", "password": "pass3"},
]
```

| 方法 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| `pytest-xdist` | 简单易用 | 依赖 pytest | pytest 项目 |
| `threading` | Python 标准库 | GIL 限制 | I/O 密集型 |
| `multiprocessing` | 真正并行 | 资源消耗大 | CPU 密集型 |
| `Selenium Grid` | 跨机器分布式 | 配置复杂 | 大规模测试 |

**关键点**：
- 并发测试需要确保每个测试独立
- pytest-xdist 是最简单的方案
- Selenium Grid 适合大规模分布式测试
- 注意资源限制（浏览器实例数量）

## 第 10 题：如何实现测试失败自动重试？

**答案**：

```python
import pytest
from selenium import webdriver

# 方法 1：使用 pytest-rerunfailures
# 安装: pip install pytest-rerunfailures
# 运行: pytest --reruns 3 --reruns-delay 2

@pytest.mark.flaky(reruns=3, reruns_delay=2)
def test_with_retry():
    """失败后自动重试 3 次，间隔 2 秒"""
    driver = webdriver.Chrome()
    driver.get("https://example.com")
    assert "Example" in driver.title
    driver.quit()

# 方法 2：自定义重试装饰器
import time
from functools import wraps

def retry(max_attempts=3, delay=1):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt < max_attempts - 1:
                        print(f"测试失败，{delay}秒后重试 ({attempt + 1}/{max_attempts})")
                        time.sleep(delay)
                    else:
                        raise
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2)
def test_with_custom_retry():
    driver = webdriver.Chrome()
    driver.get("https://example.com")
    assert "Example" in driver.title
    driver.quit()
```

**关键点**：
- pytest-rerunfailures 最简单
- 自定义装饰器提供更多控制
- 避免过度重试掩盖真实问题
