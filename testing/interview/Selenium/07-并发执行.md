**答案**：

```bash

# 2. 启动 Node

```

**测试代码**：

```python
from concurrent.futures import ThreadPoolExecutor

def test_on_browser(browser):
    """在指定浏览器测试"""
    capabilities = {
        "chrome": DesiredCapabilities.CHROME,
        "firefox": DesiredCapabilities.FIREFOX,
        "edge": DesiredCapabilities.EDGE
    }

    driver = webdriver.Remote(
        command_executor='http://localhost:4444/wd/hub',
        desired_capabilities=capabilities[browser]
    )

    driver.get("https://example.com")
    title = driver.title
    driver.quit()

    return {"browser": browser, "title": title}

# 并行执行
browsers = ["chrome", "firefox", "edge"]
with ThreadPoolExecutor(max_workers=3) as executor:
    results = executor.map(test_on_browser, browsers)

for result in results:
    print(f"{result['browser']}: {result['title']}")
```

```python
# 方法 5：Redis 分布式锁
import redis
import time

class DistributedLock:
    """Redis 分布式锁"""

    def __init__(self, redis_client, key, timeout=10):
        self.redis_client = redis_client
        self.key = f"lock:{key}"
        self.timeout = timeout
        self.identifier = str(uuid.uuid4())

    def acquire(self):
        """获取锁"""
        end_time = time.time() + self.timeout
        while time.time() < end_time:
            if self.redis_client.set(self.key, self.identifier, nx=True, ex=self.timeout):
                return True
            time.sleep(0.001)
        return False

    def release(self):
        """释放锁"""
        pipe = self.redis_client.pipeline(True)
        while True:
            try:
                pipe.watch(self.key)
                if pipe.get(self.key) == self.identifier.encode():
                    pipe.multi()
                    pipe.delete(self.key)
                    pipe.execute()
                    return True
                pipe.unwatch()
                break
            except redis.WatchError:
                pass
        return False

# 使用分布式锁保护共享资源
redis_client = redis.Redis(host='localhost', port=6379, db=0)

def test_with_distributed_lock(driver):
    """使用分布式锁"""
    lock = DistributedLock(redis_client, "test_resource")

    if lock.acquire():
        try:
            # 操作共享资源
            driver.get("https://example.com/shared")
            # ...
        finally:
            lock.release()
```

**数据隔离策略对比**：

| 策略 | 优点 | 缺点 | 适用场景 |
|------|------|------|---------|
| **唯一标识符** | 简单易用 | 可能产生大量数据 | 注册、创建类测试 |
| **Worker 分区** | 资源隔离 | 需要预先准备 | 数据库密集型测试 |
| **分布式锁** | 保证互斥 | 性能开销 | 共享资源操作 |
| **临时工作空间** | 文件隔离 | 清理麻烦 | 文件上传下载测试 |

**关键点**：
- 唯一标识符最简单有效
- 数据库分区适合大规模测试
- 分布式锁适合共享资源保护
- 每个测试执行后清理数据

## 第 5 题：如何监控和管理并发测试的性能？

**答案**：

```python
import time
import threading
from datetime import datetime
from concurrent.futures import ThreadPoolExecutor, as_completed

# 1. 性能监控类
class PerformanceMonitor:
    """性能监控器"""

    def __init__(self):
        self.start_time = None
        self.end_time = None
        self.test_results = []
        self.lock = threading.Lock()

    def start(self):
        """开始监控"""
        self.start_time = time.time()

    def end(self):
        """结束监控"""
        self.end_time = time.time()

    def record_test(self, test_name, duration, status):
        """记录测试结果"""
        with self.lock:
            self.test_results.append({
                "test_name": test_name,
                "duration": duration,
                "status": status,
                "timestamp": datetime.now().isoformat()
            })

    def get_statistics(self):
        """获取统计信息"""
        if not self.test_results:
            return {}

        total_duration = self.end_time - self.start_time if self.end_time else 0
        test_durations = [r["duration"] for r in self.test_results]
        passed = sum(1 for r in self.test_results if r["status"] == "PASS")
        failed = len(self.test_results) - passed

        return {
            "total_tests": len(self.test_results),
            "total_duration": total_duration,
            "passed": passed,
            "failed": failed,
            "avg_duration": sum(test_durations) / len(test_durations),
            "min_duration": min(test_durations),
            "max_duration": max(test_durations)
        }

# 2. 使用性能监控
from selenium import webdriver

monitor = PerformanceMonitor()
monitor.start()

import psutil

class ResourceMonitor:
    """资源使用监控"""

    def __init__(self, interval=1):
        self.interval = interval
        self.monitoring = False
        self.cpu_usage = []
        self.memory_usage = []
        self.thread = None

    def start(self):
        """开始监控"""
        self.monitoring = True
        self.thread = threading.Thread(target=self._monitor)
        self.thread.start()

    def stop(self):
        """停止监控"""
        self.monitoring = False
        if self.thread:
            self.thread.join()

    def _monitor(self):
        """监控循环"""
        while self.monitoring:
            self.cpu_usage.append(psutil.cpu_percent(interval=1))
            self.memory_usage.append(psutil.virtual_memory().percent)
            time.sleep(self.interval)

    def get_report(self):
        """获取监控报告"""
        if not self.cpu_usage:
            return {}

        return {
            "avg_cpu": sum(self.cpu_usage) / len(self.cpu_usage),
            "max_cpu": max(self.cpu_usage),
            "avg_memory": sum(self.memory_usage) / len(self.memory_usage),
            "max_memory": max(self.memory_usage)
        }

# 4. 并发控制器
from threading import Semaphore

class ConcurrencyController:
    """并发控制器"""

    def __init__(self, max_concurrent=5):
        self.semaphore = Semaphore(max_concurrent)
        self.active_tests = 0
        self.lock = threading.Lock()

    def acquire(self):
        """获取执行权限"""
        self.semaphore.acquire()
        with self.lock:
            self.active_tests += 1

    def release(self):
        """释放执行权限"""
        self.semaphore.release()
        with self.lock:
            self.active_tests -= 1

    def get_active_count(self):
        """获取活跃测试数"""
        with self.lock:
            return self.active_tests

# 使用并发控制
controller = ConcurrencyController(max_concurrent=3)

def controlled_test(test_id):
    """受控的测试"""
    controller.acquire()
    try:
        print(f"测试 {test_id} 开始，活跃数: {controller.get_active_count()}")
        driver = webdriver.Chrome()
        driver.get("https://example.com")
        time.sleep(2)  # 模拟测试
        driver.quit()
    finally:
        controller.release()

threads = []
for i in range(10):
    t = threading.Thread(target=controlled_test, args=(i,))
    threads.append(t)
    t.start()

    t.join()
```

## 第 6 题：如何实现动态扩展的并发测试？

**答案**：

```python
import queue
import threading
from selenium import webdriver

# 1. 动态线程池
class DynamicThreadPool:
    """动态线程池"""

    def __init__(self, min_workers=2, max_workers=10):
        self.min_workers = min_workers
        self.max_workers = max_workers
        self.task_queue = queue.Queue()
        self.workers = []
        self.active_workers = 0
        self.lock = threading.Lock()

        # 启动最小工作线程
        for _ in range(min_workers):
            self._add_worker()

    def _add_worker(self):
        """添加工作线程"""
        worker = threading.Thread(target=self._worker)
        worker.daemon = True
        worker.start()
        self.workers.append(worker)

    def _worker(self):
        """工作线程"""
        while True:
            try:
                task = self.task_queue.get(timeout=5)
                if task is None:
                    break

                with self.lock:
                    self.active_workers += 1

                try:
                    task()
                finally:
                    with self.lock:
                        self.active_workers -= 1
                    self.task_queue.task_done()
            except queue.Empty:
                # 队列空闲，考虑减少工作线程
                with self.lock:
                    if len(self.workers) > self.min_workers:
                        self.workers.remove(threading.current_thread())
                        break

    def submit(self, task):
        """提交任务"""
        self.task_queue.put(task)

        # 动态扩展
        with self.lock:
            if (self.task_queue.qsize() > len(self.workers) and
                len(self.workers) < self.max_workers):
                self._add_worker()

    def shutdown(self):
        """关闭线程池"""
        self.task_queue.join()
        for _ in self.workers:
            self.task_queue.put(None)
        for worker in self.workers:
            worker.join()

# 使用动态线程池
pool = DynamicThreadPool(min_workers=2, max_workers=8)

def test_task(test_id):
    driver = webdriver.Chrome()
    driver.get("https://example.com")
    print(f"测试 {test_id}: {driver.title}")
    driver.quit()

# 提交任务

# 2. 基于负载的自适应并发
class AdaptiveConcurrency:
    """自适应并发控制"""

    def __init__(self, initial_workers=4, max_workers=20):
        self.current_workers = initial_workers
        self.max_workers = max_workers
        self.avg_duration = 0
        self.test_count = 0

    def adjust_workers(self, duration):
        """根据测试耗时调整并发数"""
        self.test_count += 1
        self.avg_duration = (
            (self.avg_duration * (self.test_count - 1) + duration) /
            self.test_count
        )

        # 如果平均耗时较长，增加并发
        if self.avg_duration > 5 and self.current_workers < self.max_workers:
            self.current_workers += 1
            print(f"增加并发数到 {self.current_workers}")

        # 如果平均耗时较短，减少并发
        elif self.avg_duration < 2 and self.current_workers > 1:
            self.current_workers -= 1
            print(f"减少并发数到 {self.current_workers}")

        return self.current_workers

# 使用自适应并发
adaptive = AdaptiveConcurrency(initial_workers=4)

def adaptive_test(test_id):
    start = time.time()
    driver = webdriver.Chrome()
    driver.get("https://example.com")
    driver.quit()
    duration = time.time() - start

    new_workers = adaptive.adjust_workers(duration)
    print(f"测试 {test_id} 耗时: {duration:.2f}秒，当前并发: {new_workers}")

# 3. 云端弹性扩展（BrowserStack/Sauce Labs）
class CloudTestExecutor:
    """云端测试执行器"""

    def __init__(self, api_key, max_parallel=10):
        self.api_key = api_key
        self.max_parallel = max_parallel

    def create_driver(self):
        """创建云端 driver"""
        from selenium.webdriver.common.desired_capabilities import DesiredCapabilities

        caps = DesiredCapabilities.CHROME.copy()
        caps['browserstack.user'] = "username"
        caps['browserstack.key'] = self.api_key

        driver = webdriver.Remote(
            command_executor='https://hub-cloud.browserstack.com/wd/hub',
            desired_capabilities=caps
        )
        return driver

    def run_tests_parallel(self, test_functions):
        """并行运行测试"""
        with ThreadPoolExecutor(max_workers=self.max_parallel) as executor:
            futures = [
                executor.submit(self._run_test, func)
                for func in test_functions
            ]
            results = [f.result() for f in as_completed(futures)]
        return results

    def _run_test(self, test_func):
        """运行单个测试"""
        driver = self.create_driver()
        try:
            test_func(driver)
            return "PASS"
        except Exception as e:
            return f"FAIL: {e}"
        finally:
            driver.quit()

# 使用云端执行器
cloud_executor = CloudTestExecutor(api_key="your_api_key", max_parallel=10)

def test_example(driver):
    driver.get("https://example.com")
    assert "Example" in driver.title

test_functions = [test_example] * 20
results = cloud_executor.run_tests_parallel(test_functions)
print(f"测试结果: {results.count('PASS')} 通过, {len(results) - results.count('PASS')} 失败")
```
