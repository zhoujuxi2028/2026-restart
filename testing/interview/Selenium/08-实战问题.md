## 第 1 题：如何处理动态加载和无限滚动的页面？

**答案**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import time

driver = webdriver.Chrome()
driver.get("https://example.com/infinite-scroll")

# 方法 1：滚动加载所有内容
def load_all_content(driver, max_scrolls=50):
    """加载所有内容"""
    last_height = driver.execute_script("return document.body.scrollHeight")
    scroll_count = 0

    while scroll_count < max_scrolls:
        # 滚动到底部
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

        # 等待新内容加载
        time.sleep(2)

        # 检查是否有新内容
        new_height = driver.execute_script("return document.body.scrollHeight")
        if new_height == last_height:
            # 没有新内容了
            break

        last_height = new_height
        scroll_count += 1

    print(f"共滚动 {scroll_count} 次")
    return scroll_count

load_all_content(driver)

# 方法 2：滚动直到特定元素出现
def scroll_until_element(driver, target_locator, max_attempts=20):
    """滚动直到找到目标元素"""
    for attempt in range(max_attempts):
        try:
            element = driver.find_element(*target_locator)
            if element.is_displayed():
                # 找到目标元素
                driver.execute_script("arguments[0].scrollIntoView();", element)
                return element
        except:
            pass

        # 继续滚动
        driver.execute_script("window.scrollBy(0, 500);")
        time.sleep(1)

    raise Exception(f"未找到元素: {target_locator}")

target = scroll_until_element(driver, (By.ID, "item-100"))

# 方法 3：监控元素数量变化
def wait_for_items_loaded(driver, item_locator, expected_count, timeout=60):
    """等待特定数量的元素加载"""
    end_time = time.time() + timeout

    while time.time() < end_time:
        items = driver.find_elements(*item_locator)
        current_count = len(items)
        print(f"当前加载: {current_count}/{expected_count}")

        if current_count >= expected_count:
            return items

        # 滚动触发加载
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
        time.sleep(2)

    raise TimeoutException(f"超时：未加载到 {expected_count} 个元素")

items = wait_for_items_loaded(driver, (By.CLASS_NAME, "item"), expected_count=100)
print(f"共加载 {len(items)} 个元素")

# 方法 4：检测加载指示器
def wait_for_loading_complete(driver):
    """等待加载指示器消失"""
    wait = WebDriverWait(driver, 30)

    while True:
        # 滚动触发加载
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

        # 等待加载指示器出现
        try:
            wait.until(EC.presence_of_element_located((By.CLASS_NAME, "loading")))
            # 等待加载指示器消失
            wait.until(EC.invisibility_of_element_located((By.CLASS_NAME, "loading")))
        except:
            # 没有加载指示器了，说明加载完成
            break

        time.sleep(1)

wait_for_loading_complete(driver)

# 方法 5：虚拟滚动（只加载可见区域）
class VirtualScrollHandler:
    """虚拟滚动处理器"""

    def __init__(self, driver):
        self.driver = driver

    def get_visible_items(self, item_locator):
        """获取当前可见的元素"""
        all_items = self.driver.find_elements(*item_locator)
        visible_items = []

        for item in all_items:
            if self._is_in_viewport(item):
                visible_items.append(item)

        return visible_items

    def _is_in_viewport(self, element):
        """检查元素是否在视口中"""
        return self.driver.execute_script("""
            var elem = arguments[0];
            var rect = elem.getBoundingClientRect();
            return (
                rect.top >= 0 &&
                rect.left >= 0 &&
                rect.bottom <= (window.innerHeight || document.documentElement.clientHeight) &&
                rect.right <= (window.innerWidth || document.documentElement.clientWidth)
            );
        """, element)

    def scroll_to_index(self, item_locator, index):
        """滚动到特定索引的元素"""
        items = self.driver.find_elements(*item_locator)
        if index < len(items):
            self.driver.execute_script("arguments[0].scrollIntoView();", items[index])
            return items[index]
        return None

# 使用虚拟滚动
handler = VirtualScrollHandler(driver)
visible_items = handler.get_visible_items((By.CLASS_NAME, "item"))
print(f"可见元素数: {len(visible_items)}")

# 滚动到第 50 个元素
item_50 = handler.scroll_to_index((By.CLASS_NAME, "item"), 50)

# 方法 6：完整的无限滚动测试
class InfiniteScrollTest:
    """无限滚动测试类"""

    def __init__(self, driver, item_locator):
        self.driver = driver
        self.item_locator = item_locator
        self.loaded_items = set()

    def load_all(self, max_scrolls=100, scroll_pause=2):
        """加载所有内容"""
        last_count = 0
        no_change_count = 0

        for scroll in range(max_scrolls):
            # 滚动
            self.driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
            time.sleep(scroll_pause)

            # 获取当前元素
            items = self.driver.find_elements(*self.item_locator)
            current_count = len(items)

            # 记录新加载的元素
            for item in items[last_count:]:
                item_id = item.get_attribute("id") or item.text
                self.loaded_items.add(item_id)

            print(f"滚动 {scroll + 1}: 当前 {current_count} 个元素")

            # 检查是否有新内容
            if current_count == last_count:
                no_change_count += 1
                if no_change_count >= 3:
                    print("连续 3 次滚动无新内容，停止加载")
                    break
            else:
                no_change_count = 0

            last_count = current_count

        return len(self.loaded_items)

    def verify_items(self, expected_count):
        """验证加载的元素数量"""
        actual_count = len(self.loaded_items)
        assert actual_count >= expected_count, \
            f"加载的元素数量不足: {actual_count} < {expected_count}"

# 使用
test = InfiniteScrollTest(driver, (By.CLASS_NAME, "item"))
loaded_count = test.load_all()
print(f"共加载 {loaded_count} 个唯一元素")
test.verify_items(expected_count=100)

driver.quit()
```

**关键点**：
- 无限滚动需要检测页面高度变化
- 设置最大滚动次数避免无限循环
- 等待加载指示器消失
- 记录已加载元素避免重复

def create_stealth_driver():
    """创建隐秘的 driver"""
    options = Options()

    # 隐藏自动化
    options.add_argument('--disable-blink-features=AutomationControlled')
    options.add_experimental_option("excludeSwitches", ["enable-automation"])
    options.add_experimental_option('useAutomationExtension', False)

    # 窗口大小
    options.add_argument('--window-size=1920,1080')

    # User-Agent
    options.add_argument(f'user-agent={user_agent}')

    # 其他选项
    options.add_argument('--disable-dev-shm-usage')
    options.add_argument('--no-sandbox')

    driver = webdriver.Chrome(options=options)

    # CDP 命令
    driver.execute_cdp_cmd('Page.addScriptToEvaluateOnNewDocument', {
        'source': '''
            // 隐藏 webdriver
            Object.defineProperty(navigator, 'webdriver', {
                get: () => undefined
            });

            // 修改 plugins
            Object.defineProperty(navigator, 'plugins', {
                get: () => [1, 2, 3, 4, 5]
            });

            // 修改 languages
            Object.defineProperty(navigator, 'languages', {
                get: () => ['zh-CN', 'zh', 'en-US', 'en']
            });

            // 修改 permissions
            const originalQuery = window.navigator.permissions.query;
            window.navigator.permissions.query = (parameters) => (
                parameters.name === 'notifications' ?
                    Promise.resolve({ state: Notification.permission }) :
                    originalQuery(parameters)
            );
        '''
    })

    return driver

class CrossDomainCookieManager:
    """跨域 Cookie 管理器"""

    def __init__(self, driver):
        self.driver = driver
        self.cookies = {}

    def save_cookies_for_domain(self, domain):
        """保存当前域名的 Cookie"""
        self.cookies[domain] = self.driver.get_cookies()

    def load_cookies_for_domain(self, domain):
        """加载指定域名的 Cookie"""
        if domain in self.cookies:
            # 先访问域名
            self.driver.get(f"https://{domain}")
            # 添加 Cookie
            for cookie in self.cookies[domain]:
                self.driver.add_cookie(cookie)
            self.driver.refresh()

# 场景 1：等待文件下载完成
def wait_for_download_complete(directory, filename, timeout=60):
    """等待文件下载完成"""
    end_time = time.time() + timeout
    filepath = os.path.join(directory, filename)

    while time.time() < end_time:
        # 检查文件是否存在
        if os.path.exists(filepath):
            # 检查是否还在下载（Chrome 临时文件以 .crdownload 结尾）
            temp_file = filepath + ".crdownload"
            if not os.path.exists(temp_file):
                # 文件已完整下载
                return filepath

        time.sleep(1)

    raise TimeoutException(f"文件下载超时: {filename}")

# 场景 2：下载动态文件名
def wait_for_any_download(directory, timeout=60):
    """等待任意文件下载（不知道文件名）"""
    # 记录下载前的文件列表
    before = set(os.listdir(directory))

    end_time = time.time() + timeout

    while time.time() < end_time:
        # 当前文件列表
        after = set(os.listdir(directory))
        # 新增的文件
        new_files = after - before

        # 过滤掉临时文件
        complete_files = [
            f for f in new_files
            if not f.endswith('.crdownload') and not f.endswith('.tmp')
        ]

        if complete_files:
            # 返回第一个完整下载的文件
            return os.path.join(directory, complete_files[0])

        time.sleep(1)

    raise TimeoutException("未检测到文件下载")

import requests

def download_via_requests(driver, download_url):
    """通过 requests 下载（可以使用浏览器的 Cookie）"""
    # 获取浏览器的 Cookie
    cookies = driver.get_cookies()
    session = requests.Session()

    for cookie in cookies:
        session.cookies.set(cookie['name'], cookie['value'])

    # 下载文件
    response = session.get(download_url, stream=True)
    filename = download_url.split('/')[-1]
    filepath = os.path.join(download_dir, filename)

    with open(filepath, 'wb') as f:
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)

    return filepath

class PageObjectRegistry:
    """页面对象注册中心"""

    _pages = {}

    @classmethod
    def register(cls, name):
        """注册页面对象"""
        def decorator(page_class):
            cls._pages[name] = page_class
            return page_class
        return decorator

    @classmethod
    def get_page(cls, name, driver):
        """获取页面对象实例"""
        page_class = cls._pages.get(name)
        if not page_class:
            raise ValueError(f"未注册的页面: {name}")
        return page_class(driver)

    def __init__(self):
        self.failures = []

    def analyze_failure(self, test_name, error_message, screenshot_path):
        """分析失败原因"""
        # 常见失败模式
        if "TimeoutException" in error_message:
            suggestion = "元素加载超时，考虑增加等待时间"
        elif "NoSuchElementException" in error_message:
            suggestion = "元素定位失败，检查定位器是否正确"
        elif "StaleElementReferenceException" in error_message:
            suggestion = "元素已过时，需要重新定位"
        else:
            suggestion = "未知错误，请查看截图"

        self.failures.append({
            "test": test_name,
            "error": error_message,
            "suggestion": suggestion,
            "screenshot": screenshot_path
        })

    def generate_report(self):
        """生成失败报告"""
        report = "# 测试失败分析报告\n\n"
        for failure in self.failures:

            report += f"**建议**: {failure['suggestion']}\n"
            report += f"**截图**: {failure['screenshot']}\n\n"

# 8. 定期测试审计
class TestAudit:
    """测试审计"""

    def __init__(self):
        self.test_inventory = {}

    def scan_tests(self, test_directory):
        """扫描所有测试"""
        import ast
        import os

        for root, dirs, files in os.walk(test_directory):
            for file in files:
                if file.startswith("test_") and file.endswith(".py"):
                    filepath = os.path.join(root, file)
                    self._analyze_test_file(filepath)

    def _analyze_test_file(self, filepath):
        """分析测试文件"""
        with open(filepath, "r", encoding="utf-8") as f:
            tree = ast.parse(f.read())

        test_functions = [
            node.name for node in ast.walk(tree)
            if isinstance(node, ast.FunctionDef) and node.name.startswith("test_")
        ]

        self.test_inventory[filepath] = {
            "test_count": len(test_functions),
            "tests": test_functions
        }

    def find_unused_tests(self, execution_history):
        """查找长期未执行的测试"""
        unused_tests = []

        for filepath, info in self.test_inventory.items():
            for test_name in info["tests"]:
                full_name = f"{filepath}::{test_name}"
                if full_name not in execution_history:
                    unused_tests.append(full_name)

        return unused_tests

            "7. 检查测试覆盖率是否下降"
        ]

    @staticmethod
    def calculate_test_debt():
        """计算技术债务"""
        metrics = {
            "total_tests": 500,
            "failing_tests": 50,
            "skipped_tests": 30,
            "slow_tests": 20,  # 执行时间 > 1分钟
            "flaky_tests": 10   # 间歇性失败
        }

        debt_score = (
            metrics["failing_tests"] * 3 +
            metrics["skipped_tests"] * 2 +
            metrics["slow_tests"] * 1 +
            metrics["flaky_tests"] * 5
        )

        return {
            "score": debt_score,
            "level": "HIGH" if debt_score > 200 else "MEDIUM" if debt_score > 100 else "LOW",
            "metrics": metrics
        }
