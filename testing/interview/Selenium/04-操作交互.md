## 第 1 题：如何使用 ActionChains 执行复杂的鼠标和键盘操作？

**答案**：

`ActionChains` 用于执行链式的复杂操作，所有操作会被缓存，调用 `perform()` 后才真正执行。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver import ActionChains
from selenium.webdriver.common.keys import Keys

driver = webdriver.Chrome()
driver.get("https://example.com")

# 创建 ActionChains 对象
actions = ActionChains(driver)

# 1. 鼠标悬停（Hover）
element = driver.find_element(By.ID, "menu")
actions.move_to_element(element).perform()

# 2. 点击
element = driver.find_element(By.ID, "button")
actions.click(element).perform()

# 3. 右键点击
actions.context_click(element).perform()

# 4. 双击
actions.double_click(element).perform()

# 5. 拖拽元素
source = driver.find_element(By.ID, "draggable")
target = driver.find_element(By.ID, "droppable")
actions.drag_and_drop(source, target).perform()

# 6. 按坐标拖拽
actions.drag_and_drop_by_offset(source, xoffset=100, yoffset=50).perform()

# 7. 鼠标移动到指定坐标
actions.move_by_offset(100, 200).perform()

# 8. 鼠标移动到元素偏移位置
actions.move_to_element_with_offset(element, 10, 20).perform()

# 9. 按下鼠标左键（不释放）
actions.click_and_hold(element).perform()

# 10. 释放鼠标左键
actions.release(element).perform()

# 11. 键盘输入
input_field = driver.find_element(By.ID, "search")
actions.send_keys_to_element(input_field, "selenium").perform()

# 12. 按下 Ctrl+A（全选）
actions.key_down(Keys.CONTROL).send_keys('a').key_up(Keys.CONTROL).perform()

# 13. 链式操作
actions.move_to_element(element)\
       .click()\
       .send_keys("test")\
       .key_down(Keys.ENTER)\
       .perform()

# 14. 复杂拖拽示例
actions.click_and_hold(source)\
       .move_to_element(target)\
       .release()\
       .perform()

# 15. 绘制签名示例
canvas = driver.find_element(By.ID, "signature-pad")
actions.move_to_element_with_offset(canvas, 10, 10)\
       .click_and_hold()\
       .move_by_offset(50, 0)\
       .move_by_offset(0, 50)\
       .move_by_offset(-50, 0)\
       .release()\
       .perform()
```

**ActionChains 方法对照表**：

| 方法 | 说明 | 使用场景 |
|------|------|---------|
| `click()` | 点击 | 普通点击操作 |
| `click_and_hold()` | 按住鼠标左键 | 拖拽操作的开始 |
| `context_click()` | 右键点击 | 显示右键菜单 |
| `double_click()` | 双击 | 选中文本、打开文件 |
| `drag_and_drop()` | 拖拽到目标元素 | 拖拽排序 |
| `drag_and_drop_by_offset()` | 拖拽指定偏移 | Canvas 绘图 |
| `move_to_element()` | 移动到元素 | 悬停菜单 |
| `move_to_element_with_offset()` | 移动到元素偏移位置 | 精确定位 |
| `release()` | 释放鼠标 | 拖拽操作的结束 |
| `send_keys()` | 发送键盘输入 | 输入文本 |
| `key_down()` / `key_up()` | 按下/释放按键 | 组合键（Ctrl+C） |

**关键点**：
- 所有操作必须调用 `perform()` 才会执行
- 可以链式调用多个操作
- 适合处理复杂的用户交互场景

## 第 2 题：如何模拟键盘操作（快捷键、组合键）？

**答案**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver import ActionChains

driver = webdriver.Chrome()
driver.get("https://example.com")

element = driver.find_element(By.ID, "text-editor")

# 方法 1：使用 send_keys 直接输入
element.send_keys("Hello Selenium")

# 方法 2：特殊键
element.send_keys(Keys.ENTER)       # 回车
element.send_keys(Keys.TAB)         # Tab 键
element.send_keys(Keys.BACKSPACE)   # 退格
element.send_keys(Keys.DELETE)      # 删除
element.send_keys(Keys.ESCAPE)      # ESC
element.send_keys(Keys.SPACE)       # 空格

# 方法 3：组合键（Ctrl+A 全选）
element.send_keys(Keys.CONTROL, 'a')

# 方法 4：使用 ActionChains 执行组合键
actions = ActionChains(driver)

# Ctrl+A（全选）
actions.key_down(Keys.CONTROL).send_keys('a').key_up(Keys.CONTROL).perform()

# Ctrl+C（复制）
actions.key_down(Keys.CONTROL).send_keys('c').key_up(Keys.CONTROL).perform()

# Ctrl+V（粘贴）
actions.key_down(Keys.CONTROL).send_keys('v').key_up(Keys.CONTROL).perform()

# Ctrl+X（剪切）
actions.key_down(Keys.CONTROL).send_keys('x').key_up(Keys.CONTROL).perform()

# Ctrl+Z（撤销）
actions.key_down(Keys.CONTROL).send_keys('z').key_up(Keys.CONTROL).perform()

# Shift+Home（选择到行首）
actions.key_down(Keys.SHIFT).send_keys(Keys.HOME).key_up(Keys.SHIFT).perform()

# 方法 5：多键组合（Ctrl+Shift+T 恢复关闭的标签）
actions.key_down(Keys.CONTROL)\
       .key_down(Keys.SHIFT)\
       .send_keys('t')\
       .key_up(Keys.SHIFT)\
       .key_up(Keys.CONTROL)\
       .perform()

# 方法 6：F 键
element.send_keys(Keys.F5)  # 刷新页面
element.send_keys(Keys.F12)  # 开发者工具

# 方法 7：方向键
element.send_keys(Keys.ARROW_UP)
element.send_keys(Keys.ARROW_DOWN)
element.send_keys(Keys.ARROW_LEFT)
element.send_keys(Keys.ARROW_RIGHT)

# 方法 8：Home/End 键
element.send_keys(Keys.HOME)  # 移到行首
element.send_keys(Keys.END)   # 移到行尾

# 方法 9：Page Up/Down
element.send_keys(Keys.PAGE_UP)
element.send_keys(Keys.PAGE_DOWN)

# 实用示例：富文本编辑器操作
class RichTextEditor:
    def __init__(self, driver, editor_locator):
        self.driver = driver
        self.editor = driver.find_element(*editor_locator)
        self.actions = ActionChains(driver)

    def type_text(self, text):
        """输入文本"""
        self.editor.send_keys(text)

    def select_all(self):
        """全选"""
        self.editor.send_keys(Keys.CONTROL, 'a')

    def copy(self):
        """复制"""
        self.editor.send_keys(Keys.CONTROL, 'c')

    def paste(self):
        """粘贴"""
        self.editor.send_keys(Keys.CONTROL, 'v')

    def bold(self):
        """加粗（Ctrl+B）"""
        self.editor.send_keys(Keys.CONTROL, 'b')

    def italic(self):
        """斜体（Ctrl+I）"""
        self.editor.send_keys(Keys.CONTROL, 'i')

    def underline(self):
        """下划线（Ctrl+U）"""
        self.editor.send_keys(Keys.CONTROL, 'u')

    def undo(self):
        """撤销"""
        self.editor.send_keys(Keys.CONTROL, 'z')

    def redo(self):
        """重做（Ctrl+Y）"""
        self.editor.send_keys(Keys.CONTROL, 'y')

# 使用
editor = RichTextEditor(driver, (By.ID, "editor"))
editor.type_text("Hello World")
editor.select_all()
editor.bold()
```

**常用 Keys 枚举**：

| Keys 常量 | 说明 |
|-----------|------|
| `Keys.CONTROL` | Ctrl 键 |
| `Keys.SHIFT` | Shift 键 |
| `Keys.ALT` | Alt 键 |
| `Keys.COMMAND` | Mac Command 键 |
| `Keys.ENTER` | 回车 |
| `Keys.TAB` | Tab 键 |
| `Keys.BACKSPACE` | 退格 |
| `Keys.DELETE` | 删除 |
| `Keys.ESCAPE` | ESC |
| `Keys.SPACE` | 空格 |

**关键点**：
- 组合键需要 `key_down()` 按下，`key_up()` 释放
- 简单输入用 `send_keys()`，复杂操作用 `ActionChains`
- Mac 系统用 `Keys.COMMAND` 代替 `Keys.CONTROL`

## 第 3 题：如何处理文件上传和下载？

**答案**：

**文件上传**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
import os

driver = webdriver.Chrome()
driver.get("https://example.com/upload")

# 方法 1：直接发送文件路径（最简单）
file_input = driver.find_element(By.ID, "file-upload")
file_path = os.path.abspath("test.txt")
file_input.send_keys(file_path)

# 点击上传按钮
driver.find_element(By.ID, "upload-button").click()

# 方法 2：上传多个文件
file_input = driver.find_element(By.ID, "multi-upload")
file1 = os.path.abspath("file1.txt")
file2 = os.path.abspath("file2.txt")
file_input.send_keys(f"{file1}\n{file2}")

# 方法 3：隐藏的文件输入框
file_input = driver.find_element(By.ID, "hidden-upload")
# 使用 JavaScript 显示元素
driver.execute_script("arguments[0].style.display = 'block';", file_input)
file_input.send_keys(file_path)

# 方法 4：远程 WebDriver（Selenium Grid）
from selenium.webdriver.remote.file_detector import LocalFileDetector

driver.file_detector = LocalFileDetector()
file_input.send_keys("/local/path/to/file.txt")
```

**文件下载**：

```python
from selenium import webdriver
from selenium.webdriver.chrome.options import Options
import os
import time

# 配置下载路径
download_dir = os.path.abspath("downloads")
os.makedirs(download_dir, exist_ok=True)

options = Options()
prefs = {
    "download.default_directory": download_dir,
    "download.prompt_for_download": False,  # 不弹出下载提示
    "download.directory_upgrade": True,
    "safebrowsing.enabled": True
}
options.add_experimental_option("prefs", prefs)

driver = webdriver.Chrome(options=options)
driver.get("https://example.com/download")

# 点击下载链接
driver.find_element(By.ID, "download-link").click()

# 等待文件下载完成
def wait_for_download(directory, filename, timeout=60):
    """等待文件下载完成"""
    end_time = time.time() + timeout
    while time.time() < end_time:
        if os.path.exists(os.path.join(directory, filename)):
            # 确保文件下载完成（不是 .crdownload 临时文件）
            if not filename.endswith('.crdownload'):
                return True
        time.sleep(1)
    return False

# 使用
if wait_for_download(download_dir, "report.pdf"):
    print("文件下载完成")
else:
    print("文件下载超时")

# 验证下载的文件
downloaded_file = os.path.join(download_dir, "report.pdf")
assert os.path.exists(downloaded_file), "文件未下载"
print(f"文件大小: {os.path.getsize(downloaded_file)} bytes")
```

**处理下载对话框（使用 Chrome DevTools Protocol）**：

```python
# 启用下载事件
driver.execute_cdp_cmd("Browser.setDownloadBehavior", {
    "behavior": "allow",
    "downloadPath": download_dir
})

# 点击下载
driver.find_element(By.ID, "download-button").click()

# 获取下载状态

    return driver.execute_cdp_cmd("Page.getDownloadList", {})

time.sleep(2)
status = get_download_status(driver)
print(f"下载状态: {status}")
```

**关键点**：
- 文件上传直接用 `send_keys()` 发送绝对路径
- 文件下载需要配置浏览器选项
- 等待下载完成需要轮询文件系统
- 远程执行需要使用 `LocalFileDetector`

## 第 4 题：如何处理下拉滚动条和页面滚动？

**答案**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
import time

driver = webdriver.Chrome()
driver.get("https://example.com")

# 方法 1：滚动到页面底部
driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

# 方法 2：滚动到页面顶部
driver.execute_script("window.scrollTo(0, 0);")

# 方法 3：滚动到指定位置
driver.execute_script("window.scrollTo(0, 500);")  # 滚动到 Y=500

# 方法 4：滚动到元素位置
element = driver.find_element(By.ID, "footer")
driver.execute_script("arguments[0].scrollIntoView();", element)

# 方法 5：滚动到元素位置（顶部对齐）
driver.execute_script("arguments[0].scrollIntoView(true);", element)

# 方法 6：滚动到元素位置（底部对齐）
driver.execute_script("arguments[0].scrollIntoView(false);", element)

# 方法 7：平滑滚动
driver.execute_script("""
    arguments[0].scrollIntoView({
        behavior: 'smooth',
        block: 'center',
        inline: 'nearest'
    });
""", element)

# 方法 8：逐步滚动（模拟用户行为）
def smooth_scroll_to_bottom(driver, step=300, delay=0.1):
    """平滑滚动到底部"""
    last_height = driver.execute_script("return document.body.scrollHeight")
    current_position = 0

    while current_position < last_height:
        current_position += step
        driver.execute_script(f"window.scrollTo(0, {current_position});")
        time.sleep(delay)

        # 检查是否有新内容加载（无限滚动）
        new_height = driver.execute_script("return document.body.scrollHeight")
        if new_height > last_height:
            last_height = new_height

smooth_scroll_to_bottom(driver)

# 方法 9：滚动特定元素的滚动条
scrollable_div = driver.find_element(By.ID, "scrollable-container")
driver.execute_script("arguments[0].scrollTop = arguments[0].scrollHeight", scrollable_div)

# 方法 10：使用 Keys 滚动
body = driver.find_element(By.TAG_NAME, "body")
body.send_keys(Keys.PAGE_DOWN)  # 向下翻页
body.send_keys(Keys.PAGE_UP)    # 向上翻页
body.send_keys(Keys.END)        # 滚动到底部
body.send_keys(Keys.HOME)       # 滚动到顶部

# 方法 11：滚动到元素可见区域中心
def scroll_to_center(driver, element):
    """滚动使元素位于可见区域中心"""
    driver.execute_script("""
        var element = arguments[0];
        var elementRect = element.getBoundingClientRect();
        var absoluteElementTop = elementRect.top + window.pageYOffset;
        var middle = absoluteElementTop - (window.innerHeight / 2);
        window.scrollTo(0, middle);
    """, element)

element = driver.find_element(By.ID, "target")
scroll_to_center(driver, element)

# 方法 12：获取当前滚动位置
scroll_position = driver.execute_script("return window.pageYOffset;")
print(f"当前滚动位置: {scroll_position}px")

# 方法 13：检查是否滚动到底部
at_bottom = driver.execute_script("""
    return (window.innerHeight + window.pageYOffset) >= document.body.scrollHeight;
""")
print(f"是否到达底部: {at_bottom}")

# 方法 14：无限滚动加载（如社交媒体 Feed）
def load_all_content(driver, max_scrolls=10):
    """加载所有内容（无限滚动）"""
    last_height = driver.execute_script("return document.body.scrollHeight")

    for i in range(max_scrolls):
        # 滚动到底部
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")
        time.sleep(2)  # 等待加载

        # 检查是否有新内容
        new_height = driver.execute_script("return document.body.scrollHeight")
        if new_height == last_height:
            break  # 没有新内容，停止滚动
        last_height = new_height

    print(f"共滚动 {i + 1} 次")

load_all_content(driver)

# 方法 15：滚动到元素并点击（处理被遮挡的元素）
def scroll_and_click(driver, locator):
    """滚动到元素并点击"""
    element = driver.find_element(*locator)
    # 滚动到元素
    driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", element)
    time.sleep(0.5)  # 等待滚动完成
    # 点击
    element.click()

scroll_and_click(driver, (By.ID, "submit-button"))
```

**滚动策略对比**：

| 方法 | 优点 | 缺点 | 使用场景 |
|------|------|------|---------|
| `scrollIntoView()` | 简单直接 | 可能过于突然 | 快速定位元素 |
| `window.scrollTo()` | 精确控制 | 需要计算坐标 | 固定位置滚动 |
| 逐步滚动 | 模拟真实用户 | 速度较慢 | 触发懒加载 |
| `Keys.PAGE_DOWN` | 接近用户行为 | 滚动距离不固定 | 简单滚动 |

**关键点**：
- 推荐使用 `scrollIntoView()` 滚动到元素
- 无限滚动页面需要循环加载
- 滚动后等待一小段时间确保完成

## 第 5 题：如何处理右键菜单（Context Menu）？

**答案**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver import ActionChains
from selenium.webdriver.common.keys import Keys

driver = webdriver.Chrome()
driver.get("https://example.com")

actions = ActionChains(driver)

# 方法 1：右键点击元素
element = driver.find_element(By.ID, "target")
actions.context_click(element).perform()

# 方法 2：右键点击后选择菜单项
element = driver.find_element(By.ID, "file")
actions.context_click(element).perform()

# 等待右键菜单出现
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

wait = WebDriverWait(driver, 10)
menu_item = wait.until(
    EC.element_to_be_clickable((By.ID, "delete-option"))
)
menu_item.click()

# 方法 3：使用键盘导航右键菜单
element = driver.find_element(By.ID, "target")
actions.context_click(element).perform()

# 按下箭头键选择菜单项
actions.send_keys(Keys.ARROW_DOWN).perform()  # 向下移动
actions.send_keys(Keys.ARROW_DOWN).perform()  # 再向下
actions.send_keys(Keys.ENTER).perform()       # 确认选择

# 方法 4：自定义右键菜单（JavaScript 实现）
element = driver.find_element(By.ID, "custom-menu-trigger")
actions.context_click(element).perform()

# 查找自定义菜单项
menu = wait.until(EC.presence_of_element_located((By.CLASS_NAME, "context-menu")))
menu_item = menu.find_element(By.XPATH, "//li[text()='复制']")
menu_item.click()

# 方法 5：处理嵌套右键菜单
element = driver.find_element(By.ID, "file")
actions.context_click(element).perform()

# 悬停到"打开方式"子菜单
submenu_trigger = wait.until(
    EC.presence_of_element_located((By.ID, "open-with"))
)
actions.move_to_element(submenu_trigger).perform()

# 点击子菜单项
submenu_item = wait.until(
    EC.element_to_be_clickable((By.ID, "open-with-notepad"))
)
submenu_item.click()

# 方法 6：关闭右键菜单（不选择）
element = driver.find_element(By.ID, "target")
actions.context_click(element).perform()

# 按 ESC 关闭菜单
actions.send_keys(Keys.ESCAPE).perform()

# 或者点击页面其他地方
body = driver.find_element(By.TAG_NAME, "body")
body.click()

# 方法 7：验证右键菜单项
element = driver.find_element(By.ID, "target")
actions.context_click(element).perform()

# 获取所有菜单项
menu_items = driver.find_elements(By.CSS_SELECTOR, ".context-menu-item")
menu_texts = [item.text for item in menu_items]
print(f"菜单项: {menu_texts}")

expected_items = ["复制", "粘贴", "删除"]
assert all(item in menu_texts for item in expected_items), "菜单项不完整"

# 实用封装类
class ContextMenuHandler:
    def __init__(self, driver):
        self.driver = driver
        self.actions = ActionChains(driver)
        self.wait = WebDriverWait(driver, 10)

    def open_context_menu(self, element):
        """打开右键菜单"""
        self.actions.context_click(element).perform()

    def select_menu_item(self, element, menu_item_text):
        """打开右键菜单并选择菜单项"""
        self.open_context_menu(element)

        # 查找并点击菜单项
        menu_item = self.wait.until(
            EC.element_to_be_clickable(
                (By.XPATH, f"//li[contains(text(), '{menu_item_text}')]")
            )
        )
        menu_item.click()

    def close_context_menu(self):
        """关闭右键菜单"""
        self.actions.send_keys(Keys.ESCAPE).perform()

# 使用
handler = ContextMenuHandler(driver)
element = driver.find_element(By.ID, "file")
handler.select_menu_item(element, "删除")
```

**关键点**：
- 使用 `context_click()` 触发右键菜单
- 自定义右键菜单需要等待元素出现
- 可以用键盘或鼠标选择菜单项
- 浏览器原生右键菜单无法通过 Selenium 控制

## 第 6 题：如何处理拖拽操作（Drag and Drop）？

**答案**：

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver import ActionChains
import time

driver = webdriver.Chrome()
driver.get("https://example.com/drag-drop")

actions = ActionChains(driver)

# 方法 1：drag_and_drop（最简单）
source = driver.find_element(By.ID, "draggable")
target = driver.find_element(By.ID, "droppable")
actions.drag_and_drop(source, target).perform()

# 方法 2：drag_and_drop_by_offset（按坐标拖拽）
source = driver.find_element(By.ID, "draggable")
actions.drag_and_drop_by_offset(source, xoffset=100, yoffset=50).perform()

# 方法 3：分步拖拽（更精确控制）
source = driver.find_element(By.ID, "draggable")
```

```python
def js_drag_and_drop(driver, source, target):

        function simulateDragDrop(sourceNode, destinationNode) {
            var EVENT_TYPES = {
                DRAG_END: 'dragend',
                DRAG_START: 'dragstart',
                DROP: 'drop'
            }

            function createCustomEvent(type) {
                var event = new CustomEvent('CustomEvent')
                event.initCustomEvent(type, true, true, null)
                event.dataTransfer = {
                    data: {},
                    setData: function(type, val) {
                        this.data[type] = val
                    },
                    getData: function(type) {
                        return this.data[type]
                    }
                }
                return event
            }

            function dispatchEvent(node, type, event) {
                if (node.dispatchEvent) {
                    return node.dispatchEvent(event)
                }
            }

            var event = createCustomEvent(EVENT_TYPES.DRAG_START)
            dispatchEvent(sourceNode, EVENT_TYPES.DRAG_START, event)

            var dropEvent = createCustomEvent(EVENT_TYPES.DROP)
            dropEvent.dataTransfer = event.dataTransfer
            dispatchEvent(destinationNode, EVENT_TYPES.DROP, dropEvent)

            var dragEndEvent = createCustomEvent(EVENT_TYPES.DRAG_END)
            dragEndEvent.dataTransfer = event.dataTransfer
            dispatchEvent(sourceNode, EVENT_TYPES.DRAG_END, dragEndEvent)
        }

        simulateDragDrop(arguments[0], arguments[1]);
    """, source, target)
```

```python
def hover_and_click_safely(driver, menu_locator, item_locator, max_attempts=3):
    """安全地悬停并点击（处理菜单消失）"""
    actions = ActionChains(driver)
    wait = WebDriverWait(driver, 10)

    for attempt in range(max_attempts):
        try:
            menu = driver.find_element(*menu_locator)
            actions.move_to_element(menu).perform()

            item = wait.until(EC.element_to_be_clickable(item_locator))
            item.click()
            return True
        except:
            if attempt < max_attempts - 1:
                print(f"悬停失败，重试 {attempt + 1}/{max_attempts}")
                time.sleep(0.5)
            else:
                raise

    return False

# 使用

    driver,
    (By.ID, "menu"),
    (By.ID, "submenu-item")

class HoverMenuHandler:
    def __init__(self, driver, timeout=10):
        self.driver = driver
        self.actions = ActionChains(driver)
        self.wait = WebDriverWait(driver, timeout)

    def hover_and_click(self, menu_locator, item_locator, pause=0.5):
        """悬停菜单并点击菜单项"""
        menu = self.driver.find_element(*menu_locator)
        self.actions.move_to_element(menu).perform()
        time.sleep(pause)  # 等待菜单动画

        item = self.wait.until(EC.element_to_be_clickable(item_locator))
        item.click()

    def hover_multi_level(self, *locators):
        """多层级悬停"""
        for i, locator in enumerate(locators[:-1]):
            element = self.wait.until(EC.presence_of_element_located(locator))
            self.actions.move_to_element(element).perform()
            time.sleep(0.3)

        # 点击最后一个元素
        final_element = self.wait.until(EC.element_to_be_clickable(locators[-1]))
        final_element.click()

    def get_tooltip_text(self, element_locator, tooltip_locator):
        """获取 tooltip 文本"""
        element = self.driver.find_element(*element_locator)
        self.actions.move_to_element(element).perform()

        tooltip = self.wait.until(EC.visibility_of_element_located(tooltip_locator))
        return tooltip.text
```

```python
class SafeNavigator:
    """安全导航类"""

    def __init__(self, driver, timeout=10):
        self.driver = driver
        self.timeout = timeout
        self.wait = WebDriverWait(driver, timeout)

    def navigate_to(self, url):
        """导航到 URL"""
        self.driver.get(url)
        self._wait_for_page_load()

    def go_back(self):
        """后退"""
        current_url = self.driver.current_url
        self.driver.back()
        self.wait.until(EC.url_changes(current_url))
        self._wait_for_page_load()

    def go_forward(self):
        """前进"""
        current_url = self.driver.current_url
        self.driver.forward()
        self.wait.until(EC.url_changes(current_url))
        self._wait_for_page_load()

    def refresh(self):
        """刷新"""
        self.driver.refresh()
        self._wait_for_page_load()

    def _wait_for_page_load(self):
        """等待页面加载完成"""
        self.wait.until(
            lambda d: d.execute_script("return document.readyState") == "complete"
        )
```

```python
                   .click()\
```

```python
def load_all_lazy_content(driver, max_scrolls=10):
    """加载所有懒加载内容"""
    last_height = driver.execute_script("return document.body.scrollHeight")

    for i in range(max_scrolls):
        # 滚动到底部
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

        # 等待新内容加载
        time.sleep(2)

        # 检查是否有新内容
        new_height = driver.execute_script("return document.body.scrollHeight")
        if new_height == last_height:
            break  # 没有新内容了
        last_height = new_height

    print(f"共滚动 {i + 1} 次，加载完成")

load_all_lazy_content(driver)

# 方法 2：滚动到特定元素
def scroll_to_element_lazy_load(driver, target_locator):
    """滚动直到特定元素出现"""
    max_attempts = 20
    for attempt in range(max_attempts):
        try:
            element = driver.find_element(*target_locator)
            if element.is_displayed():
                return element
        except:
            pass

        # 继续滚动
        driver.execute_script("window.scrollBy(0, 300);")
        time.sleep(0.5)

    raise Exception(f"元素未找到: {target_locator}")

# 使用
target_element = scroll_to_element_lazy_load(driver, (By.ID, "item-50"))

# 方法 3：监控元素数量变化
from selenium.webdriver.support.ui import WebDriverWait

def wait_for_lazy_load_complete(driver, item_locator, expected_count):
    """等待懒加载完成（达到预期数量）"""
    wait = WebDriverWait(driver, 60)

    def check_count(d):
        items = d.find_elements(*item_locator)
        current_count = len(items)
        print(f"当前加载: {current_count}/{expected_count}")

        if current_count >= expected_count:
            return True

        # 继续滚动
        d.execute_script("window.scrollTo(0, document.body.scrollHeight);")
        time.sleep(1)
        return False

    wait.until(check_count)

# 使用
wait_for_lazy_load_complete(driver, (By.CLASS_NAME, "item"), expected_count=100)
```

**关键点**：
- 懒加载需要滚动触发
- 检测页面高度变化判断是否有新内容
- 设置最大滚动次数避免无限循环

```python
# 方法 5：复制粘贴工具类
class ClipboardHelper:
    def __init__(self, driver):
        self.driver = driver
        self.actions = ActionChains(driver)

    def copy_element_text(self, element):
        """复制元素文本到剪贴板"""
        element.click()
        self.actions.key_down(Keys.CONTROL).send_keys('a').key_up(Keys.CONTROL).perform()
        self.actions.key_down(Keys.CONTROL).send_keys('c').key_up(Keys.CONTROL).perform()
        time.sleep(0.5)
        return pyperclip.paste()

    def paste_to_element(self, element, text=None):
        """粘贴到元素（可指定文本）"""
        if text:
            pyperclip.copy(text)

        element.click()
        self.actions.key_down(Keys.CONTROL).send_keys('v').key_up(Keys.CONTROL).perform()

# 使用
clipboard = ClipboardHelper(driver)

# 复制
source = driver.find_element(By.ID, "source")
copied_text = clipboard.copy_element_text(source)

# 粘贴
target = driver.find_element(By.ID, "target")
clipboard.paste_to_element(target, "新文本")
```

**关键点**：
- 使用 Ctrl+C/V 或 Cmd+C/V（Mac）
- `pyperclip` 库可以直接操作系统剪贴板
- 复制粘贴操作需要短暂延迟
