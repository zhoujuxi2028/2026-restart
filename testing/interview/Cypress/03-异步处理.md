## 1. Cypress 如何处理异步操作？为什么不需要 async/await？

**答案**:

Cypress 内置异步处理机制，命令自动按顺序执行并等待完成。

```javascript
// ❌ 不需要 async/await
async function testLogin() {
  await cy.visit('/login');  // 不需要 await
  await cy.get('#email').type('user@example.com');
  await cy.click('button');
}

// ✅ Cypress 方式 - 自动队列
cy.visit('/login');
cy.get('#email').type('user@example.com');
cy.get('button').click();
// 命令按顺序执行，自动等待
```

**Cypress 的异步队列机制**:

```javascript
// 示例：理解命令队列
console.log('1. 同步代码');

cy.visit('/page');           // 加入队列
cy.get('.element').click();  // 加入队列

console.log('2. 同步代码');

// 执行顺序：
// 1. 同步代码 (立即)
// 2. 同步代码 (立即)
// 然后队列开始执行：
// visit -> get -> click

// 验证队列行为
let value = 'initial';

cy.wrap(null).then(() => {
  value = 'updated';
});

console.log(value);  // 输出: 'initial' (队列还没执行)

cy.wrap(null).then(() => {
  console.log(value);  // 输出: 'updated' (队列执行中)
});
```

**为什么不需要 async/await**:

| 特性 | 传统 Promise | Cypress |
|-----|--------------|---------|
| 语法 | `await` 关键字 | 链式调用 |
| 等待 | 手动 `await` | 自动等待 |
| 错误处理 | try/catch | 自动捕获 |
| 重试 | 手动实现 | 内置重试 |
| 顺序 | 需要 async/await | 自动排队 |

```javascript
// 传统异步代码
async function traditionalTest() {
  const response = await fetch('/api/users');
  const users = await response.json();
  const element = await findElement('.user-list');
  await element.click();
}

// Cypress 等价代码
cy.request('/api/users')
  .its('body')
  .then((users) => {
    cy.log('Users:', users.length);
  });

cy.get('.user-list').click();
// 不需要等待前面的请求完成
```

**关键点**:
- Cypress 命令返回 Chainable，不是 Promise
- 命令自动排队并按顺序执行
- 内置等待和重试机制
- 不能使用 `async/await`（会破坏队列）
- 使用 `.then()` 处理异步结果

## 2. .then() 在 Cypress 中的作用是什么？

**答案**:

`.then()` 用于访问前一个命令的结果并执行同步/异步操作。

```javascript
// 1. 访问命令结果
cy.get('button')
  .then(($btn) => {
    // $btn 是 jQuery 对象
    const text = $btn.text();
    expect(text).to.eq('Submit');
  });

// 2. 链式处理
cy.request('/api/user')
  .then((response) => {
    return response.body.id;
  })
  .then((userId) => {
    cy.visit(`/user/${userId}`);
  });

// 3. 条件逻辑
cy.get('.status')
  .invoke('text')
  .then((text) => {
    if (text.includes('Active')) {
      cy.get('.deactivate-btn').click();
    } else {
      cy.get('.activate-btn').click();
    }
  });

// 4. 使用外部变量
let userEmail;

cy.fixture('user').then((user) => {
  userEmail = user.email;
});

cy.then(() => {
  cy.get('#email').type(userEmail);
});
```

**详细应用**:

```javascript
describe('then() 命令详解', () => {
  it('访问 DOM 元素', () => {
    cy.get('.item')
      .then(($items) => {
        // $items 是 jQuery 对象
        expect($items).to.have.length(5);
        expect($items.eq(0)).to.contain('First');

        // 返回值传递给下一个 then
        return $items.length;
      })
      .then((count) => {
        cy.log(`Found ${count} items`);
      });
  });

  it('API 响应处理', () => {
    cy.request('/api/users')
      .then((response) => {
        expect(response.status).to.eq(200);
        expect(response.body).to.have.length.at.least(1);

        // 提取数据
        return response.body[0];
      })
      .then((firstUser) => {
        cy.log('First user:', firstUser.name);
        cy.visit(`/users/${firstUser.id}`);
      });
  });

  it('复杂条件判断', () => {
    cy.get('[data-cy="user-role"]')
      .invoke('text')
      .then((role) => {
        if (role === 'admin') {
          cy.get('.admin-panel').should('be.visible');
          cy.get('.admin-actions').should('exist');
        } else if (role === 'user') {
          cy.get('.user-panel').should('be.visible');
          cy.get('.admin-actions').should('not.exist');
        } else {
          cy.get('.guest-message').should('be.visible');
        }
      });
  });

  it('数据转换', () => {
    cy.get('.price')
      .invoke('text')
      .then((priceText) => {
        // "$19.99" -> 19.99
        return parseFloat(priceText.replace('$', ''));
      })
      .then((price) => {
        const tax = price * 0.1;
        const total = price + tax;
        cy.wrap(total).should('be.gt', 20);
      });
  });

  it('多步骤工作流', () => {
    // 步骤 1: 创建用户
    cy.request('POST', '/api/users', {
      name: 'John Doe',
      email: 'john@example.com'
    })
      .then((response) => {
        const userId = response.body.id;

        // 步骤 2: 上传头像
        return cy.request('POST', `/api/users/${userId}/avatar`, {
          avatar: 'base64-image-data'
        });
      })
      .then((response) => {
        // 步骤 3: 验证
        cy.visit(`/users/${response.body.userId}`);
        cy.get('.avatar-img').should('be.visible');
      });
  });
});
```

**返回值处理**:

```javascript
// 1. 返回 Cypress 命令
cy.get('.element')
  .then(($el) => {
    if ($el.hasClass('special')) {
      return cy.get('.special-content');
    } else {
      return cy.get('.normal-content');
    }
  })
  .should('be.visible');

// 2. 返回 Promise
cy.get('.async-btn')
  .click()
  .then(() => {
    return new Promise((resolve) => {
      setTimeout(() => resolve('done'), 1000);
    });
  })
  .then((result) => {
    expect(result).to.eq('done');
  });

// 3. 返回普通值
cy.wrap(5)
  .then((num) => num * 2)
  .then((num) => num + 3)
  .should('eq', 13);

// 4. 不返回值
cy.get('.element')
  .then(($el) => {
    console.log($el.text());
    // 未返回值，后续 then 收到原始 $el
  })
  .should('have.class', 'element');
```

**关键点**:
- 访问前一个命令的结果
- 支持同步和异步操作
- 可以返回值传递给下一个 `.then()`
- 返回 Cypress 命令会加入队列
- 用于条件逻辑和数据处理

## 3. Cypress 的自动等待机制如何工作？

**答案**:

Cypress 自动等待元素存在、可见、可操作，无需显式等待。

```javascript
// ❌ Selenium 需要显式等待
await driver.wait(until.elementLocated(By.id('button')), 5000);
await driver.wait(until.elementIsVisible(button), 5000);
await button.click();

// ✅ Cypress 自动等待
cy.get('#button').click();
// 自动等待:
// 1. 元素存在于 DOM
// 2. 元素可见
// 3. 元素未被覆盖
// 4. 元素未禁用
// 5. 元素未在动画中
```

**自动等待的检查项**:

```javascript
describe('自动等待机制', () => {
  it('等待元素存在', () => {
    // 等待最多 4 秒（默认超时）
    cy.get('.dynamic-element');  // 元素出现在 DOM
  });

  it('等待元素可见', () => {
    cy.get('.hidden-element')
      .should('be.visible')  // 等待可见
      .click();
  });

  it('等待元素可操作', () => {
    // 自动等待按钮可点击
    cy.get('button')
      .click();  // 等待: 可见、未禁用、未被覆盖
  });

  it('等待动画完成', () => {
    cy.get('.animated-element')
      .click();  // 等待动画完成
  });

  it('等待元素稳定', () => {
    cy.get('.moving-element')
      .click();  // 等待元素停止移动
  });
});
```

**可操作性检查**:

| 检查项 | 说明 |
|--------|------|
| 存在性 | 元素在 DOM 中 |
| 可见性 | 不是 `display: none`、`visibility: hidden` |
| 未禁用 | 没有 `disabled` 属性 |
| 未覆盖 | 不被其他元素遮挡 |
| 稳定性 | 不在动画中 |
| 可滚动 | 能滚动到视图内 |

```javascript
// Cypress 自动执行的检查
cy.get('button').click();

// 等价于以下所有检查:
// 1. 查询 DOM 直到找到元素
// 2. 滚动元素到视图内
// 3. 检查元素是否可见
// 4. 检查元素是否被覆盖
// 5. 检查元素是否禁用
// 6. 检查元素是否在动画中
// 7. 等待动画完成
// 8. 触发点击事件
```

**自动重试**:

```javascript
// 自动重试直到断言通过
cy.get('.loading')
  .should('not.exist');  // 持续重试直到元素消失

cy.get('.result')
  .should('contain', 'Success');  // 持续重试直到文本出现

cy.get('.items')
  .should('have.length', 10);  // 持续重试直到有 10 个元素

// 重试机制时间线（默认 4 秒超时）:
// 0ms: 尝试 1
// 50ms: 尝试 2
// 100ms: 尝试 3
// ...
// 4000ms: 最后尝试，失败则报错
```

**配置超时时间**:

```javascript
// 全局配置
// cypress.config.js
{
  defaultCommandTimeout: 10000,  // 10 秒
  pageLoadTimeout: 60000         // 60 秒
}

// 命令级配置
cy.get('.slow-element', { timeout: 10000 })
  .should('be.visible');

cy.get('button', { timeout: 5000 })
  .click({ timeout: 3000 });

// 断言超时
cy.get('.element')
  .should('be.visible')  // 使用 defaultCommandTimeout
  .and('contain', 'text', { timeout: 10000 });  // 自定义超时
```

**关键点**:
- 自动等待元素存在、可见、可操作
- 命令自动重试直到成功或超时
- 无需 `sleep()` 或手动等待
- 默认超时 4 秒（可配置）
- 大大简化测试代码

## 4. cy.wait() 有哪些用法？

**答案**:

`cy.wait()` 用于等待时间或网络请求完成。

```javascript
// 1. 等待固定时间（不推荐）
cy.wait(1000);  // 等待 1 秒

// 2. 等待网络请求（推荐）
cy.intercept('GET', '/api/users').as('getUsers');
cy.visit('/users');
cy.wait('@getUsers');  // 等待请求完成

// 3. 等待多个请求
cy.intercept('GET', '/api/users').as('getUsers');
cy.intercept('GET', '/api/posts').as('getPosts');
cy.visit('/dashboard');
cy.wait(['@getUsers', '@getPosts']);

// 4. 验证请求和响应
cy.wait('@getUsers').then((interception) => {
  expect(interception.response.statusCode).to.eq(200);
  expect(interception.response.body).to.have.length(10);
});

// 5. 等待特定时间后的请求
cy.wait('@getUsers').wait(500).wait('@refreshUsers');
```

**详细用法**:

```javascript
describe('wait() 命令详解', () => {
  it('等待固定时间', () => {
    cy.visit('/animation-page');
    cy.get('.start-animation').click();
    cy.wait(2000);  // 等待动画完成
    cy.get('.animation-result').should('be.visible');

    // ⚠️ 不推荐：使脚本变慢
    // 更好的方式：使用断言自动等待
    cy.get('.animation-result', { timeout: 3000 })
      .should('be.visible');
  });

  it('等待单个请求', () => {
    cy.intercept('POST', '/api/login').as('loginRequest');

    cy.visit('/login');
    cy.get('#email').type('user@example.com');
    cy.get('#password').type('password123');
    cy.get('button').click();

    cy.wait('@loginRequest')
      .its('response.statusCode')
      .should('eq', 200);
  });

  it('等待多个请求', () => {
    cy.intercept('GET', '/api/user').as('getUser');
    cy.intercept('GET', '/api/settings').as('getSettings');
    cy.intercept('GET', '/api/notifications').as('getNotifications');

    cy.visit('/dashboard');

    // 等待所有请求完成
    cy.wait(['@getUser', '@getSettings', '@getNotifications']);

    cy.get('.dashboard-content').should('be.visible');
  });

  it('请求完成后验证', () => {
    cy.intercept('GET', '/api/products').as('getProducts');

    cy.visit('/products');
    cy.wait('@getProducts').then((interception) => {
      // 访问请求对象
      expect(interception.request.url).to.include('/api/products');
      expect(interception.request.headers).to.have.property('authorization');

      // 访问响应对象
      expect(interception.response.statusCode).to.eq(200);
      expect(interception.response.body.products).to.have.length(20);

      // 使用响应数据
      const firstProduct = interception.response.body.products[0];
      cy.contains('.product-card', firstProduct.name).should('exist');
    });
  });

  it('等待请求序列', () => {
    cy.intercept('GET', '/api/categories').as('getCategories');
    cy.intercept('GET', '/api/products*').as('getProducts');

    cy.visit('/shop');
    cy.wait('@getCategories');  // 先等待分类加载

    cy.get('.category-item').first().click();
    cy.wait('@getProducts');  // 再等待产品加载

    cy.get('.product-list').should('be.visible');
  });

  it('等待并修改响应', () => {
    cy.intercept('GET', '/api/config', (req) => {
      req.reply((res) => {
        cy.wait(1000);  // 模拟慢速网络
        res.send({ config: 'modified' });
      });
    }).as('getConfig');

    cy.visit('/');
    cy.wait('@getConfig');
  });
});
```

**等待多次相同请求**:

```javascript
// 场景：无限滚动
cy.intercept('GET', '/api/items*').as('getItems');
cy.visit('/infinite-scroll');

cy.wait('@getItems');  // 第一次加载
cy.scrollTo('bottom');
cy.wait('@getItems');  // 第二次加载
cy.scrollTo('bottom');
cy.wait('@getItems');  // 第三次加载

// 或批量等待
cy.wait('@getItems');
cy.wait('@getItems');
cy.wait('@getItems');
```

**避免使用 cy.wait(ms)**:

```javascript
// ❌ 不好：固定等待
cy.get('.button').click();
cy.wait(2000);  // 浪费时间
cy.get('.result').should('exist');

// ✅ 好：智能等待
cy.get('.button').click();
cy.get('.result', { timeout: 5000 }).should('exist');  // 最多等 5 秒

// ❌ 不好：等待元素出现
cy.wait(3000);
cy.get('.dynamic-element').should('exist');

// ✅ 好：自动等待
cy.get('.dynamic-element', { timeout: 10000 }).should('exist');

// ✅ 好：等待请求
cy.intercept('GET', '/api/data').as('getData');
cy.get('.button').click();
cy.wait('@getData');
cy.get('.result').should('exist');
```

**关键点**:
- `cy.wait(ms)` 固定等待，应避免使用
- `cy.wait('@alias')` 等待网络请求，推荐
- 可验证请求和响应内容
- 支持等待多个请求
- 优先使用断言和自动等待

## 5. 如何处理 Cypress 中的 Promise？

**答案**:

Cypress 命令不是标准 Promise，需要使用 `.then()` 或 `cy.wrap()`。

```javascript
// ❌ 错误：Cypress 命令不是 Promise
const button = cy.get('button');  // Chainable，不是元素
button.click();  // 错误

// ✅ 正确：链式调用
cy.get('button').click();

// ❌ 错误：使用 async/await
async function test() {
  await cy.visit('/page');  // 破坏命令队列
  await cy.get('button').click();
}

// ✅ 正确：使用 .then()
cy.visit('/page');
cy.get('button').click();
```

**处理真正的 Promise**:

```javascript
describe('Promise 处理', () => {
  it('包装 Promise', () => {
    const promise = new Promise((resolve) => {
      setTimeout(() => resolve('result'), 1000);
    });

    cy.wrap(promise)
      .should('eq', 'result');
  });

  it('在 then 中使用 Promise', () => {
    cy.get('button').then(() => {
      return new Promise((resolve) => {
        setTimeout(() => resolve('done'), 1000);
      });
    }).then((result) => {
      expect(result).to.eq('done');
    });
  });

  it('fetch 请求', () => {
    cy.window().then((win) => {
      return win.fetch('/api/data');
    }).then((response) => {
      return response.json();
    }).then((data) => {
      expect(data).to.have.property('id');
    });
  });

  it('多个 Promise', () => {
    const promise1 = Promise.resolve(1);
    const promise2 = Promise.resolve(2);
    const promise3 = Promise.resolve(3);

    cy.wrap(Promise.all([promise1, promise2, promise3]))
      .should('deep.equal', [1, 2, 3]);
  });

  it('async函数结果', () => {
    async function fetchData() {
      const response = await fetch('/api/data');
      return response.json();
    }

    cy.wrap(fetchData())
      .its('name')
      .should('eq', 'John');
  });
});
```

**混合使用 Cypress 命令和 Promise**:

```javascript
describe('混合使用', () => {
  it('Cypress 命令后跟 Promise', () => {
    cy.visit('/page')
      .then(() => {
        // 返回 Promise
        return fetch('/api/data').then(r => r.json());
      })
      .then((data) => {
        // 使用 Promise 结果
        cy.get('#username').type(data.name);
      });
  });

  it('Promise 后跟 Cypress 命令', () => {
    cy.window()
      .then((win) => {
        return win.fetch('/api/config').then(r => r.json());
      })
      .then((config) => {
        // 返回 Cypress 命令
        return cy.get('#api-key').type(config.apiKey);
      });
  });

  it('复杂工作流', () => {
    let userId;

    // 步骤 1: API 请求获取数据
    cy.request('/api/current-user')
      .its('body.id')
      .then((id) => {
        userId = id;

        // 步骤 2: 使用原生 fetch
        return fetch(`/api/users/${id}/profile`);
      })
      .then((response) => response.json())
      .then((profile) => {
        // 步骤 3: 使用数据填充表单
        cy.visit('/profile');
        cy.get('#name').should('have.value', profile.name);
      });
  });
});
```

**注意事项**:

```javascript
// ❌ 不要混用 async/await 和 Cypress
it('错误示例', async () => {
  await cy.visit('/page');  // 破坏队列
  const element = await cy.get('button');  // element 不是你期望的
});

// ✅ 在 then 内使用 async/await
it('正确示例', () => {
  cy.visit('/page').then(async () => {
    const response = await fetch('/api/data');
    const data = await response.json();
    cy.wrap(data).as('apiData');
  });

  cy.get('@apiData').its('name').should('exist');
});

// ❌ 不要在 then 外使用 await
it('错误示例', () => {
  cy.get('button').click();

  const result = await somePromise();  // 语法错误
});

// ✅ 在 then 内使用 await
it('正确示例', () => {
  cy.get('button').click().then(async () => {
    const result = await somePromise();
    expect(result).to.exist;
  });
});
```

**关键点**:
- Cypress 命令返回 Chainable，不是 Promise
- 使用 `cy.wrap()` 包装 Promise
- 在 `.then()` 内可以返回 Promise
- 避免在测试顶层使用 `async/await`
- 可以在 `.then()` 内使用 `async/await`

## 6. 如何使用 cy.intercept() 等待和模拟网络请求？

**答案**:

`cy.intercept()` 拦截、等待、模拟和修改网络请求。

```javascript
// 1. 监听请求
cy.intercept('GET', '/api/users').as('getUsers');
cy.visit('/users');
cy.wait('@getUsers');

// 2. 模拟响应
cy.intercept('GET', '/api/users', {
  statusCode: 200,
  body: [
    { id: 1, name: 'John' },
    { id: 2, name: 'Jane' }
  ]
}).as('getUsers');

// 3. 使用 fixture
cy.intercept('GET', '/api/users', {
  fixture: 'users.json'
});

// 4. 动态响应
cy.intercept('GET', '/api/users', (req) => {
  req.reply({
    statusCode: 200,
    body: { users: [] },
    headers: { 'X-Custom': 'value' }
  });
});

// 5. 修改请求
cy.intercept('POST', '/api/login', (req) => {
  req.body.timestamp = Date.now();
  req.continue();
});
```

**详细应用**:

```javascript
describe('intercept 详解', () => {
  it('基本拦截和等待', () => {
    cy.intercept('GET', '/api/posts').as('getPosts');

    cy.visit('/blog');
    cy.wait('@getPosts').then((interception) => {
      expect(interception.response.statusCode).to.eq(200);
      expect(interception.response.body).to.have.length.gt(0);
    });
  });

  it('模拟成功响应', () => {
```

## 7. Cypress 中如何处理竞态条件？

**答案**:

使用断言、别名和 `cy.wait()` 处理竞态条件。

```javascript
// 问题：竞态条件
cy.get('.button').click();  // 触发 API 请求
cy.get('.result').should('contain', 'Success');  // 可能在响应前检查

// ✅ 解决方案 1: 等待网络请求
cy.intercept('POST', '/api/action').as('apiAction');
cy.get('.button').click();
cy.wait('@apiAction');  // 等待完成
cy.get('.result').should('contain', 'Success');

// ✅ 解决方案 2: 等待 loading 消失
cy.get('.button').click();
cy.get('.loading').should('be.visible');
cy.get('.loading').should('not.exist');
cy.get('.result').should('contain', 'Success');

// ✅ 解决方案 3: 延长超时
cy.get('.button').click();
cy.get('.result', { timeout: 10000 })
  .should('contain', 'Success');
```

**常见竞态场景**:

```javascript
describe('竞态条件处理', () => {
  it('场景 1: 快速点击', () => {
    // 问题：连续点击可能发送多个请求
    cy.get('.increment').click().click().click();

    // 解决：等待每次请求
    cy.intercept('POST', '/api/increment').as('increment');

    cy.get('.increment').click();
    cy.wait('@increment');

    cy.get('.increment').click();
    cy.wait('@increment');

    cy.get('.increment').click();
    cy.wait('@increment');

    cy.get('.count').should('contain', '3');
  });

  it('场景 2: 并发请求', () => {
    // 多个同时发起的请求
    cy.intercept('GET', '/api/user').as('getUser');
    cy.intercept('GET', '/api/settings').as('getSettings');
    cy.intercept('GET', '/api/notifications').as('getNotifications');

    cy.visit('/dashboard');

    // 等待所有请求完成
    cy.wait(['@getUser', '@getSettings', '@getNotifications']);

    cy.get('.dashboard').should('be.visible');
  });

  it('场景 3: 表单提交后导航', () => {
    cy.intercept('POST', '/api/submit').as('submit');

    cy.get('form').within(() => {
      cy.get('[name="email"]').type('user@example.com');
      cy.get('button[type="submit"]').click();
    });

    // 等待提交完成
    cy.wait('@submit');

    // 再验证导航
    cy.url().should('include', '/success');
    cy.contains('Form submitted').should('be.visible');
  });

  it('场景 4: 动态内容加载', () => {
    cy.visit('/page');

    // 等待初始加载
    cy.get('.content').should('exist');

    // 触发更新
    cy.get('.refresh-btn').click();

    // 问题：内容可能瞬间更新，难以检测变化
    // 解决：使用请求别名
    cy.intercept('GET', '/api/content').as('getContent');
    cy.get('.refresh-btn').click();
    cy.wait('@getContent');

    cy.get('.timestamp').should('not.contain', '00:00:00');
  });

  it('场景 5: 轮询请求', () => {
    // 应用每 5 秒轮询一次
    cy.intercept('GET', '/api/status').as('pollStatus');

    cy.visit('/monitor');

    // 等待第一次轮询
    cy.wait('@pollStatus');
    cy.get('.status').should('contain', 'Running');

    // 等待第二次轮询
    cy.wait('@pollStatus');
  });

  it('场景 6: 依赖请求序列', () => {
    // 请求 A -> 请求 B -> 请求 C
    cy.intercept('GET', '/api/categories').as('getCategories');
    cy.intercept('GET', '/api/products*').as('getProducts');
    cy.intercept('GET', '/api/reviews*').as('getReviews');

    cy.visit('/shop');
    cy.wait('@getCategories');  // 1. 先加载分类

    cy.get('.category').first().click();
    cy.wait('@getProducts');  // 2. 再加载产品

    cy.get('.product').first().click();
    cy.wait('@getReviews');  // 3. 最后加载评论

    cy.get('.review-list').should('be.visible');
  });
});
```

**最佳实践**:

```javascript
// ❌ 不好：使用 cy.wait(ms)
cy.get('.button').click();
cy.wait(3000);  // 盲目等待
cy.get('.result').should('exist');

// ✅ 好：使用网络别名
cy.intercept('POST', '/api/action').as('action');
cy.get('.button').click();
cy.wait('@action');
cy.get('.result').should('exist');

// ❌ 不好：假设顺序
cy.get('.item-1').should('exist');
cy.get('.item-2').should('exist');  // 可能在 item-2 出现前检查

// ✅ 好：明确等待
cy.get('.item-1').should('exist');
cy.get('.loading').should('not.exist');
cy.get('.item-2').should('exist');

// ✅ 或使用 within
cy.get('.container').within(() => {
  cy.get('.item').should('have.length', 2);
});
```

**关键点**:
- 使用 `cy.wait('@alias')` 等待请求
- 等待 loading 状态消失
- 使用充分的超时时间
- 避免 `cy.wait(ms)` 固定等待
- 验证中间状态（loading、disabled等）

## 8. 如何在测试间共享数据？

**答案**:

使用别名 (alias)、闭包变量或 Cypress 环境变量。

```javascript
// 方法 1: 使用别名
describe('使用别名共享数据', () => {
  before(() => {
    cy.request('/api/user').its('body').as('currentUser');
  });

  it('测试 1', function() {
    cy.log(this.currentUser.name);
    cy.get('#welcome').should('contain', this.currentUser.name);
  });

  it('测试 2', function() {
    cy.get('#email').should('have.value', this.currentUser.email);
  });
});

// 方法 2: 使用闭包变量
describe('使用闭包变量', () => {
  let userId;

  before(() => {
    cy.request('POST', '/api/users', { name: 'John' })
      .its('body.id')
      .then((id) => {
        userId = id;
      });
  });

  it('测试 1', () => {
    cy.visit(`/users/${userId}`);
    cy.contains('John').should('be.visible');
  });

  it('测试 2', () => {
    cy.request(`/api/users/${userId}`)
      .its('body.name')
      .should('eq', 'John');
  });
});

// 方法 3: 使用 Cypress.env()
describe('使用环境变量', () => {
  before(() => {
    cy.request('/api/token')
      .its('body.token')
      .then((token) => {
        Cypress.env('authToken', token);
      });
  });

  it('测试 1', () => {
    const token = Cypress.env('authToken');
    cy.request({
      url: '/api/protected',
      headers: { Authorization: `Bearer ${token}` }
    });
  });
});
```

## 9. Cypress 中的重试机制如何工作？

**答案**:

Cypress 自动重试断言和查询命令，直到成功或超时。

```javascript
// 自动重试的命令
cy.get('.element')           // 重试直到找到
  .should('be.visible')      // 重试直到可见
  .and('contain', 'text')    // 重试直到包含文本
  .click();                  // 不重试（操作命令）

// 重试流程：
// 0ms: 查询 .element
// 50ms: 未找到，重试
// 100ms: 找到但不可见，重试
// 150ms: 可见但不包含文本，重试
// 200ms: 所有条件满足，执行 click
```

**可重试 vs 不可重试命令**:

```javascript
// ✅ 可重试（查询和断言）
cy.get('.element');                    // 查询
cy.contains('text');                   // 查询
cy.get('.element').should('exist');   // 断言
cy.get('.element').find('.child');    // 查询

// ❌ 不可重试（操作命令）
cy.click();                            // 操作
cy.type('text');                       // 操作
cy.request('/api');                    // 操作
cy.visit('/page');                     // 操作

// 影响
cy.get('.element').click();  // click 不重试
cy.get('.element')
  .should('be.visible')      // 先重试确保可见
  .click();                  // 再执行 click
```

**重试示例**:

```javascript
describe('重试机制', () => {
  it('等待元素出现', () => {
    cy.visit('/dynamic-page');

    // 自动重试最多 4 秒
    cy.get('.dynamic-element', { timeout: 4000 })
      .should('be.visible');
  });

  it('等待文本变化', () => {
    cy.get('.counter').should('contain', '0');
    cy.get('.increment').click();

    // 重试直到计数器更新
    cy.get('.counter').should('contain', '1');
  });

  it('等待样式变化', () => {
    cy.get('.element')
      .should('have.css', 'opacity', '1')  // 重试
      .and('have.css', 'display', 'block');  // 重试
  });

  it('复杂条件重试', () => {
    cy.get('.list-item')
      .should('have.length', 5)  // 重试直到有 5 个
      .first()
      .should('contain', 'First')  // 重试直到包含文本
      .and('have.class', 'active');  // 重试直到有类名
  });

  it('自定义重试逻辑', () => {
    cy.get('.status').should(($status) => {
      const text = $status.text();
      // 复杂验证，失败时自动重试
      expect(text).to.match(/^(Ready|Active|Completed)$/);
    });
  });
});
```

**控制重试行为**:

```javascript
// 1. 设置超时
cy.get('.slow-element', { timeout: 10000 })  // 等待 10 秒
  .should('be.visible');

// 2. 断言级超时
cy.get('.element')
  .should('be.visible')  // 使用默认超时
  .and('contain', 'text', { timeout: 5000 });  // 5 秒超时

// 3. 禁用重试（通过 .then()）
cy.get('.element').then(($el) => {
  // then 回调不重试
  expect($el).to.be.visible;  // 立即检查，失败不重试
});

// 4. 全局超时配置
// cypress.config.js
{
  defaultCommandTimeout: 8000,  // 默认 8 秒
  requestTimeout: 10000         // 请求 10 秒
}
```

**重试陷阱**:

```javascript
// ❌ 问题：操作命令不重试
cy.get('.button').click();  // 即使按钮不可点击也不重试

// ✅ 解决：先断言再操作
cy.get('.button')
  .should('be.visible')
  .and('be.enabled')
  .click();

// ❌ 问题：then 回调不重试
cy.get('.element').then(($el) => {
  expect($el).to.have.class('active');  // 不重试，立即失败
});

// ✅ 解决：使用 should
cy.get('.element')
  .should('have.class', 'active');  // 自动重试

// ❌ 问题：过早读取值
let text;
cy.get('.element').then(($el) => {
  text = $el.text();  // 立即读取，不重试
});
cy.wrap(null).should(() => {
  expect(text).to.eq('Expected');  // text 可能是旧值
});

// ✅ 解决：在需要时读取
cy.get('.element')
  .invoke('text')
  .should('eq', 'Expected');  // 持续重试
```

**关键点**:
- 查询和断言命令自动重试
- 操作命令不重试
- 默认超时 4 秒（可配置）
- `.then()` 回调不重试
- 先断言可操作性，再执行操作

## 10. 如何处理条件测试和动态内容？

**答案**:

使用 `.then()` 和条件逻辑处理动态内容。

```javascript
// 基本条件测试
cy.get('body').then(($body) => {
  if ($body.find('.modal').length > 0) {
    // 模态框存在，关闭它
    cy.get('.modal .close-btn').click();
  } else {
    // 模态框不存在，继续
    cy.log('No modal found');
  }
});

// 检查元素属性
cy.get('.element').then(($el) => {
  if ($el.hasClass('disabled')) {
    cy.log('Element is disabled');
  } else {
    cy.wrap($el).click();
  }
});

// 根据文本内容条件执行
cy.get('.status').then(($status) => {
  const text = $status.text();

  if (text.includes('Online')) {
    cy.get('.chat-btn').should('be.visible');
  } else {
    cy.get('.offline-message').should('exist');
  }
});
```

**实际场景**:

```javascript
describe('条件测试', () => {
  it('关闭可选的弹窗', () => {
    cy.visit('/page');

    // 如果有欢迎弹窗，关闭它
    cy.get('body').then(($body) => {
      if ($body.find('.welcome-modal').length) {
        cy.get('.welcome-modal .close').click();
      }
    });

    cy.get('.main-content').should('be.visible');
  });

  it('处理动态按钮状态', () => {
    cy.get('.submit-btn').then(($btn) => {
      if ($btn.is(':disabled')) {
        // 按钮禁用，先填充必填字段
        cy.get('#required-field').type('value');
      }
    });

    // 现在按钮应该可用
    cy.get('.submit-btn').should('be.enabled').click();
  });

  it('根据用户角色测试', () => {
    cy.get('.user-role').then(($role) => {
      const role = $role.text();

      if (role === 'Admin') {
        cy.get('.admin-panel').should('be.visible');
        cy.get('.delete-user-btn').should('exist');
      } else if (role === 'User') {
        cy.get('.admin-panel').should('not.exist');
        cy.get('.user-settings').should('be.visible');
      } else {
        cy.get('.guest-message').should('be.visible');
      }
    });
  });

  it('处理分页内容', () => {
    function checkAllPages() {
      cy.get('.item').should('have.length.at.least', 1);

      cy.get('body').then(($body) => {
        if ($body.find('.next-page:not(.disabled)').length) {
          // 有下一页
          cy.get('.next-page').click();
          checkAllPages();  // 递归检查
        }
      });
    }

    cy.visit('/paginated-list');
    checkAllPages();
  });

  it('处理可选的 cookie 横幅', () => {
    cy.visit('/');

    // 等待页面加载

    // 检查并关闭 cookie 横幅

      cy.intercept('POST', '/api/login', {
        fixture: 'login-response.json'
      });
      cy.visit('/dashboard');
    }
  });
});
```

```javascript
Cypress.Commands.add('closeModalIfExists', () => {
  return cy.get('body').then(($body) => {

    if ($modal.length && $modal.is(':visible')) {
      cy.get('.modal .close-btn').click();
      cy.get('.modal').should('not.be.visible');
    }
  });

// 使用

cy.get('.element').then(($el) => {
  const classes = $el.attr('class');

  return cy.wrap(
    classes.includes('state-1') ? 'State 1' :
    classes.includes('state-2') ? 'State 2' :
    'Unknown State'
  );
}).then((state) => {
  cy.log('Current state:', state);
});

// 异步条件
cy.get('.api-status').then(($status) => {
  if ($status.text() === 'Ready') {
    return cy.request('/api/data');
  } else {
    return cy.wrap(null).wait(1000).then(() => {
      return cy.get('.api-status');  // 重新检查
    });
  }
});
```

**关键点**:
- 使用 `.then()` 访问元素后进行条件判断
- jQuery 方法（`.find()`, `.is()`, `.hasClass()`）不会重试
- 条件逻辑应封装为自定义命令
- 避免过度依赖条件测试
- 优先编写确定性测试
