## 1. cypress.config.js 有哪些常用配置选项？

**答案**:

`cypress.config.js` 用于配置 Cypress 的行为和环境。

```javascript
// cypress.config.js
const { defineConfig } = require('cypress');

module.exports = defineConfig({
  // E2E 测试配置
  e2e: {
    // 基础 URL
    baseUrl: 'http://localhost:3000',

    // 测试文件模式
    specPattern: 'cypress/e2e/**/*.cy.{js,jsx,ts,tsx}',

    // 支持文件
    supportFile: 'cypress/support/e2e.js',

    // 超时设置
    defaultCommandTimeout: 10000,  // 命令超时 (默认 4000ms)
    pageLoadTimeout: 60000,        // 页面加载超时 (默认 60000ms)
    requestTimeout: 10000,         // 请求超时 (默认 5000ms)
    responseTimeout: 30000,        // 响应超时 (默认 30000ms)

    // 视口设置
    viewportWidth: 1280,
    viewportHeight: 720,

    // 视频和截图
    video: true,                   // 录制视频
    screenshotOnRunFailure: true,  // 失败时截图
    videosFolder: 'cypress/videos',
    screenshotsFolder: 'cypress/screenshots',

    // 测试隔离
    testIsolation: true,           // 每个测试间重载页面

    // 浏览器设置
    chromeWebSecurity: false,      // 禁用同源策略

    // 实验性功能
    experimentalStudio: true,      // 启用测试录制器

    // Node 事件
    setupNodeEvents(on, config) {
      // 插件配置
      return config;
    }
  },

  // 组件测试配置
  component: {
    devServer: {
      framework: 'react',
      bundler: 'webpack'
    },
    specPattern: 'src/**/*.cy.{js,jsx,ts,tsx}'
  },

  // 环境变量
  env: {
    apiUrl: 'https://api.example.com',
    username: 'testuser'
  },

  // 重试配置
  retries: {
    runMode: 2,    // cypress run 时重试次数
    openMode: 0    // cypress open 时重试次数
  },

  // 项目 ID（Cypress Dashboard）
  projectId: 'abc123',

  // 其他配置
  watchForFileChanges: true,    // 监听文件变化
  numTestsKeptInMemory: 50,     // 内存中保留的测试数
  trashAssetsBeforeRuns: true   // 运行前清理资源
});
```

**命令行覆盖配置**:

```bash
# 覆盖 baseUrl
npx cypress run --config baseUrl=https://staging.example.com

# 覆盖多个配置
npx cypress run --config viewportWidth=1920,viewportHeight=1080

# 设置环境变量
npx cypress run --env apiUrl=https://api.staging.com,debug=true

# 使用配置文件
npx cypress run --config-file cypress.staging.json
```

**关键点**:
- `baseUrl`: 基础 URL
- 超时设置: 命令、页面加载、请求等
- 视口: 浏览器窗口大小
- 视频和截图: 测试记录
- 环境变量: 跨环境配置
- `setupNodeEvents`: 插件和任务注册
- 命令行可覆盖配置

## 2. 如何创建和使用自定义命令？

**答案**:

在 `cypress/support/commands.js` 中定义自定义命令。

```javascript
// cypress/support/commands.js

// 1. 简单命令
Cypress.Commands.add('login', (email, password) => {
  cy.visit('/login');
  cy.get('#email').type(email);
  cy.get('#password').type(password);
  cy.get('button[type="submit"]').click();
});

// 使用
cy.login('user@example.com', 'password123');

// 2. 子命令（链式调用）
Cypress.Commands.add('setText', { prevSubject: 'element' }, ($element, text) => {
  return cy.wrap($element).clear().type(text);
});

// 使用
cy.get('input').setText('new text');

// 3. 双命令（可选链式调用）
Cypress.Commands.add('console', { prevSubject: 'optional' }, (subject, method) => {
  console[method || 'log'](subject);
  return subject;
});

// 使用
cy.console('Hello');
cy.get('div').console('log');

// 4. 覆盖现有命令
Cypress.Commands.overwrite('visit', (originalFn, url, options) => {
  // 添加认证头
  options = options || {};
  options.headers = {
    ...options.headers,
    'Authorization': `Bearer ${Cypress.env('token')}`
  };

  return originalFn(url, options);
});

// 5. 异步命令
Cypress.Commands.add('waitForResource', (resourceName) => {
  return cy.window().then((win) => {
    return new Cypress.Promise((resolve) => {
      const checkResource = () => {
        if (win.resources && win.resources[resourceName]) {
          resolve(win.resources[resourceName]);
        } else {
          setTimeout(checkResource, 100);
        }
      };
      checkResource();
    });
  });
});
```

**实际应用示例**:

```javascript
  cy.get('.skeleton-loader', { timeout: 10000 }).should('not.exist');

Cypress.Commands.add('shouldBeVisible', { prevSubject: true }, (subject) => {

    .should('exist')
    .and('be.visible')
    .and('not.be.disabled');
  return subject;
});

// 网络命令
Cypress.Commands.add('stubGraphQL', (operationName, data) => {
  cy.intercept('POST', '/graphql', (req) => {
    if (req.body.operationName === operationName) {
      req.alias = operationName;
      req.reply({ data });
    }
  });

// 使用
```

**TypeScript 类型定义**:

```typescript
// cypress/support/index.d.ts
declare namespace Cypress {
  interface Chainable {
    /**
     * 通过 UI 登录
     * @param email 用户邮箱
     * @param password 用户密码
     * @example cy.loginByUI('user@example.com', 'password123')
     */
    loginByUI(email: string, password: string): Chainable<void>;

    /**
     * 通过 API 登录
     * @param email 用户邮箱
     * @param password 用户密码
     * @example cy.loginByAPI('user@example.com', 'password123')
     */
    loginByAPI(email: string, password: string): Chainable<void>;

    /**
     * 填充表单
     * @param formData 表单数据对象
     * @example cy.fillForm({ firstName: 'John', email: 'john@example.com' })
     */
    fillForm(formData: Record<string, string>): Chainable<void>;

    /**
     * 设置文本（清除后输入）
     * @param text 要输入的文本
     * @example cy.get('input').setText('new text')
     */
    setText(text: string): Chainable<Element>;
  }
}
```

## 3. Cypress 插件系统如何工作？

**答案**:

插件在 Node.js 环境运行，通过 `setupNodeEvents` 注册。

```javascript
// cypress.config.js
const { defineConfig } = require('cypress');

module.exports = defineConfig({
  e2e: {
    setupNodeEvents(on, config) {
      // 1. 注册任务
      on('task', {
        // 数据库操作
        'db:seed'() {
          return require('./cypress/db').seed();
        },

        // 文件操作
        'readFile'(filename) {
          return require('fs').readFileSync(filename, 'utf8');
        },

        // 日志
        log(message) {
          console.log(message);
          return null;  // task 必须返回值
        },

        // 异步任务
        async 'api:createUser'(userData) {
          const axios = require('axios');
          const response = await axios.post('/api/users', userData);
          return response.data;
        }
      });

      // 2. 浏览器启动配置
      on('before:browser:launch', (browser, launchOptions) => {
        if (browser.name === 'chrome') {
          launchOptions.args.push('--disable-dev-shm-usage');
          launchOptions.args.push('--disable-gpu');
        }
        return launchOptions;
      });

      // 3. 文件预处理器
      on('file:preprocessor', require('@cypress/webpack-preprocessor'));

      // 4. 截图处理
      on('after:screenshot', (details) => {
        console.log('Screenshot:', details.path);
        return details;
      });

      // 5. 测试运行事件
      on('before:run', (details) => {
        console.log('Starting test run:', details.specs.length, 'specs');
      });

      on('after:run', (results) => {
        console.log('Test run finished:', results.totalFailed, 'failures');
      });

      // 6. 配置修改
      config.baseUrl = process.env.BASE_URL || config.baseUrl;
      config.env.apiUrl = process.env.API_URL || config.env.apiUrl;

      return config;
    }
  }
});

// 在测试中使用任务
cy.task('db:seed');
cy.task('log', 'Test started');
cy.task('api:createUser', { name: 'John', email: 'john@example.com' });
```

**常用插件示例**:

```javascript
const { defineConfig } = require('cypress');
```

**复杂插件示例**:

```javascript
// cypress/plugins/custom-plugin.js
module.exports = (on, config) => {
  // 数据库连接
  const db = require('./db-connection');

  on('task', {
    // 创建测试数据
    async 'db:createUser'(userData) {
      const user = await db.users.create(userData);
      return user;
    },

    // 清理数据
    async 'db:clearUsers'() {
      await db.users.deleteMany({ isTest: true });
      return null;
    },

    // 查询数据
    async 'db:findUser'(query) {
      const user = await db.users.findOne(query);
      return user;
    }
  });

  // 环境变量加载
  require('dotenv').config();
  config.env.dbUrl = process.env.DATABASE_URL;

  // 条件配置
  if (config.env.environment === 'staging') {
    config.baseUrl = 'https://staging.example.com';
    config.retries = { runMode: 2, openMode: 0 };
  }

  // 浏览器配置
  on('before:browser:launch', (browser, launchOptions) => {
    if (browser.name === 'chrome') {
      // 添加扩展
      launchOptions.extensions.push('path/to/extension');

      // 自定义标志
      launchOptions.args.push('--start-maximized');
    }
    return launchOptions;
  });

  // 测试文件过滤
  on('before:spec', (spec) => {
    console.log('Running spec:', spec.relative);
  });

  return config;
};

// cypress.config.js
const customPlugin = require('./cypress/plugins/custom-plugin');

module.exports = defineConfig({
  e2e: {
    setupNodeEvents(on, config) {
      return customPlugin(on, config);
    }
  }
});
```

**插件事件**:

| 事件 | 触发时机 | 用途 |
|-----|---------|------|
| `task` | cy.task() 调用时 | 执行 Node 任务 |
| `before:browser:launch` | 浏览器启动前 | 配置浏览器选项 |
| `before:run` | 测试运行前 | 初始化设置 |
| `after:run` | 测试运行后 | 清理和报告 |
| `before:spec` | 规格文件运行前 | 文件级设置 |
| `after:spec` | 规格文件运行后 | 文件级清理 |
| `after:screenshot` | 截图后 | 处理截图 |
| `file:preprocessor` | 文件加载前 | 编译 TS/JSX |

**关键点**:
- 插件在 Node.js 环境运行
- 通过 `setupNodeEvents` 注册
- 使用 `cy.task()` 调用 Node 任务
- 可以修改配置和浏览器选项
- 支持文件预处理（TypeScript 等）
- 监听测试生命周期事件

## 4. 如何使用环境变量和配置文件？

**答案**:

环境变量通过多种方式配置，优先级依次为：命令行 > 系统变量 > cypress.env.json > cypress.config.js。

```javascript
// 1. cypress.config.js
module.exports = defineConfig({
  e2e: {
    env: {
      apiUrl: 'https://api.example.com',
      username: 'testuser',
      timeout: 5000
    }
  }
});

// 2. cypress.env.json（不提交到 git）
{
  "apiUrl": "https://api.staging.com",
  "password": "secret123",
  "apiKey": "abc-xyz-123"
}

// 3. 命令行
// npx cypress run --env apiUrl=https://prod.api.com,username=admin

// 4. 系统环境变量（需加 CYPRESS_ 前缀）
// export CYPRESS_API_URL=https://api.example.com
// export CYPRESS_USERNAME=testuser

// 在测试中使用
describe('环境变量使用', () => {
  it('访问环境变量', () => {
    const apiUrl = Cypress.env('apiUrl');
    const username = Cypress.env('username');

    cy.log(`API URL: ${apiUrl}`);
    cy.log(`Username: ${username}`);

    cy.request(`${apiUrl}/users`).then((response) => {
      expect(response.status).to.eq(200);
    });
  });

  it('设置环境变量', () => {
    // 运行时设置
    Cypress.env('newKey', 'newValue');
    expect(Cypress.env('newKey')).to.eq('newValue');
  });

  it('获取所有环境变量', () => {
    const allEnv = Cypress.env();
    cy.log(JSON.stringify(allEnv));
  });
});
```

```javascript
// cypress/reporters/custom-reporter.js
const Mocha = require('mocha');

class CustomReporter extends Mocha.reporters.Base {
  constructor(runner, options) {
    super(runner, options);

    const stats = this.stats;
    const reporterOptions = options.reporterOptions || {};

    runner.on('start', () => {
      console.log('\n=== 测试开始 ===\n');
    });

    runner.on('pass', (test) => {
      console.log(`✓ ${test.title} (${test.duration}ms)`);
    });

    runner.on('fail', (test, err) => {
      console.log(`✗ ${test.title}`);
      console.log(`  错误: ${err.message}`);
    });

    runner.on('end', () => {
      console.log('\n=== 测试结束 ===');
      console.log(`通过: ${stats.passes}/${stats.tests}`);
      console.log(`失败: ${stats.failures}`);
      console.log(`耗时: ${stats.duration}ms`);
    });
  }
}

module.exports = CustomReporter;
```

## 8. 如何优化 Cypress 性能？

**答案**:

通过配置和最佳实践优化 Cypress 测试性能。

```javascript
// cypress.config.js - 性能优化配置
module.exports = defineConfig({
  e2e: {
    // 1. 视频和截图优化
    video: false,  // CI 环境可以启用
    videoCompression: 32,
    screenshotOnRunFailure: true,

    // 2. 内存管理
    numTestsKeptInMemory: 10,  // 减少内存占用（默认 50）
    experimentalMemoryManagement: true,

    // 3. 测试隔离
    testIsolation: true,  // 每个测试后重载页面

    // 4. 超时优化
    defaultCommandTimeout: 8000,
    pageLoadTimeout: 30000,

    // 5. 浏览器优化
    chromeWebSecurity: false,  // 禁用安全检查加快速度

    setupNodeEvents(on, config) {
      // 浏览器启动优化
      on('before:browser:launch', (browser, launchOptions) => {
        if (browser.name === 'chrome') {
          launchOptions.args.push('--disable-dev-shm-usage');
          launchOptions.args.push('--disable-gpu');
          launchOptions.args.push('--no-sandbox');
          launchOptions.args.push('--disable-extensions');
        }
        return launchOptions;
      });

      return config;
    }
  }
});
```

**测试代码优化**:

