## 1. cy.request() 和 cy.intercept() 有什么区别？

**答案**:

`cy.request()` 发送真实 HTTP 请求，`cy.intercept()` 拦截和模拟浏览器请求。

```javascript
// cy.request() - 直接 HTTP 请求
cy.request('GET', '/api/users')
  .its('status')
  .should('eq', 200);

// cy.intercept() - 拦截浏览器请求
cy.intercept('GET', '/api/users', {
  fixture: 'users.json'
});
cy.visit('/users-page');  // 页面请求会被拦截
```

**详细对比**:

| 特性 | cy.request() | cy.intercept() |
|-----|-------------|----------------|
| 请求方式 | 直接 HTTP 请求 | 拦截浏览器请求 |
| 执行位置 | Cypress 后端 | 浏览器内 |
| CORS 限制 | 无 | 有 |
| 用途 | API 测试、数据准备 | 模拟响应、监控请求 |
| 同源策略 | 不受限 | 受浏览器限制 |

```javascript
describe('request vs intercept', () => {
  it('cy.request() - API 测试', () => {
    // 直接请求 API，不通过浏览器
    cy.request({
      method: 'POST',
      url: '/api/login',
      body: {
        email: 'user@example.com',
        password: 'password123'
      }
    }).then((response) => {
      expect(response.status).to.eq(200);
      expect(response.body).to.have.property('token');

      // 使用返回的 token
      const token = response.body.token;
      window.localStorage.setItem('authToken', token);
    });
  });

  it('cy.intercept() - 模拟响应', () => {
    // 拦截浏览器发出的请求
    cy.intercept('GET', '/api/products', {
      statusCode: 200,
      body: [
        { id: 1, name: 'Product 1', price: 19.99 },
        { id: 2, name: 'Product 2', price: 29.99 }
      ]
    }).as('getProducts');

    cy.visit('/shop');
    cy.wait('@getProducts');

    cy.get('.product-card').should('have.length', 2);
  });

  it('结合使用', () => {
    // 1. 使用 request 创建测试数据
    cy.request('POST', '/api/users', {
      name: 'Test User',
      email: 'test@example.com'
    }).then((response) => {
      const userId = response.body.id;

      // 2. 使用 intercept 模拟后续请求
      cy.intercept('GET', `/api/users/${userId}`, {
        body: {
          id: userId,
          name: 'Test User',
          role: 'admin'  // 修改响应
        }
      });

      cy.visit(`/users/${userId}`);
      cy.contains('admin').should('be.visible');
    });
  });
});
```

**使用场景**:

```javascript

    // 创建测试数据

    // 纯 API 测试
    cy.request('/api/health').its('status').should('eq', 200);
  });

  it('认证状态设置', () => {
    // 通过 API 登录
    cy.request('POST', '/api/login', credentials)

        window.localStorage.setItem('token', token);

// cy.intercept() 适用场景
describe('intercept 场景', () => {
  it('模拟错误响应', () => {
    cy.intercept('POST', '/api/submit', {
      statusCode: 500,
      body: { error: 'Internal Server Error' }
    });

    cy.visit('/form');
    cy.get('button').click();
    cy.contains('Error').should('be.visible');
  });

  it('测试加载状态', () => {
    cy.intercept('GET', '/api/data', (req) => {
      req.reply({
        delay: 3000,  // 3 秒延迟
        body: { data: [] }
      });
    });

    cy.visit('/page');
    cy.get('.loading').should('be.visible');
    cy.get('.content', { timeout: 5000 }).should('be.visible');
  });

  it('验证请求参数', () => {
    cy.intercept('GET', '/api/search*').as('search');

    cy.visit('/search');
    cy.get('#query').type('cypress');
    cy.get('button').click();

    cy.wait('@search').its('request.url').should('include', 'q=cypress');
  });
});
```

**关键点**:
- `cy.request()`: 直接 HTTP 请求，不通过浏览器
- `cy.intercept()`: 拦截浏览器请求，可模拟响应
- `cy.request()` 用于 API 测试和数据准备
- `cy.intercept()` 用于模拟和监控网络请求
- 两者常结合使用

## 2. 如何使用 cy.intercept() 的高级功能？

**答案**:

`cy.intercept()` 支持请求匹配、响应修改、动态处理等高级功能。

```javascript
// 1. 基本拦截
cy.intercept('GET', '/api/users').as('getUsers');

// 2. 匹配对象
cy.intercept({
  method: 'POST',
  url: '/api/users',
  headers: {
    'content-type': 'application/json'
  }
}).as('createUser');

// 3. 通配符和正则
cy.intercept('GET', '/api/users/*').as('getUserById');
cy.intercept('GET', /\/api\/products\/\d+/).as('getProduct');

// 4. 动态响应
cy.intercept('GET', '/api/users', (req) => {
  req.reply({
    statusCode: 200,
    body: {
      users: [{ id: 1, name: 'John' }]
    },
    headers: {
      'X-Custom-Header': 'value'
    }
  });
});

// 5. 修改请求
cy.intercept('POST', '/api/data', (req) => {
  req.body.timestamp = Date.now();
  req.headers['X-Request-ID'] = 'test-123';
  req.continue();
});

// 6. 修改响应
cy.intercept('GET', '/api/config', (req) => {
  req.reply((res) => {
    res.body.feature_flag = true;
    res.body.theme = 'dark';
  });
});
```

```javascript
describe('intercept 高级功能', () => {
  it('条件拦截', () => {
    cy.intercept('GET', '/api/data', (req) => {
      // 根据查询参数决定响应
      if (req.query.type === 'admin') {
        req.reply({ fixture: 'admin-data.json' });

        req.reply({ fixture: 'user-data.json' });

    });

    cy.visit('/data?type=admin');
    cy.contains('Admin Dashboard').should('be.visible');
  });

  it('模拟网络延迟', () => {
    cy.intercept('GET', '/api/slow', (req) => {
      req.reply({
        delay: 5000,  // 5 秒延迟
        body: { data: 'slow response' }
      });
    }).as('slowRequest');

    cy.visit('/page');
    cy.get('.loading').should('be.visible');

    cy.wait('@slowRequest', { timeout: 10000 });
    cy.get('.content').should('contain', 'slow response');
  });

  it('模拟网络错误', () => {
    cy.intercept('GET', '/api/data', {
      forceNetworkError: true
    }).as('networkError');

    cy.visit('/page');
    cy.wait('@networkError');

    cy.get('.error-message')
      .should('be.visible')
      .and('contain', 'Network error');
  });

  it('序列响应（多次调用不同响应）', () => {
    let requestCount = 0;

    cy.intercept('GET', '/api/status', (req) => {
      requestCount++;

      if (requestCount === 1) {
        req.reply({ status: 'pending' });
      } else if (requestCount === 2) {
        req.reply({ status: 'processing' });
      } else {
        req.reply({ status: 'completed' });
      }
    }).as('getStatus');

    cy.visit('/status-page');
    cy.wait('@getStatus');
    cy.contains('pending').should('be.visible');

    cy.get('.refresh-btn').click();
    cy.wait('@getStatus');
    cy.contains('processing').should('be.visible');

    cy.get('.refresh-btn').click();
    cy.wait('@getStatus');
    cy.contains('completed').should('be.visible');
  });

  it('修改请求头', () => {
    cy.intercept('GET', '/api/protected', (req) => {
      req.headers['authorization'] = 'Bearer fake-token';
      req.continue();
    }).as('getProtected');

    cy.visit('/protected-page');
    cy.wait('@getProtected');
  });

  it('修改响应状态码', () => {
    cy.intercept('POST', '/api/submit', (req) => {
      req.reply((res) => {
        // 将 200 改为 201
        res.statusCode = 201;
        res.body.message = 'Created successfully';
      });
    });

    cy.visit('/form');
    cy.get('button').click();
    cy.contains('Created successfully').should('be.visible');
  });

  it('记录和验证请求', () => {
    const requests = [];

    cy.intercept('POST', '/api/log', (req) => {
      requests.push(req.body);
      req.reply({ statusCode: 200 });
    }).as('logEvent');

    cy.visit('/page');
    cy.get('.action-1').click();
    cy.get('.action-2').click();

    cy.wait('@logEvent');
    cy.wait('@logEvent');

    cy.wrap(null).then(() => {
      expect(requests).to.have.length(2);
      expect(requests[0].action).to.eq('action-1');
      expect(requests[1].action).to.eq('action-2');
    });
  });

  it('动态 fixture 选择', () => {
    cy.intercept('GET', '/api/user/*', (req) => {
      const userId = req.url.split('/').pop();
      req.reply({ fixture: `users/user-${userId}.json` });
    });

    cy.visit('/users/123');
    cy.contains('User 123 Profile').should('be.visible');
  });

  it('模拟分页', () => {
    cy.intercept('GET', '/api/items*', (req) => {
      const page = req.query.page || 1;
      const limit = req.query.limit || 10;

      req.reply({
        statusCode: 200,
        body: {
          items: Array.from({ length: limit }, (_, i) => ({
            id: (page - 1) * limit + i + 1,
            name: `Item ${(page - 1) * limit + i + 1}`
          })),
          page: parseInt(page),
          totalPages: 5
        }
      });
    }).as('getItems');

    cy.visit('/items');
    cy.wait('@getItems');
    cy.get('.item').should('have.length', 10);

    cy.get('.next-page').click();
    cy.wait('@getItems');
    cy.get('.item').first().should('contain', 'Item 11');
  });

  it('条件转发到真实 API', () => {
    cy.intercept('GET', '/api/data', (req) => {
      if (Cypress.env('USE_REAL_API')) {
        // 转发到真实 API
        req.continue();
      } else {
        // 使用 mock 数据
        req.reply({ fixture: 'data.json' });
      }
    });
  });
});
```

## 3. 如何测试 API 端点？

**答案**:

使用 `cy.request()` 进行 API 端点测试。

```javascript
describe('API 测试', () => {
  it('GET 请求', () => {
    cy.request('GET', '/api/users')
      .then((response) => {
        expect(response.status).to.eq(200);
        expect(response.body).to.be.an('array');
        expect(response.body).to.have.length.gt(0);
      });
  });

  it('POST 请求', () => {
    cy.request({
      method: 'POST',
      url: '/api/users',
      body: {
        name: 'John Doe',
        email: 'john@example.com',
        role: 'user'
      }
    }).then((response) => {
      expect(response.status).to.eq(201);
      expect(response.body).to.have.property('id');
      expect(response.body.name).to.eq('John Doe');
    });
  });

  it('PUT 请求', () => {
    cy.request('PUT', '/api/users/123', {
      name: 'Jane Doe'
    }).then((response) => {
      expect(response.status).to.eq(200);
      expect(response.body.name).to.eq('Jane Doe');
    });
  });

  it('DELETE 请求', () => {
    cy.request('DELETE', '/api/users/123')
      .its('status')
      .should('eq', 204);
  });
});
```

```javascript
        });
    });

      }).then((response) => {
        expect(response.status).to.eq(404);
      });
    });

    it('应该支持分页', () => {
      cy.request('GET', '/api/users?page=1&limit=10')
        .then((response) => {
          expect(response.status).to.eq(200);
          expect(response.body.data).to.have.length.lte(10);

    it('应该支持搜索', () => {
      cy.request('GET', '/api/users?search=test')
        .then((response) => {
          expect(response.status).to.eq(200);
          response.body.forEach((user) => {
            expect(user.name.toLowerCase()).to.include('test');
          });
        });

  });

    it('应该更新用户信息', () => {

          name: 'Updated Name'

        .should('eq', 204);

      cy.request({
        method: 'GET',
        url: `/api/users/${userId}`,
        failOnStatusCode: false
      }).its('status').should('eq', 404);
    });
  });
});
```

## 4. 如何处理 CORS 和跨域请求？

**答案**:

Cypress 可以禁用 Web 安全或配置代理来处理 CORS。

```javascript
// cypress.config.js - 禁用 Web 安全（不推荐生产使用）
module.exports = defineConfig({
  e2e: {
    chromeWebSecurity: false  // 禁用同源策略
  }
});

// 测试中访问跨域资源
describe('跨域测试', () => {
  it('访问跨域 API', () => {
    cy.visit('https://example.com');
    cy.request('https://api.another-domain.com/data')
      .its('status')
      .should('eq', 200);
  });
});
```

**使用代理配置**:

```javascript
// cypress.config.js
module.exports = defineConfig({
  e2e: {
    // 配置代理
    setupNodeEvents(on, config) {
      // 使用中间件处理代理
      return require('./cypress/plugins/index.js')(on, config);
    }
  }
});

// cypress/plugins/index.js
const { defineConfig } = require('cypress');

module.exports = (on, config) => {
  // 配置代理规则
  config.env.apiProxy = 'https://api.external.com';
```

```javascript
describe('CORS 处理', () => {
  it('cy.request 不受 CORS 限制', () => {
    // cy.request 在 Cypress 后端执行，不受浏览器 CORS 限制
    cy.request({
      method: 'GET',
      url: 'https://api.external.com/data',
      headers: {
        'Authorization': 'Bearer token'
      }
    }).then((response) => {
      expect(response.status).to.eq(200);

      // 将数据注入页面
      cy.visit('/page');
      cy.window().then((win) => {
        win.externalData = response.body;
      });
    });
  });

  it('拦截并添加 CORS 头', () => {
    cy.intercept('GET', 'https://api.external.com/data', (req) => {
      req.reply((res) => {
        res.headers['access-control-allow-origin'] = '*';
        res.headers['access-control-allow-methods'] = 'GET, POST';
      });
    });

    cy.visit('/page');
    // 页面发起的跨域请求会被拦截并添加 CORS 头
  });
});
```

**处理预检请求**:

```javascript
describe('OPTIONS 预检请求', () => {
  beforeEach(() => {

  });
});
```

## 6. 如何处理 GraphQL 请求？

**答案**:

使用 `cy.request()` 或 `cy.intercept()` 测试 GraphQL。

```javascript
describe('GraphQL 测试', () => {
  it('Query 请求', () => {
    const query = `
      query GetUser($id: ID!) {
        user(id: $id) {
          id
          name
          email
        }
      }
    `;

    cy.request({
      method: 'POST',
      url: '/graphql',
      body: {
        query,
        variables: { id: '123' }
      }
    }).then((response) => {
      expect(response.status).to.eq(200);
      expect(response.body.data.user).to.have.property('id', '123');
      expect(response.body.data.user).to.have.property('name');
    });
  });

  it('Mutation 请求', () => {
    const mutation = `
      mutation CreateUser($input: CreateUserInput!) {
        createUser(input: $input) {
          id
          name
          email
        }
      }
    `;

    cy.request({
      method: 'POST',
      url: '/graphql',
      body: {
        query: mutation,
        variables: {
          input: {
            name: 'John Doe',
            email: 'john@example.com'
          }
        }
      }
    }).then((response) => {
      expect(response.status).to.eq(200);
      expect(response.body.data.createUser).to.have.property('id');
      expect(response.body.data.createUser.name).to.eq('John Doe');
    });
  });

  it('处理 GraphQL 错误', () => {
    const query = `
      query GetUser($id: ID!) {
        user(id: $id) {
          id
          name
        }
      }
    `;

    cy.request({
      method: 'POST',
      url: '/graphql',
      body: {
        query,
        variables: { id: 'invalid' }
      },
      failOnStatusCode: false
    }).then((response) => {
      expect(response.body.errors).to.exist;
      expect(response.body.errors[0].message).to.include('User not found');
    });
  });
});
```

**拦截 GraphQL 请求**:

```javascript
describe('拦截 GraphQL', () => {
  it('拦截特定查询', () => {
    cy.intercept('POST', '/graphql', (req) => {
      if (req.body.query.includes('GetUser')) {
        req.reply({
          data: {
            user: {
              id: '123',
              name: 'Mocked User',
              email: 'mocked@example.com'
            }
          }
        });
      }
    }).as('getUser');

    cy.visit('/user/123');
    cy.wait('@getUser');
    cy.contains('Mocked User').should('be.visible');
  });

  it('拦截多个 GraphQL 操作', () => {
    cy.intercept('POST', '/graphql', (req) => {
      const operationName = req.body.operationName;

      if (operationName === 'GetUsers') {
        req.alias = 'getUsers';
        req.reply({ fixture: 'graphql/users.json' });
      } else if (operationName === 'CreateUser') {
        req.alias = 'createUser';
        req.reply({ fixture: 'graphql/created-user.json' });
      }
    });

    cy.visit('/users');
    cy.wait('@getUsers');

    cy.get('.create-user-btn').click();
    cy.wait('@createUser');
  });

  it('模拟 GraphQL 错误', () => {
    cy.intercept('POST', '/graphql', {
      body: {
        errors: [
          {
            message: 'Unauthorized',
            extensions: {
              code: 'UNAUTHENTICATED'
            }
          }
        ]
      }
    });

    cy.visit('/protected-page');
    cy.contains('Please log in').should('be.visible');
  });
});
```
