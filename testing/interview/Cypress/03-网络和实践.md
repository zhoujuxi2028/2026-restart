## 1. 如何使用 `cy.intercept()` 拦截和 Mock 网络请求？

**答案：**
`cy.intercept()` 是 Cypress 用于拦截、监控和 Mock HTTP 请求的核心 API。

**代码示例：**
```javascript
// 基本拦截：监控请求
cy.intercept('GET', '/api/users').as('getUsers')
cy.visit('/users')
cy.wait('@getUsers')  // 等待请求完成

// Mock 响应：返回固定数据
cy.intercept('GET', '/api/users', {
  statusCode: 200,
  body: [
    { id: 1, name: '张三' },
    { id: 2, name: '李四' }
  ]
})

// Mock 错误响应
cy.intercept('POST', '/api/login', {
  statusCode: 401,
  body: { error: '用户名或密码错误' }
})

// 使用 fixture 文件
cy.intercept('GET', '/api/products', {
  fixture: 'products.json'
})

// 动态响应：根据请求修改
cy.intercept('GET', '/api/users/*', (req) => {
  req.reply((res) => {
    res.body.timestamp = Date.now()
    res.send()
  })
})

// 延迟响应：测试加载状态
cy.intercept('GET', '/api/data', (req) => {
  req.reply({
    delay: 2000,  // 延迟 2 秒
    body: { message: '加载完成' }
  })
})

// 实际应用：测试不同网络状态
describe('用户列表', () => {
  it('应该显示加载中状态', () => {
    cy.intercept('GET', '/api/users', (req) => {
      req.reply({ delay: 1000, body: [] })
    }).as('getUsers')

    cy.visit('/users')
    cy.get('[data-cy="loading"]').should('be.visible')
    cy.wait('@getUsers')
    cy.get('[data-cy="loading"]').should('not.exist')
  })

  it('应该处理请求失败', () => {
    cy.intercept('GET', '/api/users', {
      statusCode: 500,
      body: { error: '服务器错误' }
    })

    cy.visit('/users')
    cy.contains('加载失败').should('be.visible')
  })
})
```

## 2. 如何测试 API 请求和响应？

**答案：**
使用 `cy.request()` 直接发送 HTTP 请求，适合 API 测试和后端集成测试。

**代码示例：**
```javascript
// 基本 GET 请求
cy.request('GET', '/api/users')
  .its('status')
  .should('eq', 200)

// 完整请求配置
cy.request({
  method: 'POST',
  url: '/api/login',
  body: {
    username: 'testuser',
    password: 'password123'
  },
  headers: {
    'Content-Type': 'application/json'
  }
}).then((response) => {
  expect(response.status).to.eq(200)
  expect(response.body).to.have.property('token')
  expect(response.body.user.username).to.eq('testuser')
})

// 保存响应数据
cy.request('POST', '/api/login', {
  username: 'admin',
  password: 'admin123'
}).then((response) => {
  // 保存 token 用于后续请求
  window.localStorage.setItem('token', response.body.token)
})

// 使用 token 发送认证请求
cy.request({
  method: 'GET',
  url: '/api/profile',
  headers: {
    'Authorization': `Bearer ${window.localStorage.getItem('token')}`
  }
}).its('status').should('eq', 200)

// 实际应用：完整的 CRUD 测试
describe('用户 API', () => {
  let userId

  it('应该创建用户', () => {
    cy.request('POST', '/api/users', {
      name: '测试用户',
      email: 'test@example.com'
    }).then((response) => {
      expect(response.status).to.eq(201)
      expect(response.body).to.have.property('id')
      userId = response.body.id
    })
  })

  it('应该获取用户列表', () => {
    cy.request('GET', '/api/users')
      .its('body')
      .should('be.an', 'array')
      .and('have.length.greaterThan', 0)
  })

  it('应该更新用户', () => {
    cy.request('PUT', `/api/users/${userId}`, {
      name: '更新后的用户名'
    }).then((response) => {
      expect(response.status).to.eq(200)
      expect(response.body.name).to.eq('更新后的用户名')
    })
  })

  it('应该删除用户', () => {
    cy.request('DELETE', `/api/users/${userId}`)
      .its('status')
      .should('eq', 204)
  })
})
```

## 3. 如何配置 Cypress？常用配置项有哪些？

**答案：**
在 `cypress.config.js` 中配置全局设置，影响所有测试。

**代码示例：**
```javascript
// cypress.config.js
const { defineConfig } = require('cypress')

module.exports = defineConfig({
  e2e: {
    // 基础 URL
    baseUrl: 'http://localhost:3000',

    // 视口大小
    viewportWidth: 1280,
    viewportHeight: 720,

    // 超时配置
    defaultCommandTimeout: 4000,      // 命令超时（毫秒）
    requestTimeout: 5000,             // 请求超时
    responseTimeout: 30000,           // 响应超时
    pageLoadTimeout: 60000,           // 页面加载超时

    // 重试配置
    retries: {
      runMode: 2,    // cypress run 时重试次数
      openMode: 0    // cypress open 时重试次数
    },

    // 视频和截图
    video: true,                      // 录制视频
    screenshotOnRunFailure: true,     // 失败时截图
    videosFolder: 'cypress/videos',   // 视频目录
    screenshotsFolder: 'cypress/screenshots',  // 截图目录

    // 测试文件配置
    specPattern: 'cypress/e2e/**/*.cy.{js,jsx,ts,tsx}',
    excludeSpecPattern: ['**/__snapshots__/*', '**/__image_snapshots__/*'],

    // 环境变量
    env: {
      apiUrl: 'http://localhost:8080/api',
      username: 'testuser',
      password: 'testpass'
    },

    // 设置节点事件
    setupNodeEvents(on, config) {
      // 任务注册
      on('task', {
        log(message) {
          console.log(message)
          return null
        },
        clearDatabase() {
          // 清理数据库逻辑
          return null
        }
      })

      return config
    },
  },

  // 组件测试配置
  component: {
    devServer: {
      framework: 'react',
      bundler: 'vite',
    },
    specPattern: 'src/**/*.cy.{js,jsx,ts,tsx}',
  },
})

// 使用环境变量
cy.visit(Cypress.env('apiUrl'))
cy.request({
  url: Cypress.env('apiUrl') + '/users',
  headers: {
    'Authorization': `Bearer ${Cypress.env('authToken')}`
  }
})
```

**环境变量优先级：**
```javascript
// 1. cypress.config.js
env: { apiUrl: 'http://localhost:8080' }

// 2. cypress.env.json（不提交到版本控制）
{
  "apiUrl": "http://staging.example.com",
  "authToken": "secret-token-123"
}

// 3. 命令行参数（最高优先级）
npx cypress run --env apiUrl=http://production.example.com
```

## 4. 如何创建和使用自定义命令？

**答案：**
在 `cypress/support/commands.js` 中使用 `Cypress.Commands.add()` 创建可复用命令。

**代码示例：**
```javascript
// cypress/support/commands.js

// 简单自定义命令
Cypress.Commands.add('login', (username, password) => {
  cy.visit('/login')
  cy.get('[data-cy="username"]').type(username)
  cy.get('[data-cy="password"]').type(password)
  cy.get('[data-cy="submit"]').click()
})

// 使用 API 登录（更快）
Cypress.Commands.add('loginByApi', (username, password) => {
  cy.request('POST', '/api/login', {
    username,
    password
  }).then((response) => {
    window.localStorage.setItem('token', response.body.token)
  })
})

// 父命令：直接调用
Cypress.Commands.add('clickButton', (label) => {
  cy.contains('button', label).click()
})

// 子命令：链式调用
Cypress.Commands.add('selectDropdown', { prevSubject: 'element' }, (subject, value) => {
  cy.wrap(subject).click()
  cy.contains(value).click()
})

// 双重命令：可选链式调用
Cypress.Commands.add('console', { prevSubject: 'optional' }, (subject, method = 'log') => {
  if (subject) {
    console[method]('Subject:', subject)
    return subject
  } else {
    console[method]('Called without subject')
  }
})

// 实际应用：复杂业务逻辑封装
Cypress.Commands.add('createUser', (userData) => {
  cy.request('POST', '/api/users', userData).then((response) => {
    return response.body
  })
})

Cypress.Commands.add('addToCart', (productId, quantity = 1) => {
  cy.get(`[data-cy="product-${productId}"]`).within(() => {
    cy.get('[data-cy="quantity"]').clear().type(quantity)
    cy.get('[data-cy="add-to-cart"]').click()
  })
  cy.get('[data-cy="cart-count"]').should('contain', quantity)
})

// 使用自定义命令
describe('用户流程', () => {
  beforeEach(() => {
    cy.loginByApi('testuser', 'password123')
  })

  it('应该能够购买商品', () => {
    cy.visit('/products')
    cy.addToCart('product-1', 2)
    cy.addToCart('product-2', 1)
    cy.clickButton('结算')
    cy.url().should('include', '/checkout')
  })
})

// 子命令使用
cy.get('[data-cy="country-select"]')
  .selectDropdown('中国')
```

## 5. 如何使用 Fixture 管理测试数据？

**答案：**
Fixture 用于存储静态测试数据，存放在 `cypress/fixtures/` 目录。

**代码示例：**
```javascript
// cypress/fixtures/users.json
{
  "admin": {
    "username": "admin",
    "password": "admin123",
    "role": "administrator"
  },
  "testUser": {
    "username": "testuser",
    "password": "test123",
    "role": "user"
  },
  "users": [
    { "id": 1, "name": "张三", "email": "zhangsan@example.com" },
    { "id": 2, "name": "李四", "email": "lisi@example.com" }
  ]
}

// 使用 fixture 数据
describe('用户管理', () => {
  let testData

  // 方法 1：在 before 中加载
  before(() => {
    cy.fixture('users.json').then((data) => {
      testData = data
    })
  })

  it('应该使用管理员登录', () => {
    cy.login(testData.admin.username, testData.admin.password)
  })

  // 方法 2：直接在测试中加载
  it('应该显示用户列表', () => {
    cy.fixture('users.json').then((data) => {
      cy.intercept('GET', '/api/users', {
        statusCode: 200,
        body: data.users
      })
    })

    cy.visit('/users')
    cy.get('[data-cy="user-item"]').should('have.length', 2)
  })

  // 方法 3：使用 as 别名
  it('应该创建新用户', () => {
    cy.fixture('users.json').as('userData')
    cy.get('@userData').then((data) => {
      cy.get('[data-cy="name"]').type(data.testUser.username)
    })
  })
})

// 与 intercept 结合使用
cy.intercept('GET', '/api/products', {
  fixture: 'products.json'  // 直接引用 fixture 文件
})

// 动态生成 fixture 数据
// cypress/fixtures/dynamic-users.js
module.exports = (count = 10) => {
  return Array.from({ length: count }, (_, i) => ({
    id: i + 1,
    name: `用户${i + 1}`,
    email: `user${i + 1}@example.com`
  }))
}

// 使用动态数据
cy.fixture('dynamic-users.js').then((generateUsers) => {
  const users = generateUsers(50)
  cy.intercept('GET', '/api/users', { body: users })
})
```

## 6. 如何处理文件上传和下载？

**答案：**
使用 `selectFile()` 上传文件，使用插件或请求拦截处理下载。

**代码示例：**
```javascript
// 文件上传
describe('文件上传', () => {
  it('应该上传图片', () => {
    // 从 fixtures 目录选择文件
    cy.get('[data-cy="file-input"]')
      .selectFile('cypress/fixtures/avatar.png')

    cy.get('[data-cy="upload-btn"]').click()
    cy.contains('上传成功').should('be.visible')
  })

  it('应该拖放上传文件', () => {
    cy.get('[data-cy="drop-zone"]')
      .selectFile('cypress/fixtures/document.pdf', {
        action: 'drag-drop'  // 模拟拖放
      })
  })

  it('应该上传多个文件', () => {
    cy.get('[data-cy="file-input"]')
      .selectFile([
        'cypress/fixtures/image1.jpg',
        'cypress/fixtures/image2.jpg'
      ])
  })

  it('应该从 Buffer 上传', () => {
    cy.fixture('avatar.png', 'base64').then((fileContent) => {
      cy.get('[data-cy="file-input"]')
        .selectFile({
          contents: Cypress.Buffer.from(fileContent, 'base64'),
          fileName: 'avatar.png',
          mimeType: 'image/png'
        })
    })
  })
})

// 文件下载验证
describe('文件下载', () => {
  it('应该下载 CSV 文件', () => {
    // 拦截下载请求
    cy.intercept('GET', '/api/export/users.csv').as('downloadFile')

    cy.get('[data-cy="export-btn"]').click()
    cy.wait('@downloadFile').then((interception) => {
      expect(interception.response.statusCode).to.eq(200)
      expect(interception.response.headers['content-type'])
        .to.include('text/csv')
    })
  })

  it('应该验证下载的文件内容', () => {
    cy.get('[data-cy="download-btn"]').click()

    // 使用 readFile 验证下载的文件
    const downloadsFolder = Cypress.config('downloadsFolder')
    cy.readFile(`${downloadsFolder}/report.pdf`).should('exist')
  })
})
```

## 7. 如何在 CI/CD 中运行 Cypress 测试？

**答案：**
配置 CI/CD 流水线运行无头测试，生成报告和视频。

**代码示例：**
```yaml
# .github/workflows/cypress.yml (GitHub Actions)
name: Cypress Tests

on: [push, pull_request]

jobs:
  cypress-run:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout
        uses: actions/checkout@v3

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'

      - name: Install dependencies
        run: npm ci

      - name: Start application
        run: npm start &

      - name: Wait for server
        run: npx wait-on http://localhost:3000

      - name: Run Cypress tests
        run: npx cypress run --browser chrome

      - name: Upload screenshots
        uses: actions/upload-artifact@v3
        if: failure()
        with:
          name: cypress-screenshots
          path: cypress/screenshots

      - name: Upload videos
        uses: actions/upload-artifact@v3
        if: always()
        with:
          name: cypress-videos
          path: cypress/videos

# 使用 Cypress Dashboard
- name: Run Cypress with Dashboard
  run: npx cypress run --record --key ${{ secrets.CYPRESS_RECORD_KEY }}
  env:
    CYPRESS_RECORD_KEY: ${{ secrets.CYPRESS_RECORD_KEY }}
```

```javascript
// package.json - CI 脚本
{
  "scripts": {
    "test": "cypress run",
    "test:chrome": "cypress run --browser chrome",
    "test:firefox": "cypress run --browser firefox",
    "test:headed": "cypress run --headed",
    "test:ci": "start-server-and-test start http://localhost:3000 cypress:run",
    "cypress:run": "cypress run --config video=true"
  }

// 并行执行测试（需要 Cypress Dashboard）
npx cypress run --record --parallel --ci-build-id $CI_BUILD_ID

// Docker 中运行
// Dockerfile
FROM cypress/included:12.0.0

COPY . .
RUN npm ci
CMD ["npx", "cypress", "run"]
```

## 8. Cypress 测试的最佳实践有哪些？

**答案：**
遵循最佳实践可以提高测试的可维护性、可靠性和执行速度。

**核心原则：**
```javascript
// ✅ 1. 使用 data-cy 属性
<button data-cy="submit-btn">提交</button>
cy.get('[data-cy="submit-btn"]').click()

// ❌ 避免使用易变的选择器
cy.get('.btn.btn-primary.submit')  // 样式改变会导致失败

// ✅ 2. 不要使用 cy.wait(固定时间)
cy.wait(5000)  // ❌ 不可靠

cy.get('[data-cy="result"]', { timeout: 10000 })
  .should('be.visible')  // ✅ 使用自动等待

// ✅ 3. 每个测试独立
beforeEach(() => {
  cy.visit('/')
  cy.loginByApi('testuser', 'pass123')
})

// ❌ 避免测试间依赖
it('test 1', () => {
  cy.createUser()  // 创建用户
})
it('test 2', () => {
  cy.deleteUser()  // 依赖 test 1 ❌
})

// ✅ 4. 使用 API 做准备工作
cy.request('POST', '/api/users', userData)  // 快速创建数据
cy.visit('/users')  // 然后测试 UI

// ❌ 避免纯 UI 操作
cy.visit('/users/new')
cy.get('[data-cy="name"]').type('张三')  // 慢
cy.get('[data-cy="submit"]').click()

// ✅ 5. 合理组织测试
describe('用户管理', () => {
  context('管理员权限', () => {
    beforeEach(() => cy.loginAsAdmin())
    it('应该能够删除用户', () => {})
  })

  context('普通用户权限', () => {
    beforeEach(() => cy.loginAsUser())
    it('不应该看到删除按钮', () => {})
  })
})

// ✅ 6. 使用自定义命令
Cypress.Commands.add('loginByApi', (username, password) => {
  cy.request('POST', '/api/login', { username, password })
    .then((response) => {
      window.localStorage.setItem('token', response.body.token)
    })
})

// ✅ 7. Mock 外部依赖
cy.intercept('GET', '/api/weather', {
  fixture: 'weather.json'
})

// ✅ 8. 使用有意义的断言
cy.get('[data-cy="error"]')
  .should('be.visible')
  .and('contain', '用户名不能为空')
  .and('have.class', 'error-message')
```

## 10. Cypress 常见陷阱和解决方案是什么？

**答案：**
了解常见错误可以避免调试时间浪费。

**常见问题：**
```javascript
// ❌ 陷阱 1：在 Cypress 命令前使用异步代码
it('错误示例', async () => {
  const data = await fetchData()  // ❌ 不要混用
  cy.get('[data-cy="input"]').type(data)
})

// ✅ 解决方案：使用 .then()
it('正确示例', () => {
  cy.task('fetchData').then((data) => {
    cy.get('[data-cy="input"]').type(data)
  })
})

// ❌ 陷阱 2：直接使用 Cypress 命令的返回值
const button = cy.get('[data-cy="button"]')  // ❌ 返回 Chainable
button.click()  // 不会按预期工作

// ✅ 解决方案：链式调用
cy.get('[data-cy="button"]').click()

// ❌ 陷阱 3：在循环中使用 Cypress 命令
const items = [1, 2, 3]
items.forEach((item) => {
  cy.get(`[data-item="${item}"]`).click()  // ❌ 可能不按预期执行
})

// ✅ 解决方案：使用 .each()
cy.get('[data-cy="item"]').each(($item) => {
  cy.wrap($item).click()
})

// ❌ 陷阱 4：测试间共享状态
let sharedData

it('test 1', () => {
  cy.get('[data-cy="value"]').then(($el) => {
    sharedData = $el.text()  // ❌ 不可靠
  })
})

it('test 2', () => {
  console.log(sharedData)  // 可能为 undefined
})

// ✅ 解决方案：使用 alias 或重新获取
it('test 1', () => {
  cy.get('[data-cy="value"]')
    .invoke('text')
    .as('sharedData')
})

it('test 2', () => {
  cy.get('@sharedData').then((data) => {
    console.log(data)
  })
})

// ❌ 陷阱 5：条件测试不稳定
cy.get('body').then(($body) => {
  if ($body.find('.modal').length > 0) {
    cy.get('.modal').click()  // ❌ 可能找不到
  }
})

// ✅ 解决方案：明确断言
cy.get('body').then(($body) => {
  if ($body.find('.modal').length > 0) {
    cy.get('.modal').should('be.visible').click()
  }
})

// ❌ 陷阱 6：忽略命令超时
cy.get('[data-cy="slow-element"]').click()  // 4 秒后超时

// ✅ 解决方案：增加超时或使用 should
cy.get('[data-cy="slow-element"]', { timeout: 10000 })
  .should('be.visible')
  .click()

// ❌ 陷阱 7：不当使用 cy.wait()
cy.wait(3000)  // ❌ 固定等待不可靠

// ✅ 解决方案：等待特定条件
cy.intercept('GET', '/api/data').as('getData')
cy.wait('@getData')

// 或使用 should 重试
cy.get('[data-cy="result"]').should('contain', '加载完成')
```
