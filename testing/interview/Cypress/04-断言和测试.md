## 1. Cypress 支持哪些断言库和断言风格？

**答案**:

Cypress 内置 Chai、Chai-jQuery 和 Sinon-Chai 断言库。

```javascript
// 1. Chai BDD 断言
expect(value).to.equal(42);
expect(arr).to.have.length(5);
expect(obj).to.have.property('name');

// 2. Chai-jQuery 断言（DOM 元素）
cy.get('.element')
  .should('be.visible')
  .and('have.class', 'active')
  .and('contain', 'text');

// 3. Sinon-Chai 断言（spy/stub）
expect(spy).to.be.called;
expect(stub).to.be.calledWith('arg');

// 4. 传统 TDD 断言
assert.equal(value, 42);
assert.isTrue(condition);
assert.isObject(obj);
```

**断言风格对比**:

```javascript
describe('断言风格', () => {
  it('BDD 风格 - expect', () => {
    const value = 42;
    const arr = [1, 2, 3];
    const obj = { name: 'John', age: 30 };

    // 相等性
    expect(value).to.equal(42);
    expect(value).to.eq(42);
    expect(arr).to.deep.equal([1, 2, 3]);

    // 类型
    expect(value).to.be.a('number');
    expect(arr).to.be.an('array');
    expect(obj).to.be.an('object');

    // 真值
    expect(true).to.be.true;
    expect(false).to.be.false;
    expect(null).to.be.null;
    expect(undefined).to.be.undefined;

    // 包含
    expect(arr).to.include(2);
    expect('hello').to.include('ell');
    expect(obj).to.have.property('name');

    // 长度
    expect(arr).to.have.length(3);
    expect('hello').to.have.lengthOf(5);

    // 大小比较
    expect(value).to.be.gt(40);  // greater than
    expect(value).to.be.gte(42); // greater than or equal
    expect(value).to.be.lt(50);  // less than
    expect(value).to.be.lte(42); // less than or equal
  });

  it('TDD 风格 - assert', () => {
    const value = 42;
    const arr = [1, 2, 3];

    assert.equal(value, 42);
    assert.strictEqual(value, 42);
    assert.deepEqual(arr, [1, 2, 3]);

    assert.isNumber(value);
    assert.isArray(arr);
    assert.isTrue(value === 42);
    assert.isNotNull(value);

    assert.include(arr, 2);
    assert.lengthOf(arr, 3);
  });

  it('Cypress 链式断言', () => {
    cy.get('.element')
      .should('exist')
      .and('be.visible')
      .and('have.text', 'Hello')
      .and('have.css', 'color', 'rgb(255, 0, 0)');
  });
});
```

**常用 Chai-jQuery 断言**:

| 断言 | 说明 | 示例 |
|-----|------|------|
| `be.visible` | 元素可见 | `cy.get('.el').should('be.visible')` |
| `be.hidden` | 元素隐藏 | `cy.get('.el').should('be.hidden')` |
| `exist` | 元素存在 | `cy.get('.el').should('exist')` |
| `have.class` | 包含类名 | `cy.get('.el').should('have.class', 'active')` |
| `have.text` | 文本内容 | `cy.get('.el').should('have.text', 'Hello')` |
| `contain` | 包含文本 | `cy.get('.el').should('contain', 'Hell')` |
| `have.value` | 输入值 | `cy.get('input').should('have.value', 'text')` |
| `be.checked` | 已选中 | `cy.get('input').should('be.checked')` |
| `be.disabled` | 已禁用 | `cy.get('button').should('be.disabled')` |
| `have.attr` | 包含属性 | `cy.get('a').should('have.attr', 'href')` |

**关键点**:
- 内置 Chai、Chai-jQuery、Sinon-Chai
- 支持 BDD (`expect`) 和 TDD (`assert`) 风格
- Cypress 命令使用 `.should()` 和 `.and()`
- Chai-jQuery 专门用于 DOM 断言
- 推荐使用 BDD 风格和链式断言

## 2. .should() 和 expect() 有什么区别？何时使用？

**答案**:

`.should()` 自动重试，`expect()` 立即断言。

```javascript
// .should() - 自动重试
cy.get('.element')
  .should('be.visible')  // 重试直到可见或超时
  .and('contain', 'text');

// expect() - 立即断言
cy.get('.element').then(($el) => {
  expect($el).to.be.visible;  // 立即检查，不重试
});

// 区别示例
cy.get('.loading').should('not.exist');  // ✅ 等待 loading 消失

cy.get('.loading').then(($loading) => {
  expect($loading).to.not.exist;  // ❌ 可能在消失前检查
});
```

**使用场景对比**:

```javascript
describe('should vs expect', () => {
  it('should() - 用于 DOM 元素（自动重试）', () => {
    // ✅ 推荐：等待异步变化
    cy.get('.counter')
      .should('contain', '5');  // 重试直到计数器显示 5

    cy.get('.status')
      .should('have.class', 'active');  // 重试直到有类名

    cy.get('.list-item')
      .should('have.length', 10);  // 重试直到有 10 个元素
  });

  it('expect() - 用于值和对象（不重试）', () => {
    // ✅ 适合：已知值的断言
    cy.request('/api/user').then((response) => {
      expect(response.status).to.eq(200);
      expect(response.body).to.have.property('id');
      expect(response.body.name).to.be.a('string');
    });

    cy.wrap({ name: 'John', age: 30 }).then((obj) => {
      expect(obj.name).to.eq('John');
      expect(obj.age).to.be.gt(18);
    });
  });

  it('should() 函数形式（可重试）', () => {
    // ✅ 复杂验证需要重试
    cy.get('.items').should(($items) => {
      expect($items).to.have.length(5);
      expect($items.first()).to.contain('First');
      expect($items.last()).to.contain('Last');
    });

    cy.get('.price').should(($price) => {
      const text = $price.text();
      const price = parseFloat(text.replace('$', ''));
      expect(price).to.be.gt(0);
      expect(price).to.be.lt(1000);
    });
  });

  it('混合使用', () => {
    cy.get('.user-card')
      .should('be.visible')  // should 等待可见
      .then(($card) => {
        // then 内使用 expect
        const name = $card.find('.name').text();
        const role = $card.find('.role').text();

        expect(name).to.match(/^[A-Z]/);  // 名字首字母大写
        expect(role).to.be.oneOf(['Admin', 'User', 'Guest']);
      });
  });
});
```

**何时使用 should()**:

```javascript
// ✅ DOM 元素状态验证
cy.get('.element').should('be.visible');

// ✅ 等待异步更新
cy.get('.counter').should('contain', '10');

// ✅ 链式多个断言
cy.get('.element')
  .should('exist')
  .and('be.visible')
  .and('have.text', 'Hello');

// ✅ 复杂条件需要重试
cy.get('.items').should(($items) => {
  expect($items.length).to.be.gt(0);
});
```

**何时使用 expect()**:

```javascript
// ✅ API 响应验证
cy.request('/api/data').then((response) => {
  expect(response.status).to.eq(200);
});

// ✅ 纯 JavaScript 值
cy.wrap(42).then((num) => {
  expect(num).to.be.gt(40);
});

// ✅ 在 .then() 回调中
cy.get('.element').then(($el) => {
  const text = $el.text();
  expect(text.length).to.be.gt(10);
});

// ✅ 复杂逻辑计算
cy.get('.prices').then(($prices) => {
  const prices = [...$prices].map(el => parseFloat(el.textContent));
  const total = prices.reduce((sum, p) => sum + p, 0);
  expect(total).to.be.gt(100);
});
```

**关键点**:
- `.should()` 自动重试，用于 DOM 元素
- `expect()` 不重试，用于值和对象
- `.should()` 可以接受函数参数实现复杂断言
- 优先使用 `.should()` 以利用自动重试
- `.then()` 回调中使用 `expect()`

## 2. 如何编写自定义断言？

**答案**:

使用 Chai 的 `addChaiPlugin` 或 `addMethod` 添加自定义断言。

```javascript
// cypress/support/commands.js

// 方法 1: 简单自定义断言
Cypress.Commands.add('beWithinRange', { prevSubject: true }, (subject, min, max) => {
  expect(subject).to.be.at.least(min);
  expect(subject).to.be.at.most(max);
  return subject;
});

// 使用
cy.wrap(42).should('beWithinRange', 40, 50);

// 方法 2: Chai 插件方式
chai.Assertion.addMethod('beEven', function() {
  const obj = this._obj;
  this.assert(
    obj % 2 === 0,
    'expected #{this} to be even',
    'expected #{this} to not be even'
  );
});

// 使用
expect(42).to.be.beEven;

// 方法 3: 复杂自定义断言
Cypress.Commands.add('haveValidEmail', { prevSubject: true }, (subject) => {
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  const email = subject.val ? subject.val() : subject.text();

  expect(email).to.match(emailRegex);
  return subject;
});

// 使用
cy.get('#email').should('haveValidEmail');
```

**实际应用示例**:

```javascript
describe('自定义断言', () => {
  // 日期范围断言
  Cypress.Commands.add('beDateInRange', { prevSubject: true }, (subject, startDate, endDate) => {
    const date = new Date(subject);
    const start = new Date(startDate);
    const end = new Date(endDate);

    expect(date.getTime()).to.be.at.least(start.getTime());
    expect(date.getTime()).to.be.at.most(end.getTime());

    return subject;
  });

  it('验证日期范围', () => {
    cy.wrap('2026-02-10').should('beDateInRange', '2026-01-01', '2026-12-31');
  });

  // 价格格式断言
  Cypress.Commands.add('beValidPrice', { prevSubject: true }, (subject) => {
    const text = subject.text ? subject.text() : subject;
    const priceRegex = /^\$\d+(\.\d{2})?$/;

    expect(text).to.match(priceRegex);
    return subject;
  });

  it('验证价格格式', () => {
    cy.get('.price').should('beValidPrice');
    cy.wrap('$19.99').should('beValidPrice');
  });

  // URL 断言
  Cypress.Commands.add('beValidUrl', { prevSubject: true }, (subject) => {
    try {
      new URL(subject);
      return subject;
    } catch (e) {
      throw new Error(`Invalid URL: ${subject}`);
    }
  });

  it('验证 URL', () => {
    cy.get('a').invoke('attr', 'href').should('beValidUrl');
  });

  // 元素集合断言
  Cypress.Commands.add('allContainText', { prevSubject: true }, (subject, text) => {
    cy.wrap(subject).each(($el) => {
      expect($el.text()).to.include(text);
    });
    return subject;
  });

  it('验证所有元素包含文本', () => {
    cy.get('.list-item').should('allContainText', 'Item');
  });

  // 响应时间断言
  Cypress.Commands.add('respondWithin', { prevSubject: true }, (subject, maxTime) => {
    const duration = subject.duration;
    expect(duration).to.be.lt(maxTime);
    return subject;
  });

  it('验证响应时间', () => {
    cy.request('/api/fast').should('respondWithin', 1000);
  });
});
```

**复杂自定义断言**:

```javascript
// 表单验证断言
Cypress.Commands.add('beValidForm', { prevSubject: true }, (subject) => {
  cy.wrap(subject).within(() => {
    // 所有必填字段已填充
    cy.get('input[required], select[required], textarea[required]').each(($el) => {
      expect($el.val()).to.not.be.empty;
    });

    // 没有验证错误
    cy.get('.error').should('not.exist');

    // 提交按钮可用
    cy.get('button[type="submit"]').should('not.be.disabled');
  });

  return subject;
});

// 使用
cy.get('form').should('beValidForm');

// 可访问性断言
Cypress.Commands.add('beAccessible', { prevSubject: true }, (subject) => {
  cy.wrap(subject).should('have.attr', 'role');
  cy.wrap(subject).should('have.attr', 'aria-label');
  return subject;
});

// JSON 结构断言
Cypress.Commands.add('matchSchema', { prevSubject: true }, (subject, schema) => {
  function validateSchema(obj, schema) {
    for (const key in schema) {
      expect(obj).to.have.property(key);
      expect(obj[key]).to.be.a(schema[key]);
    }
  }

  validateSchema(subject, schema);
  return subject;
});

// 使用
cy.request('/api/user').its('body').should('matchSchema', {
  id: 'number',
  name: 'string',
  email: 'string',
  active: 'boolean'
});
```

**TypeScript 类型定义**:

```typescript
// cypress/support/index.d.ts
declare namespace Cypress {
  interface Chainer<Subject> {
    (chainer: 'beWithinRange', min: number, max: number): Chainable<Subject>;
    (chainer: 'haveValidEmail'): Chainable<Subject>;
    (chainer: 'beValidPrice'): Chainable<Subject>;
    (chainer: 'beValidUrl'): Chainable<Subject>;
    (chainer: 'beValidForm'): Chainable<Subject>;
    (chainer: 'matchSchema', schema: object): Chainable<Subject>;
  }
}
```

**关键点**:
- 使用 `Cypress.Commands.add` 添加自定义断言
- `prevSubject: true` 表示需要前置主体
- 可以组合多个内置断言
- 返回 subject 以支持链式调用
- 提供 TypeScript 类型定义
- 封装常用验证逻辑

## 4. 如何组织测试套件和测试用例？

**答案**:

使用 `describe`、`context`、`it` 组织层次化测试结构。

```javascript
// 基本结构
describe('功能模块', () => {
  context('特定场景', () => {
    it('应该执行某个操作', () => {
      // 测试代码
    });
  });
});

// 实际示例
describe('用户认证', () => {
  beforeEach(() => {
    cy.visit('/login');
  });

  context('登录功能', () => {
    it('使用有效凭证应该成功登录', () => {
      cy.get('#email').type('user@example.com');
      cy.get('#password').type('password123');
      cy.get('button[type="submit"]').click();
      cy.url().should('include', '/dashboard');
    });

    it('使用无效凭证应该显示错误', () => {
      cy.get('#email').type('user@example.com');
      cy.get('#password').type('wrongpassword');
      cy.get('button[type="submit"]').click();
      cy.get('.error').should('contain', 'Invalid credentials');
    });
  });

  context('密码重置', () => {
    beforeEach(() => {
      cy.contains('Forgot password?').click();
    });

    it('应该发送重置邮件', () => {
      cy.get('#email').type('user@example.com');
      cy.get('button').contains('Send Reset Link').click();
      cy.contains('Email sent').should('be.visible');
    });
  });
});
```

## 5. 如何使用测试钩子（before、beforeEach、after、afterEach）？

**答案**:

钩子函数用于设置和清理测试环境。

```javascript
describe('测试钩子示例', () => {
  // 所有测试前执行一次
  before(() => {
    cy.log('=== 测试套件开始 ===');
    cy.task('db:seed');  // 初始化数据库
  });

  // 每个测试前执行
  beforeEach(() => {
    cy.log('--- 测试开始 ---');
    cy.visit('/page');
    cy.clearCookies();
    cy.clearLocalStorage();
  });

  // 每个测试后执行
  afterEach(() => {
    cy.log('--- 测试结束 ---');
    cy.screenshot();
  });

  // 所有测试后执行一次
  after(() => {
    cy.log('=== 测试套件结束 ===');
    cy.task('db:clean');  // 清理数据库
  });

  it('测试 1', () => {
    cy.get('.element').should('exist');
  });

  it('测试 2', () => {
    cy.get('.element').click();
  });
});

// 执行顺序：
// 1. before
// 2. beforeEach
// 3. 测试 1
// 4. afterEach
// 5. beforeEach
// 6. 测试 2
// 7. afterEach
// 8. after
```

**实际应用场景**:

```javascript
describe('用户认证流程', () => {
  let testUser;

  before(() => {
    // 创建测试用户（所有测试共享）
    cy.request('POST', '/api/users', {
      email: 'test@example.com',
      password: 'Test123!',
      name: 'Test User'
    }).then((response) => {
      testUser = response.body;
    });
  });

  beforeEach(() => {
    // 每个测试前登录
    cy.session('user-session', () => {
      cy.visit('/login');
      cy.get('#email').type(testUser.email);
      cy.get('#password').type('Test123!');
      cy.get('button[type="submit"]').click();
      cy.url().should('include', '/dashboard');
    });
  });

  afterEach(function() {
    // 测试失败时记录详细信息
    if (this.currentTest.state === 'failed') {
      cy.task('log', {
        test: this.currentTest.title,
        error: this.currentTest.err.message
      });
    }
  });

  after(() => {
    // 清理测试用户
    cy.request('DELETE', `/api/users/${testUser.id}`);
  });

  it('可以查看个人资料', () => {
    cy.visit('/profile');
    cy.contains(testUser.name).should('be.visible');
  });

  it('可以更新设置', () => {
    cy.visit('/settings');
    cy.get('#theme').select('dark');
    cy.get('button').contains('Save').click();
  });
});
```

**嵌套钩子**:

```javascript
    if (this.currentTest.state === 'failed') {

    }
```

## 6. 如何跳过测试或仅运行特定测试？

**答案**:

使用 `.skip` 和 `.only` 控制测试执行。

```javascript
// 跳过单个测试
it.skip('暂时跳过这个测试', () => {
  // 不会执行
});

// 跳过整个套件
describe.skip('跳过整个套件', () => {
  it('测试 1', () => {});
  it('测试 2', () => {});
});

// 仅运行一个测试
it.only('只运行这个测试', () => {
  cy.visit('/page');
});

// 仅运行一个套件
describe.only('只运行这个套件', () => {
  it('测试 1', () => {});
  it('测试 2', () => {});
});
```

**实际应用**:

```javascript
describe('功能测试', () => {
  // 调试时只运行当前开发的测试
  it.only('正在开发的功能', () => {
    cy.visit('/new-feature');
    cy.get('.new-element').should('exist');
  });

  // 已知失败的测试，暂时跳过
  it.skip('已知 bug，待修复', () => {
    // Bug #123: 登录失败
    cy.visit('/login');
    cy.get('#email').type('user@example.com');
  });

  // 正常测试
  it('完成的功能', () => {
    cy.visit('/page');
  });

  // 条件跳过
  const isCI = Cypress.env('CI');
  (isCI ? it.skip : it)('本地测试', () => {
    cy.visit('/localhost-only');
  });
});
```

**动态跳过**:

```javascript
describe('跨浏览器测试', () => {
  it('所有浏览器运行', () => {
    cy.visit('/page');
  });

  // 仅 Chrome 运行
  if (Cypress.isBrowser('chrome')) {
    it('Chrome 特定功能', () => {
      cy.visit('/chrome-feature');
    });
  }

  // 跳过 Firefox
  if (Cypress.isBrowser('firefox')) {
    it.skip('Firefox 暂不支持', () => {
      cy.visit('/unsupported');
    });
  }

  // 使用配置
  const skipTest = Cypress.env('environment') === 'production';

  (skipTest ? it.skip : it)('开发环境测试', () => {
    cy.visit('/dev-feature');
  });
});
```

**最佳实践**:

```javascript
// ✅ 好的做法：添加跳过原因
it.skip('跳过原因：等待 API v2 上线', () => {
  // 测试代码
});

// ✅ 使用 Jira ticket
it.skip('BUG-123: 修复登录验证问题', () => {
  // 测试代码
});

// ❌ 不好的做法：没有说明
it.skip('测试', () => {
  // 为什么跳过？
});

// ✅ 使用 .only 调试后记得删除
it.only('调试中...', () => {
  // 记得在提交前移除 .only
});

// ✅ 使用环境变量控制
const runSlow Tests = Cypress.env('RUN_SLOW_TESTS');

(runSlowTests ? describe : describe.skip)('慢速测试套件', () => {
  it('长时间运行的测试', () => {
    // 耗时测试
  });
});
```

**命令行控制**:

```bash
# 运行指定规格文件
npx cypress run --spec "cypress/e2e/login.cy.js"

# 运行多个文件
npx cypress run --spec "cypress/e2e/login.cy.js,cypress/e2e/register.cy.js"

# 使用通配符
npx cypress run --spec "cypress/e2e/auth/**/*.cy.js"

# 运行包含特定文本的测试（通过 grep 插件）
npx cypress run --env grep="登录"
```

**自定义跳过逻辑**:

```javascript
// cypress/support/commands.js

  if (Cypress.isBrowser(browser)) {
    cy.state('runnable').skip();

  cy.skipOn('firefox');
  cy.visit('/page');

// 条件运行

  if (!condition) {
    cy.state('runnable').skip();
```

**关键点**:
- `.skip` 跳过测试或套件
- `.only` 仅运行特定测试或套件
- 支持动态条件跳过
- 添加跳过原因注释
- 提交前移除 `.only`
- 使用命令行参数控制执行范围

## 8. 如何测试可访问性（Accessibility）？

**答案**:

使用 `cypress-axe` 插件进行可访问性测试。

```bash
# 安装
npm install --save-dev cypress-axe axe-core
```

```javascript
// cypress/support/e2e.js
import 'cypress-axe';

// 测试文件
describe('可访问性测试', () => {
  beforeEach(() => {
    cy.visit('/');
    cy.injectAxe();  // 注入 axe-core
  });

  it('页面应该没有可访问性问题', () => {
    cy.checkA11y();  // 检查整个页面
  });

  it('导航区域应该可访问', () => {
    cy.checkA11y('nav');  // 检查特定元素
  });

  it('忽略特定问题', () => {
    cy.checkA11y(null, {
      rules: {
        'color-contrast': { enabled: false }  // 忽略颜色对比度检查
      }
    });
  });

  it('仅检查特定规则', () => {
    cy.checkA11y(null, {
      runOnly: {
        type: 'tag',
        values: ['wcag2a', 'wcag2aa']  // 仅 WCAG 2 AA 标准
      }
    });
  });
});
```

**实际应用示例**:

```javascript

    cy.visit('/contact-form');
    cy.injectAxe();
  });

    cy.get('input[type="text"]').each(($input) => {

      cy.get(`#${id}`).should('be.visible').and('contain', 'Invalid email');
    });
```
