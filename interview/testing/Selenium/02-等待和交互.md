## 1. Selenium 的三种等待机制是什么？如何选择？

**答案：**
等待机制用于处理页面元素加载的时间差，避免 `NoSuchElementException` 异常。

| 等待类型 | 作用范围 | 优点 | 缺点 | 使用场景 |
|---------|---------|------|------|---------|
| 强制等待 | 全局暂停 | 简单 | 浪费时间、不智能 | 调试临时使用 |
| 隐式等待 | 全局生效 | 配置一次即可 | 可能增加整体执行时间 | 简单场景 |
| 显式等待 | 针对特定元素 | 灵活、智能 | 代码量稍多 | 生产环境推荐 |

```python
import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get("https://www.example.com")

# 1. 强制等待（time.sleep）- 不推荐
time.sleep(5)  # 无条件等待5秒
element = driver.find_element(By.ID, "username")
# ❌ 缺点：固定等待时间，即使元素已加载也会等待

# 2. 隐式等待（Implicit Wait）- 适合简单场景
driver.implicitly_wait(10)  # 设置全局等待时间10秒
# 在10秒内每隔0.5秒查找一次元素，找到就继续，超时抛异常
element = driver.find_element(By.ID, "username")
element.send_keys("testuser")
# ✅ 优点：全局生效，代码简洁
# ⚠️ 注意：只设置一次，影响所有 find_element 调用

# 3. 显式等待（Explicit Wait）- 推荐
wait = WebDriverWait(driver, 10)  # 最多等待10秒
element = wait.until(
    EC.presence_of_element_located((By.ID, "username"))
)
element.send_keys("testuser")
# ✅ 优点：针对特定元素，灵活控制

# 显式等待 - 自定义条件
wait = WebDriverWait(driver, 10, poll_frequency=0.5)  # 每0.5秒检查一次
element = wait.until(
    lambda d: d.find_element(By.ID, "username"),
    message="用户名输入框未找到"
)

# 显式等待 - 忽略特定异常
from selenium.common.exceptions import NoSuchElementException, StaleElementReferenceException

wait = WebDriverWait(
    driver,
    10,
    ignored_exceptions=[NoSuchElementException, StaleElementReferenceException]
)
element = wait.until(EC.element_to_be_clickable((By.ID, "submit")))

# 最佳实践：隐式等待 + 显式等待组合使用（不推荐混用，但实际项目中常见）
driver.implicitly_wait(5)  # 全局兜底
wait = WebDriverWait(driver, 15)  # 特殊元素使用显式等待
element = wait.until(EC.visibility_of_element_located((By.ID, "dynamic-content")))

driver.quit()
```

**选择建议：**
- 开发/调试阶段：可以临时使用 `time.sleep()`
- 简单项目：使用隐式等待
- 生产环境：**优先使用显式等待**（更精确、更可控）

## 2. 显式等待有哪些常用的 Expected Conditions？

**答案：**
Expected Conditions（EC）是 Selenium 提供的预定义等待条件，位于 `selenium.webdriver.support.expected_conditions` 模块。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get("https://www.example.com")
wait = WebDriverWait(driver, 10)

# 1. presence_of_element_located - 元素出现在DOM中（不管是否可见）
element = wait.until(
    EC.presence_of_element_located((By.ID, "hidden-div"))
)

# 2. visibility_of_element_located - 元素可见（高度和宽度>0）
element = wait.until(
    EC.visibility_of_element_located((By.ID, "visible-div"))
)

# 3. element_to_be_clickable - 元素可点击（可见且enabled）
button = wait.until(
    EC.element_to_be_clickable((By.ID, "submit-button"))
)
button.click()

# 4. invisibility_of_element_located - 元素不可见或不存在
wait.until(
    EC.invisibility_of_element_located((By.ID, "loading-spinner"))
)

# 5. text_to_be_present_in_element - 元素包含指定文本
wait.until(
    EC.text_to_be_present_in_element(
        (By.ID, "status"),
        "已完成"
    )
)

# 6. text_to_be_present_in_element_value - 元素value属性包含指定文本
wait.until(
    EC.text_to_be_present_in_element_value(
        (By.ID, "username"),
        "admin"
    )
)

# 7. frame_to_be_available_and_switch_to_it - iframe可用并切换
wait.until(
    EC.frame_to_be_available_and_switch_to_it((By.ID, "iframe1"))
)

# 8. alert_is_present - 弹窗出现
alert = wait.until(EC.alert_is_present())
alert.accept()

# 9. title_contains - 页面标题包含指定文本
wait.until(EC.title_contains("欢迎"))

# 10. title_is - 页面标题完全匹配
wait.until(EC.title_is("用户登录 - Example"))

# 11. url_contains - URL包含指定文本
wait.until(EC.url_contains("/dashboard"))

# 12. url_to_be - URL完全匹配
wait.until(EC.url_to_be("https://www.example.com/home"))

# 13. staleness_of - 元素从DOM中移除（处理元素刷新）
old_element = driver.find_element(By.ID, "refresh-content")
refresh_button = driver.find_element(By.ID, "refresh")
refresh_button.click()
wait.until(EC.staleness_of(old_element))

# 14. presence_of_all_elements_located - 所有元素出现
elements = wait.until(
    EC.presence_of_all_elements_located((By.CLASS_NAME, "product-item"))
)
print(f"找到 {len(elements)} 个商品")

# 15. visibility_of - 已找到的元素可见
element = driver.find_element(By.ID, "content")
wait.until(EC.visibility_of(element))

# 16. element_located_to_be_selected - 元素被选中（checkbox/radio）
wait.until(
    EC.element_located_to_be_selected((By.ID, "agree-checkbox"))
)

# 17. element_selection_state_to_be - 元素选中状态为指定值
wait.until(
    EC.element_selection_state_to_be(
        driver.find_element(By.ID, "agree-checkbox"),
        True  # True=选中，False=未选中
    )
)

# 18. number_of_windows_to_be - 窗口数量达到指定值
driver.find_element(By.LINK_TEXT, "打开新窗口").click()
wait.until(EC.number_of_windows_to_be(2))

# 自定义等待条件
def element_has_css_class(locator, css_class):
    """自定义条件：元素包含指定CSS类"""
    def _predicate(driver):
        element = driver.find_element(*locator)
        return css_class in element.get_attribute("class")
    return _predicate

wait.until(element_has_css_class((By.ID, "status"), "success"))

# 组合条件 - 任意条件满足（OR）
from selenium.webdriver.support import expected_conditions as EC

wait.until(
    lambda d: EC.presence_of_element_located((By.ID, "btn1"))(d) or
              EC.presence_of_element_located((By.ID, "btn2"))(d)
)

driver.quit()
```

**常用场景总结：**
- 点击按钮前：`element_to_be_clickable`
- 等待加载完成：`invisibility_of_element_located`（加载动画）
- 验证内容：`text_to_be_present_in_element`
- 等待页面跳转：`url_contains` 或 `url_to_be`
- 切换 iframe：`frame_to_be_available_and_switch_to_it`
- 处理弹窗：`alert_is_present`

## 3. 如何使用 ActionChains 执行复杂交互？

**答案：**
ActionChains 用于执行鼠标悬停、拖拽、右键点击、双击等复杂操作。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.common.keys import Keys

driver = webdriver.Chrome()
driver.get("https://www.example.com")

# 创建 ActionChains 对象
actions = ActionChains(driver)

# 1. 鼠标悬停（hover）
menu = driver.find_element(By.ID, "menu")
actions.move_to_element(menu).perform()

# 2. 鼠标悬停并点击子菜单
menu = driver.find_element(By.ID, "menu")
submenu = driver.find_element(By.ID, "submenu-item")
actions.move_to_element(menu).click(submenu).perform()

# 3. 双击
element = driver.find_element(By.ID, "double-click-target")
actions.double_click(element).perform()

# 4. 右键点击（上下文菜单）
element = driver.find_element(By.ID, "right-click-target")
actions.context_click(element).perform()

# 5. 拖拽元素
source = driver.find_element(By.ID, "draggable")
target = driver.find_element(By.ID, "droppable")
actions.drag_and_drop(source, target).perform()

# 6. 拖拽指定偏移量
element = driver.find_element(By.ID, "slider")
actions.drag_and_drop_by_offset(element, 100, 0).perform()  # 向右移动100像素

# 7. 点击并按住
element = driver.find_element(By.ID, "hold-button")
actions.click_and_hold(element).perform()
# 执行其他操作...
actions.release(element).perform()  # 释放

# 8. 组合键操作（Ctrl+A 全选）
input_field = driver.find_element(By.ID, "text-input")
actions.click(input_field).key_down(Keys.CONTROL).send_keys('a').key_up(Keys.CONTROL).perform()

# 9. 链式调用（多个操作连续执行）
username = driver.find_element(By.ID, "username")
password = driver.find_element(By.ID, "password")
submit = driver.find_element(By.ID, "submit")

actions.click(username)\
    .send_keys("testuser")\
    .click(password)\
    .send_keys("password123")\
    .click(submit)\
    .perform()

# 10. 滚动到元素（Selenium 4+）
element = driver.find_element(By.ID, "footer")
actions.scroll_to_element(element).perform()

# 11. 滚动指定距离
actions.scroll_by_amount(0, 500).perform()  # 向下滚动500像素

# 12. 悬停并等待子菜单出现
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

menu = driver.find_element(By.ID, "menu")
actions.move_to_element(menu).perform()

wait = WebDriverWait(driver, 5)
submenu = wait.until(
    EC.visibility_of_element_located((By.ID, "submenu"))
)
submenu.click()

# 13. 模拟绘制（Canvas）
canvas = driver.find_element(By.ID, "canvas")
actions.click_and_hold(canvas)\
    .move_by_offset(50, 50)\
    .move_by_offset(50, -50)\
    .move_by_offset(-50, -50)\
    .move_by_offset(-50, 50)\
    .release()\
    .perform()

# 14. 范围选择（Shift + Click）
first_item = driver.find_element(By.XPATH, "//li[1]")
last_item = driver.find_element(By.XPATH, "//li[10]")
actions.click(first_item)\
    .key_down(Keys.SHIFT)\
    .click(last_item)\
    .key_up(Keys.SHIFT)\
    .perform()

# 15. 重置 ActionChains
actions.reset_actions()  # 清除之前的操作链

driver.quit()
```

**注意事项：**
- 所有 ActionChains 操作必须调用 `.perform()` 才会执行
- 链式调用可以一次执行多个操作
- `key_down()` 必须配合 `key_up()` 使用

## 4. 如何处理键盘操作？常用的 Keys 有哪些?

**答案：**
使用 `send_keys()` 方法结合 `Keys` 类进行键盘操作。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.common.action_chains import ActionChains

driver = webdriver.Chrome()
driver.get("https://www.example.com")

element = driver.find_element(By.ID, "search-input")

# 1. 输入文本
element.send_keys("Selenium 测试")

# 2. 回车键
element.send_keys(Keys.ENTER)

# 3. 清空输入框（Ctrl+A + Delete）
element.send_keys(Keys.CONTROL + "a")  # 全选
element.send_keys(Keys.DELETE)         # 删除

# 4. Tab 键（切换焦点）
element.send_keys(Keys.TAB)

# 5. 退格键
element.send_keys("test")
element.send_keys(Keys.BACKSPACE)  # 删除最后一个字符，结果：tes

# 6. 空格键
element.send_keys(Keys.SPACE)

# 7. 方向键
element.send_keys(Keys.ARROW_DOWN)
element.send_keys(Keys.ARROW_UP)
element.send_keys(Keys.ARROW_LEFT)
element.send_keys(Keys.ARROW_RIGHT)

# 8. 功能键
element.send_keys(Keys.F5)  # 刷新页面

# 9. 组合键 - Ctrl+C（复制）
element.send_keys(Keys.CONTROL + "c")

# 10. 组合键 - Ctrl+V（粘贴）
element.send_keys(Keys.CONTROL + "v")

# 11. 组合键 - Ctrl+A（全选）
element.send_keys(Keys.CONTROL + "a")

# 12. Escape 键（关闭弹窗）
driver.find_element(By.TAG_NAME, "body").send_keys(Keys.ESCAPE)

# 13. Home/End 键
element.send_keys(Keys.HOME)  # 光标移到开头
element.send_keys(Keys.END)   # 光标移到末尾

# 14. Page Up/Down
driver.find_element(By.TAG_NAME, "body").send_keys(Keys.PAGE_DOWN)
driver.find_element(By.TAG_NAME, "body").send_keys(Keys.PAGE_UP)

# 15. 使用 ActionChains 执行组合键（更精确）
actions = ActionChains(driver)
actions.key_down(Keys.CONTROL)\
    .send_keys('a')\
    .key_up(Keys.CONTROL)\
    .perform()

# 16. 表单提交（两种方式）
# 方式1：回车键
element.send_keys(Keys.RETURN)
# 方式2：调用 submit()
form = driver.find_element(By.TAG_NAME, "form")
form.submit()

# 17. 多行输入
textarea = driver.find_element(By.ID, "comment")
textarea.send_keys("第一行", Keys.ENTER, "第二行", Keys.ENTER, "第三行")

# 18. 清空输入框的多种方式
# 方式1：clear() 方法
element.clear()
# 方式2：组合键
element.send_keys(Keys.CONTROL + "a")
element.send_keys(Keys.DELETE)
# 方式3：JavaScript
driver.execute_script("arguments[0].value = '';", element)

# 常用 Keys 枚举总结
"""
Keys.ENTER / Keys.RETURN    # 回车
Keys.TAB                     # Tab键
Keys.SPACE                   # 空格
Keys.BACKSPACE               # 退格
Keys.DELETE                  # 删除
Keys.ESCAPE                  # Esc
Keys.CONTROL / Keys.COMMAND  # Ctrl (Mac: Command)
Keys.SHIFT                   # Shift
Keys.ALT                     # Alt
Keys.ARROW_UP/DOWN/LEFT/RIGHT # 方向键
Keys.HOME / Keys.END         # Home/End
Keys.PAGE_UP / Keys.PAGE_DOWN # Page Up/Down
Keys.F1 ~ Keys.F12           # 功能键
"""

driver.quit()
```

**实际应用场景：**
- 搜索框自动补全：输入文本后按 `ARROW_DOWN` 选择建议项，再按 `ENTER`
- 表单快速填写：使用 `TAB` 键在字段间切换
- 文本编辑：`CTRL+A` 全选、`CTRL+C` 复制、`CTRL+V` 粘贴

## 5. 如何处理文件上传？

**答案：**
文件上传有两种情况：普通 input 上传和复杂上传组件。

```python
import os
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get("https://www.example.com/upload")

# 方法1：直接使用 send_keys 发送文件路径（最常用）
# HTML: <input type="file" id="file-input" />
file_path = "/home/user/document.pdf"
file_input = driver.find_element(By.ID, "file-input")
file_input.send_keys(file_path)

# 方法2：上传多个文件（multiple 属性）
# HTML: <input type="file" id="file-input" multiple />
file_path1 = "/home/user/file1.pdf"
file_path2 = "/home/user/file2.pdf"
file_input = driver.find_element(By.ID, "file-input")
file_input.send_keys(f"{file_path1}\n{file_path2}")  # 使用换行符分隔

# 方法3：使用绝对路径
current_dir = os.path.dirname(os.path.abspath(__file__))
file_path = os.path.join(current_dir, "test_files", "sample.pdf")
file_input = driver.find_element(By.ID, "file-input")
file_input.send_keys(file_path)

# 方法4：处理隐藏的 input 元素
# 有些上传组件会隐藏原生 input，使用 JS 显示它
file_input = driver.find_element(By.ID, "hidden-file-input")
driver.execute_script("arguments[0].style.display = 'block';", file_input)
file_input.send_keys(file_path)

# 方法5：使用 ActionChains 点击上传按钮（如果有触发按钮）
from selenium.webdriver.common.action_chains import ActionChains

# 先上传文件
file_input = driver.find_element(By.ID, "file-input")
file_input.send_keys(file_path)

# 再点击提交按钮
submit_button = driver.find_element(By.ID, "upload-submit")
submit_button.click()

# 等待上传完成
wait = WebDriverWait(driver, 30)
success_message = wait.until(
    EC.visibility_of_element_located((By.CLASS_NAME, "upload-success"))
)
print(f"上传结果: {success_message.text}")

# 方法6：处理拖拽上传（模拟拖拽文件到区域）
# 注意：Selenium 无法直接模拟系统文件拖拽，需要使用 JavaScript
drop_zone = driver.find_element(By.ID, "drop-zone")
file_path = "/home/user/document.pdf"

# 使用 JavaScript 模拟文件拖拽（需要特殊脚本）
js_drop_file = """
    var target = arguments[0];
    var offsetX = arguments[1];
    var offsetY = arguments[2];
    var document = target.ownerDocument || document;
    var input = document.createElement('INPUT');
    input.type = 'file';
    input.onchange = function () {
      var rect = target.getBoundingClientRect();
      var x = rect.left + (offsetX || (rect.width >> 1));
      var y = rect.top + (offsetY || (rect.height >> 1));
      var dataTransfer = { files: this.files };
      ['dragenter', 'dragover', 'drop'].forEach(function (name) {
        var evt = document.createEvent('MouseEvent');
        evt.initMouseEvent(name, !0, !0, document.defaultView, 0, 0, 0, x, y, !1, !1, !1, !1, 0, null);
        evt.dataTransfer = dataTransfer;
        target.dispatchEvent(evt);
      });
      setTimeout(function () { document.body.removeChild(input); }, 25);
    };
    document.body.appendChild(input);
    return input;
"""
file_input = driver.execute_script(js_drop_file, drop_zone, 0, 0)
file_input.send_keys(file_path)

# 方法7：使用 AutoIt 处理 Windows 文件对话框（Windows系统）
# 1. 安装 AutoIt
# 2. 编写 .au3 脚本并编译为 .exe
# 3. 在 Python 中调用
import subprocess

upload_button = driver.find_element(By.ID, "upload-button")
upload_button.click()  # 弹出文件选择对话框

# 调用 AutoIt 脚本处理对话框
subprocess.Popen(["FileUpload.exe", file_path])

# 方法8：验证上传成功
def verify_upload(driver, expected_filename):
    """验证文件上传成功"""
    wait = WebDriverWait(driver, 10)
    uploaded_file = wait.until(
        EC.presence_of_element_located((By.XPATH, f"//span[contains(text(), '{expected_filename}')]"))
    )
    return uploaded_file.is_displayed()

if verify_upload(driver, "document.pdf"):
    print("文件上传成功")

# 封装上传函数
def upload_file(driver, locator, file_path, timeout=30):
    """
    通用文件上传函数
    :param driver: WebDriver实例
    :param locator: 元素定位器 (By.ID, "file-input")
    :param file_path: 文件绝对路径
    :param timeout: 等待超时时间
    """
    if not os.path.isabs(file_path):
        raise ValueError("文件路径必须是绝对路径")

    if not os.path.exists(file_path):
        raise FileNotFoundError(f"文件不存在: {file_path}")

    wait = WebDriverWait(driver, timeout)
    file_input = wait.until(EC.presence_of_element_located(locator))

    # 如果元素不可见，使用 JS 显示
    if not file_input.is_displayed():
        driver.execute_script("arguments[0].style.display = 'block';", file_input)

    file_input.send_keys(file_path)
    print(f"文件已上传: {os.path.basename(file_path)}")

# 使用封装函数
upload_file(driver, (By.ID, "file-input"), "/home/user/document.pdf")

driver.quit()
```

**注意事项：**
1. 文件路径必须是绝对路径
2. Linux/Mac 路径：`/home/user/file.pdf`
3. Windows 路径：`C:\\Users\\user\\file.pdf` 或 `C:/Users/user/file.pdf`
4. 隐藏的 input 元素需要用 JS 显示
5. 复杂上传组件可能需要 AutoIt（Windows）或 pyautogui

## 6. 如何处理文件下载？

**答案：**
文件下载需要配置浏览器的下载行为，避免弹出下载对话框。

```python
import os
import time
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.chrome.options import Options as ChromeOptions

# 设置下载目录
download_dir = "/home/user/downloads"
os.makedirs(download_dir, exist_ok=True)

# 配置 Chrome 下载选项
```

## 7. 如何切换 iframe？

**答案：**
iframe 是独立的 HTML 文档，需要先切换到 iframe 内才能定位其中的元素。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get("https://www.example.com")

# 方法1：通过 iframe 的 id 或 name 切换
driver.switch_to.frame("iframe-id")
# 或
driver.switch_to.frame("iframe-name")

# 方法2：通过 iframe 元素对象切换（推荐）
iframe = driver.find_element(By.ID, "iframe-id")
driver.switch_to.frame(iframe)

# 方法3：通过索引切换（不推荐，不稳定）
driver.switch_to.frame(0)  # 切换到第一个iframe

# 方法4：结合显式等待切换iframe
wait = WebDriverWait(driver, 10)
wait.until(EC.frame_to_be_available_and_switch_to_it((By.ID, "iframe-id")))

# 操作 iframe 内的元素
input_field = driver.find_element(By.ID, "username")
input_field.send_keys("testuser")

# 方法5：切换回主文档（default_content）
driver.switch_to.default_content()

# 方法6：切换到父级 frame（parent_frame）
driver.switch_to.parent_frame()

# 方法7：嵌套 iframe 切换
# HTML结构：
# <iframe id="frame1">
#   <iframe id="frame2">
#     <input id="nested-input" />
#   </iframe>
# </iframe>

# 先切换到外层iframe
driver.switch_to.frame("frame1")
# 再切换到内层iframe
driver.switch_to.frame("frame2")
# 操作元素
driver.find_element(By.ID, "nested-input").send_keys("test")
# 返回主文档
driver.switch_to.default_content()

# 方法8：使用 XPath 定位 iframe
iframe = driver.find_element(By.XPATH, "//iframe[@title='Example']")
driver.switch_to.frame(iframe)

# 方法9：判断元素是否在 iframe 中
def find_element_in_iframes(driver, locator):
    """在所有iframe中查找元素"""
    # 先在主文档中查找
    try:
        return driver.find_element(*locator)
    except:
        pass

    # 遍历所有iframe
    iframes = driver.find_elements(By.TAG_NAME, "iframe")
    for iframe in iframes:
        driver.switch_to.frame(iframe)
        try:
            element = driver.find_element(*locator)
            return element
        except:
            driver.switch_to.default_content()

    driver.switch_to.default_content()
    raise Exception("元素未找到")

# 方法10：封装 iframe 切换上下文管理器
from contextlib import contextmanager

@contextmanager
def iframe_context(driver, iframe_locator):
    """iframe 上下文管理器，自动切换和恢复"""
    iframe = driver.find_element(*iframe_locator)
    driver.switch_to.frame(iframe)
    try:
        yield
    finally:
        driver.switch_to.default_content()

# 使用上下文管理器
with iframe_context(driver, (By.ID, "iframe-id")):
    # 在iframe内操作
    driver.find_element(By.ID, "username").send_keys("testuser")
# 自动切换回主文档

driver.quit()
```

**注意事项：**
1. 切换到 iframe 后，只能操作 iframe 内的元素
2. 操作完 iframe 后，记得切换回主文档
3. 多层嵌套 iframe 需要逐层切换
4. 使用 `frame_to_be_available_and_switch_to_it` 等待 iframe 加载

## 8. 如何处理多窗口/标签页切换？

**答案：**
使用 `window_handles` 获取所有窗口句柄，通过 `switch_to.window()` 切换。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get("https://www.example.com")

# 获取当前窗口句柄
main_window = driver.current_window_handle
print(f"主窗口句柄: {main_window}")

# 方法1：点击链接打开新窗口
driver.find_element(By.LINK_TEXT, "打开新窗口").click()

# 等待新窗口出现
wait = WebDriverWait(driver, 10)
wait.until(EC.number_of_windows_to_be(2))

# 获取所有窗口句柄
all_windows = driver.window_handles
print(f"所有窗口句柄: {all_windows}")

# 切换到新窗口（最后一个）
new_window = [window for window in all_windows if window != main_window][0]
driver.switch_to.window(new_window)
print(f"当前窗口标题: {driver.title}")

# 在新窗口中操作
driver.find_element(By.ID, "new-window-element").click()

# 关闭当前窗口
driver.close()

# 切换回主窗口
driver.switch_to.window(main_window)

# 方法2：使用索引切换窗口
all_windows = driver.window_handles
driver.switch_to.window(all_windows[0])  # 切换到第一个窗口
driver.switch_to.window(all_windows[1])  # 切换到第二个窗口

# 方法3：遍历所有窗口
for window in driver.window_handles:
    driver.switch_to.window(window)
    print(f"窗口标题: {driver.title}")
    print(f"窗口URL: {driver.current_url}")

# 方法4：根据窗口标题切换
def switch_to_window_by_title(driver, title):
    """根据标题切换到指定窗口"""
    for window in driver.window_handles:
        driver.switch_to.window(window)
        if title in driver.title:
            return True
    return False

if switch_to_window_by_title(driver, "新窗口标题"):
    print("成功切换到目标窗口")

# 方法5：根据 URL 切换
def switch_to_window_by_url(driver, url):
    """根据URL切换到指定窗口"""
    for window in driver.window_handles:
        driver.switch_to.window(window)
        if url in driver.current_url:
            return True
    return False

if switch_to_window_by_url(driver, "example.com"):
    print("成功切换到目标窗口")

# 方法6：打开新标签页（Selenium 4+）
driver.switch_to.new_window('tab')  # 打开新标签页
driver.get("https://www.example.com")

# 方法7：打开新窗口（Selenium 4+）
driver.switch_to.new_window('window')  # 打开新窗口
driver.get("https://www.example.com")

# 方法8：关闭所有其他窗口，只保留主窗口
main_window = driver.current_window_handle
for window in driver.window_handles:
    if window != main_window:
        driver.switch_to.window(window)
        driver.close()
driver.switch_to.window(main_window)

# 方法9：封装窗口切换上下文管理器
from contextlib import contextmanager

@contextmanager
def window_context(driver, window_handle):
    """窗口上下文管理器，自动切换和恢复"""
    original_window = driver.current_window_handle
    driver.switch_to.window(window_handle)
    try:
        yield
    finally:
        if original_window in driver.window_handles:
            driver.switch_to.window(original_window)

# 使用上下文管理器
main_window = driver.current_window_handle
driver.find_element(By.LINK_TEXT, "打开新窗口").click()
wait.until(EC.number_of_windows_to_be(2))
new_window = [w for w in driver.window_handles if w != main_window][0]

with window_context(driver, new_window):
    # 在新窗口中操作
    print(f"新窗口标题: {driver.title}")
    driver.find_element(By.ID, "element").click()
# 自动切换回主窗口

# 方法10：等待新窗口并自动切换
def wait_and_switch_to_new_window(driver, timeout=10):
    """等待新窗口出现并自动切换"""
    original_windows = driver.window_handles
    wait = WebDriverWait(driver, timeout)
    wait.until(lambda d: len(d.window_handles) > len(original_windows))

    new_windows = [w for w in driver.window_handles if w not in original_windows]
    if new_windows:
        driver.switch_to.window(new_windows[0])
        return new_windows[0]
    return None

# 使用
driver.find_element(By.LINK_TEXT, "打开新窗口").click()
new_window = wait_and_switch_to_new_window(driver)
if new_window:
    print(f"成功切换到新窗口: {driver.title}")

driver.quit()
```

**注意事项：**
1. 窗口句柄是唯一标识符（字符串），每次打开窗口都会生成新的句柄
2. `close()` 关闭当前窗口，`quit()` 关闭所有窗口并退出 WebDriver
3. 关闭窗口后，焦点不会自动切换，需要手动切换

## 9. 如何处理 JavaScript 弹窗（alert、confirm、prompt）？

**答案：**
使用 `driver.switch_to.alert` 切换到弹窗并进行操作。

```python

from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver.find_element(By.ID, "alert-button").click()

# 等待弹窗出现
wait = WebDriverWait(driver, 10)
alert = wait.until(EC.alert_is_present())

# 获取弹窗文本
alert_text = alert.text

# 接受弹窗（点击"确定"）
alert.accept()

# 方法2：处理 confirm 弹窗
driver.find_element(By.ID, "confirm-button").click()
alert = wait.until(EC.alert_is_present())
print(f"确认弹窗: {alert.text}")

# 接受（点击"确定"）
alert.accept()
# 或取消（点击"取消"）
# alert.dismiss()

# 方法3：处理 prompt 弹窗（需要输入）
driver.find_element(By.ID, "prompt-button").click()
alert = wait.until(EC.alert_is_present())
print(f"输入弹窗: {alert.text}")

# 输入文本
alert.send_keys("用户输入的内容")
# 接受
alert.accept()

# 方法4：使用 try-except 处理弹窗
try:
    alert = driver.switch_to.alert
    print(f"弹窗内容: {alert.text}")
    alert.accept()
except NoAlertPresentException:
    print("没有弹窗")

# 方法5：封装弹窗处理函数
def handle_alert(driver, action="accept", input_text=None, timeout=10):
    """
    处理弹窗
    :param action: accept（接受）或 dismiss（取消）
    :param input_text: prompt 弹窗需要输入的文本
    :param timeout: 等待超时时间
    """
    try:
        wait = WebDriverWait(driver, timeout)
        alert = wait.until(EC.alert_is_present())

        alert_text = alert.text
        print(f"弹窗内容: {alert_text}")

        if input_text:
            alert.send_keys(input_text)

        if action == "accept":
            alert.accept()
        elif action == "dismiss":
            alert.dismiss()

        return alert_text
    except NoAlertPresentException:
        print("没有弹窗")
        return None

# 使用封装函数
driver.find_element(By.ID, "alert-button").click()
handle_alert(driver, action="accept")

driver.find_element(By.ID, "prompt-button").click()
handle_alert(driver, action="accept", input_text="测试输入")

# 方法6：判断弹窗是否存在
def is_alert_present(driver):
    """判断是否有弹窗"""
    try:
        driver.switch_to.alert
        return True
    except NoAlertPresentException:
        return False

if is_alert_present(driver):
    alert = driver.switch_to.alert
    alert.accept()

# 方法7：使用 JavaScript 触发弹窗（测试用）
driver.execute_script("alert('这是一个测试弹窗');")
alert = driver.switch_to.alert
print(alert.text)
alert.accept()

# 方法8：自动处理弹窗（设置监听）
# 注意：Selenium 无法直接监听弹窗，需要在操作前预判
def click_and_handle_alert(driver, element_locator, alert_action="accept"):
    """点击元素并自动处理弹窗"""
    element = driver.find_element(*element_locator)
    element.click()

    wait = WebDriverWait(driver, 2)
    try:
        alert = wait.until(EC.alert_is_present())
        if alert_action == "accept":
            alert.accept()
        else:
            alert.dismiss()
        return True
    except:
        return False

# 使用
click_and_handle_alert(driver, (By.ID, "delete-button"), alert_action="accept")

driver.quit()
```

**弹窗类型总结：**
- **alert**：只有"确定"按钮，使用 `accept()`
- **confirm**：有"确定"和"取消"按钮，使用 `accept()` 或 `dismiss()`
- **prompt**：需要输入文本，使用 `send_keys()` 后 `accept()`

**注意事项：**
1. 必须先切换到弹窗才能操作
2. 弹窗处理后，焦点自动返回主页面
3. 如果弹窗不存在，会抛出 `NoAlertPresentException`
4. 自定义的 Modal 对话框（div 模拟）不是 JavaScript 弹窗，使用常规定位方法

## 10. 如何处理滚动条？

**答案：**
使用 JavaScript 或 ActionChains 控制页面滚动。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.common.keys import Keys
import time

driver = webdriver.Chrome()
driver.get("https://www.example.com")

# 方法1：滚动到页面底部
driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

# 方法2：滚动到页面顶部
driver.execute_script("window.scrollTo(0, 0);")

# 方法3：滚动指定距离
driver.execute_script("window.scrollBy(0, 500);")  # 向下滚动500像素

# 方法4：滚动到指定元素（元素在视口中可见）
element = driver.find_element(By.ID, "footer")
driver.execute_script("arguments[0].scrollIntoView();", element)

# 方法5：滚动到元素并居中显示
element = driver.find_element(By.ID, "target")
driver.execute_script("arguments[0].scrollIntoView({block: 'center'});", element)

# 方法6：使用 ActionChains 滚动（Selenium 4.2+）
actions = ActionChains(driver)
footer = driver.find_element(By.ID, "footer")
actions.scroll_to_element(footer).perform()

# 方法7：滚动指定距离（ActionChains）
actions.scroll_by_amount(0, 500).perform()  # 向下滚动500像素

# 方法8：使用键盘操作滚动
body = driver.find_element(By.TAG_NAME, "body")
body.send_keys(Keys.PAGE_DOWN)  # 向下翻页
body.send_keys(Keys.PAGE_UP)    # 向上翻页
body.send_keys(Keys.END)        # 滚动到底部
body.send_keys(Keys.HOME)       # 滚动到顶部

# 方法9：处理局部滚动条（div 内的滚动条）
scrollable_div = driver.find_element(By.ID, "scrollable-container")
driver.execute_script("arguments[0].scrollTop = arguments[0].scrollHeight;", scrollable_div)

# 方法10：逐步滚动（模拟人工滚动，加载懒加载内容）
def slow_scroll_to_bottom(driver, scroll_pause_time=0.5):
    """逐步滚动到页面底部"""
    last_height = driver.execute_script("return document.body.scrollHeight")

    while True:
        # 滚动到底部
        driver.execute_script("window.scrollTo(0, document.body.scrollHeight);")

        # 等待页面加载
        time.sleep(scroll_pause_time)

        # 计算新的滚动高度
        new_height = driver.execute_script("return document.body.scrollHeight")

        # 如果没有新内容加载，退出循环
        if new_height == last_height:
            break

        last_height = new_height

# 使用
slow_scroll_to_bottom(driver)

# 方法11：滚动加载所有懒加载图片
def load_all_lazy_images(driver):
    """滚动加载所有懒加载图片"""
    images = driver.find_elements(By.TAG_NAME, "img")
    for img in images:
        driver.execute_script("arguments[0].scrollIntoView();", img)
        time.sleep(0.2)  # 等待图片加载

load_all_lazy_images(driver)

# 方法12：获取滚动条位置
scroll_y = driver.execute_script("return window.pageYOffset;")
print(f"当前滚动位置: {scroll_y}")

# 方法13：判断是否滚动到底部
is_at_bottom = driver.execute_script("""
    return (window.innerHeight + window.pageYOffset) >= document.body.scrollHeight;
""")
print(f"是否在页面底部: {is_at_bottom}")

# 方法14：平滑滚动
driver.execute_script("""
    window.scrollTo({
        top: document.body.scrollHeight,
        behavior: 'smooth'
    });
""")

# 方法15：横向滚动
driver.execute_script("window.scrollTo(500, 0);")  # 横向滚动500像素

driver.quit()
```

**常见场景：**
- **懒加载列表**：使用 `slow_scroll_to_bottom()` 逐步加载
- **元素不可见**：使用 `scrollIntoView()` 滚动到元素位置
- **无限滚动**：循环滚动并等待新内容加载
- **截图前**：滚动到元素位置确保元素可见

**注意事项：**
1. `scrollIntoView()` 会将元素滚动到视口顶部或底部
2. 滚动后可能需要等待元素加载完成
3. 某些网站有反爬机制，需要模拟人工滚动速度
