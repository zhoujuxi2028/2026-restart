## 1. 如何在 Selenium 中执行 JavaScript？有哪些应用场景？

## 2. 什么是 Page Object Model（POM）？如何实现？

**答案：**
POM 是一种设计模式，将页面元素和操作封装到页面类中，提高代码可维护性和可复用性。

**核心思想：**
- 每个页面对应一个类
- 页面元素作为类属性
- 页面操作作为类方法
- 测试用例只调用页面对象的方法

```python
# pages/base_page.py - 基础页面类
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

class BasePage:
    """所有页面类的基类"""

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)

    def find_element(self, locator):
        """查找单个元素"""
        return self.wait.until(EC.presence_of_element_located(locator))

    def find_elements(self, locator):
        """查找多个元素"""
        return self.wait.until(EC.presence_of_all_elements_located(locator))

    def click(self, locator):
        """点击元素"""
        element = self.wait.until(EC.element_to_be_clickable(locator))
        element.click()

    def input_text(self, locator, text):
        """输入文本"""
        element = self.find_element(locator)
        element.clear()
        element.send_keys(text)

    def get_text(self, locator):
        """获取元素文本"""
        return self.find_element(locator).text

    def is_element_visible(self, locator):
        """判断元素是否可见"""
        try:
            self.wait.until(EC.visibility_of_element_located(locator))
            return True
        except:
            return False


# pages/login_page.py - 登录页面类
from selenium.webdriver.common.by import By
from pages.base_page import BasePage

class LoginPage(BasePage):
    """登录页面"""

    # 页面元素定位器（推荐使用元组格式）
    USERNAME_INPUT = (By.ID, "username")
    PASSWORD_INPUT = (By.ID, "password")
    LOGIN_BUTTON = (By.ID, "login-button")
    ERROR_MESSAGE = (By.CLASS_NAME, "error-message")

    def __init__(self, driver):
        super().__init__(driver)
        self.url = "https://www.example.com/login"

    def open(self):
        """打开登录页面"""
        self.driver.get(self.url)
        return self

    def login(self, username, password):
        """执行登录操作"""
        self.input_text(self.USERNAME_INPUT, username)
        self.input_text(self.PASSWORD_INPUT, password)
        self.click(self.LOGIN_BUTTON)
        return self

    def get_error_message(self):
        """获取错误提示"""
        return self.get_text(self.ERROR_MESSAGE)

    def is_login_button_visible(self):
        """判断登录按钮是否可见"""
        return self.is_element_visible(self.LOGIN_BUTTON)

# 进阶：使用 Page Factory 模式（类似 Java）
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

class LoginPageFactory:
    """使用属性装饰器的 POM"""

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)

    @property
    def username_input(self):
        return self.wait.until(
            EC.presence_of_element_located((By.ID, "username"))
        )

    @property
    def password_input(self):
        return self.wait.until(
            EC.presence_of_element_located((By.ID, "password"))
        )

    @property
    def login_button(self):
        return self.wait.until(
            EC.element_to_be_clickable((By.ID, "login-button"))
        )

    def login(self, username, password):
        self.username_input.send_keys(username)
        self.password_input.send_keys(password)
        self.login_button.click()
```

**POM 优点：**
- 代码复用：多个测试用例共享页面对象
- 易于维护：页面元素变化只需修改页面类
- 可读性强：测试用例清晰表达业务逻辑
- 降低耦合：测试用例不直接操作 WebDriver

**最佳实践：**
1. 每个页面一个类
2. 定位器作为类常量
3. 页面操作返回 `self` 或下一个页面对象（方法链）
4. 使用基础页面类封装通用方法
5. 测试用例只调用页面对象方法，不直接操作元素

**答案：**
数据驱动测试是指将测试数据与测试逻辑分离，使用相同的测试逻辑处理不同的测试数据。

## 4. 什么是 Selenium Grid？如何实现并发测试？

**答案：**
Selenium Grid 是一个分布式测试框架，支持在多台机器上并行运行测试，提高测试效率。

**架构：**
- **Hub（中心节点）**：接收测试请求，分配到各个 Node
- **Node（执行节点）**：实际执行测试的机器

```bash
# 启动 Selenium Grid 4（Standalone 模式）
java -jar selenium-server-4.x.jar standalone

# 启动 Hub（分布式模式）
java -jar selenium-server-4.x.jar hub

# 启动 Node（连接到 Hub）
java -jar selenium-server-4.x.jar node --hub http://hub-ip:4444
```

```python
# 方法4：使用多线程并发
import threading
from selenium import webdriver

def run_test(browser_name):
    """单个测试任务"""
    driver = webdriver.Remote(
        command_executor='http://localhost:4444/wd/hub',
        desired_capabilities={'browserName': browser_name}
    )
    driver.get("https://www.example.com")
    print(f"{browser_name}: {driver.title}")
    driver.quit()

# 创建线程并发执行
threads = []
browsers = ['chrome', 'firefox', 'edge']

for browser in browsers:
    thread = threading.Thread(target=run_test, args=(browser,))
    threads.append(thread)
    thread.start()

for thread in threads:
    thread.join()

class GridTestRunner:
    """Grid 并发测试运行器"""

    def __init__(self, hub_url, max_workers=5):
        self.hub_url = hub_url
        self.max_workers = max_workers

    def run_tests(self, test_cases):
        """并发执行多个测试用例"""
        with ThreadPoolExecutor(max_workers=self.max_workers) as executor:
            futures = [executor.submit(self._run_single_test, test) for test in test_cases]

            for future in futures:
                try:
                    result = future.result()
                    print(f"测试结果: {result}")
                except Exception as e:
                    print(f"测试失败: {e}")

    def _run_single_test(self, test_case):
        """执行单个测试"""
        driver = webdriver.Remote(
            command_executor=self.hub_url,
            desired_capabilities={'browserName': 'chrome'}
        )
        try:
            # 执行测试逻辑
            driver.get(test_case['url'])
            return {'test': test_case['name'], 'status': 'PASS'}
        except Exception as e:
            return {'test': test_case['name'], 'status': 'FAIL', 'error': str(e)}
        finally:
            driver.quit()

# 使用
runner = GridTestRunner('http://localhost:4444/wd/hub', max_workers=10)
test_cases = [
    {'name': 'test1', 'url': 'https://www.example.com/1'},
    {'name': 'test2', 'url': 'https://www.example.com/2'},
]
runner.run_tests(test_cases)
```

**Selenium Grid 优点：**
- 并发执行：缩短测试时间
- 跨平台：支持不同操作系统
- 跨浏览器：同时测试多个浏览器
- 可扩展：动态添加/删除节点

**注意事项：**
1. 并发测试需要确保测试数据独立
2. 共享资源（如数据库）需要加锁
3. Grid Hub 需要足够的内存和 CPU

```python
        try:
            return func()
        except StaleElementReferenceException:
            if attempt == max_attempts - 1:

        """安全查找元素"""
        try:
            wait = WebDriverWait(self.driver, timeout)
            return wait.until(EC.presence_of_element_located(locator))
        except TimeoutException:
            print(f"元素未找到: {locator}")
            return None

    def safe_click(self, locator, timeout=10):
        """安全点击元素"""
        try:
            element = self.wait.until(EC.element_to_be_clickable(locator))
            element.click()
            return True
        except ElementClickInterceptedException:
            # 使用 JS 点击
            element = self.driver.find_element(*locator)
            self.driver.execute_script("arguments[0].click();", element)
            return True
        except TimeoutException:
            print(f"元素不可点击: {locator}")
            return False

    def safe_input(self, locator, text, timeout=10):
        """安全输入文本"""
        try:
            element = self.wait.until(EC.visibility_of_element_located(locator))
            element.clear()
            element.send_keys(text)
            return True
        except (TimeoutException, ElementNotInteractableException):
            print(f"无法输入文本: {locator}")
            return False

    def safe_get_text(self, locator, timeout=10):
        """安全获取文本"""
        try:
            element = self.wait.until(EC.visibility_of_element_located(locator))
            return element.text
        except TimeoutException:
            print(f"无法获取文本: {locator}")
            return ""

# 8. 通用异常处理装饰器
import functools

def handle_exceptions(func):
    """异常处理装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except NoSuchElementException:
            print(f"{func.__name__}: 元素未找到")
        except TimeoutException:
            print(f"{func.__name__}: 等待超时")
        except StaleElementReferenceException:
            print(f"{func.__name__}: 元素过期，重试")
            time.sleep(0.5)
            return func(*args, **kwargs)  # 重试一次
        except Exception as e:
            print(f"{func.__name__}: 未知异常 - {e}")
        return None
    return wrapper

@handle_exceptions
def find_and_click(driver, locator):
    element = driver.find_element(*locator)
    element.click()

driver.quit()
```

**异常处理最佳实践：**
1. 使用显式等待避免 `NoSuchElementException`
2. 遇到 `StaleElementReferenceException` 重新定位元素
3. 元素被遮挡时使用 JavaScript 点击
4. 封装异常处理函数提高代码可读性
5. 记录异常日志便于调试

## 10. 实战：如何处理一个复杂的登录场景（验证码、滑块、二次验证）？

**答案：**
综合运用多种技术处理复杂登录场景。

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.action_chains import ActionChains
import time

class ComplexLoginHandler:
    """复杂登录场景处理器"""

    def __init__(self, driver):
        self.driver = driver
        self.wait = WebDriverWait(driver, 10)

    def login_with_image_captcha(self, username, password):
        """处理图片验证码登录"""
        # 输入用户名密码
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)

        # 获取验证码图片
        captcha_img = self.driver.find_element(By.ID, "captcha-image")
        captcha_img.screenshot("captcha.png")

        # 方法1：手动输入（测试环境）
        captcha_code = input("请输入验证码: ")

        # 方法2：使用 OCR 识别（需要 pytesseract）
        # import pytesseract
        # from PIL import Image
        # captcha_code = pytesseract.image_to_string(Image.open("captcha.png"))

        # 方法3：使用第三方打码平台（如 2Captcha）
        # captcha_code = solve_captcha_via_api("captcha.png")

        self.driver.find_element(By.ID, "captcha").send_keys(captcha_code)
        self.driver.find_element(By.ID, "login-button").click()

    def login_with_slider_captcha(self, username, password):
        """处理滑块验证码"""
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)
        self.driver.find_element(By.ID, "login-button").click()

        # 等待滑块出现
        slider = self.wait.until(
            EC.presence_of_element_located((By.CLASS_NAME, "slider-button"))
        )

        # 方法1：简单拖拽（可能被识别为机器人）
        actions = ActionChains(self.driver)
        actions.click_and_hold(slider).move_by_offset(300, 0).release().perform()

        # 方法2：模拟人类滑动（更真实）
        self._drag_slider_human_like(slider, 300)

        # 方法3：使用 JavaScript 绕过（某些场景有效）
        self.driver.execute_script("""
            arguments[0].style.left = '300px';
        """, slider)

    def _drag_slider_human_like(self, element, distance):
        """模拟人类滑动滑块"""
        actions = ActionChains(self.driver)
        actions.click_and_hold(element).perform()

        # 分段移动，模拟人类不均匀速度
        moved = 0
        while moved < distance:
            move_distance = min(10 + (distance - moved) // 10, distance - moved)
            actions.move_by_offset(move_distance, 0).perform()
            moved += move_distance
            time.sleep(0.01 + (0.05 * (1 - moved / distance)))  # 速度逐渐变慢

        actions.release().perform()

    def login_with_sms_code(self, username, password):
        """处理短信验证码登录"""
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)

        # 点击发送验证码
        self.driver.find_element(By.ID, "send-sms").click()

        # 方法1：手动输入
        sms_code = input("请输入短信验证码: ")

        # 方法2：从测试环境获取（如果有 API）
        # sms_code = self._get_sms_code_from_test_api(username)

        # 方法3：使用固定验证码（测试环境）
        # sms_code = "123456"

        self.driver.find_element(By.ID, "sms-code").send_keys(sms_code)
        self.driver.find_element(By.ID, "login-button").click()

    def login_with_2fa(self, username, password):
        """处理双因素认证（2FA）"""
        # 第一步：用户名密码登录
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)
        self.driver.find_element(By.ID, "login-button").click()

        # 第二步：输入 TOTP 验证码
        # 方法1：使用 pyotp 生成 TOTP
        import pyotp
        totp_secret = "YOUR_TOTP_SECRET"
        totp = pyotp.TOTP(totp_secret)
        verification_code = totp.now()

        # 等待 2FA 页面出现
        self.wait.until(
            EC.presence_of_element_located((By.ID, "verification-code"))
        )

        self.driver.find_element(By.ID, "verification-code").send_keys(verification_code)
        self.driver.find_element(By.ID, "verify-button").click()

    def login_with_google_recaptcha(self, username, password):
        """处理 Google reCAPTCHA"""
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)

        # 方法1：切换到 reCAPTCHA iframe
        recaptcha_frame = self.wait.until(
            EC.presence_of_element_located((By.CSS_SELECTOR, "iframe[src*='recaptcha']"))
        )
        self.driver.switch_to.frame(recaptcha_frame)

        # 点击 reCAPTCHA 复选框
        checkbox = self.wait.until(
            EC.element_to_be_clickable((By.CLASS_NAME, "recaptcha-checkbox"))
        )
        checkbox.click()

        self.driver.switch_to.default_content()

        # 方法2：使用 2Captcha API 自动解决
        # 需要付费服务
        # site_key = self._get_recaptcha_site_key()
        # solution = solve_recaptcha(site_key, self.driver.current_url)
        # self.driver.execute_script(f"document.getElementById('g-recaptcha-response').innerHTML='{solution}';")

        self.driver.find_element(By.ID, "login-button").click()

    def login_with_cookies(self, username, password):
        """使用 cookies 绕过登录（适合已登录的场景）"""
        # 第一次登录
        self.driver.get("https://www.example.com/login")
        self.driver.find_element(By.ID, "username").send_keys(username)
        self.driver.find_element(By.ID, "password").send_keys(password)
        self.driver.find_element(By.ID, "login-button").click()

        # 保存 cookies
        import pickle
        with open("cookies.pkl", "wb") as f:
            pickle.dump(self.driver.get_cookies(), f)

        # 后续直接加载 cookies
        self.driver.get("https://www.example.com")
        with open("cookies.pkl", "rb") as f:
            cookies = pickle.load(f)
            for cookie in cookies:
                self.driver.add_cookie(cookie)

        self.driver.refresh()

    def comprehensive_login(self, username, password):
        """综合登录流程（处理多种可能性）"""
        try:
            # 打开登录页面
            self.driver.get("https://www.example.com/login")

            # 输入用户名密码
            self.driver.find_element(By.ID, "username").send_keys(username)
            self.driver.find_element(By.ID, "password").send_keys(password)

            # 检查是否有验证码
            if self._is_element_present(By.ID, "captcha-image"):
                print("检测到图片验证码")
                self.login_with_image_captcha(username, password)
                return

            if self._is_element_present(By.CLASS_NAME, "slider-captcha"):
                print("检测到滑块验证码")
                self.login_with_slider_captcha(username, password)
                return

            # 点击登录
            self.driver.find_element(By.ID, "login-button").click()

            # 检查是否需要短信验证
            if self._is_element_present(By.ID, "sms-code"):
                print("需要短信验证")
                sms_code = input("请输入短信验证码: ")
                self.driver.find_element(By.ID, "sms-code").send_keys(sms_code)
                self.driver.find_element(By.ID, "verify-button").click()

            # 检查是否需要 2FA
            if self._is_element_present(By.ID, "verification-code"):
                print("需要双因素认证")
                self.login_with_2fa(username, password)

            # 验证登录成功
            self.wait.until(EC.url_contains("/dashboard"))
            print("登录成功")

        except Exception as e:
            print(f"登录失败: {e}")
            self.driver.save_screenshot("login_error.png")

    def _is_element_present(self, by, value):
        """检查元素是否存在"""
        try:
            self.driver.find_element(by, value)
            return True
        except:
            return False
```

3. **短信验证码**：测试环境使用固定码或 API 获取
4. **双因素认证**：使用 pyotp 生成 TOTP
5. **reCAPTCHA**：手动点击或第三方服务
6. **使用 cookies**：避免重复登录，提高效率
7. **综合处理**：检测各种可能的验证方式，动态处理
